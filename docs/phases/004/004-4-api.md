## API Handlers

```ts
// apps/data-service/src/hono/handlers/signals-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  AddToWatchlistRequestSchema,
  RemoveFromWatchlistRequestSchema,
  GetSignalsRequestSchema,
} from "@repo/data-ops/zod-schema/signals"
import * as watchlistQueries from "@repo/data-ops/queries/watchlist"
import * as signalQueries from "@repo/data-ops/queries/signals"
import { ingestStockTwitsForUser } from "@repo/data-ops/services/signal-aggregator"

const signalsRouter = new Hono<{ Bindings: Env }>()

// Watchlist endpoints
signalsRouter.get("/watchlist", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const watchlist = await watchlistQueries.getUserWatchlist(c.get("db"), userId)
  return c.json({ data: watchlist })
})

signalsRouter.post(
  "/watchlist",
  apiTokenMiddleware,
  zValidator("json", AddToWatchlistRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { symbol } = c.req.valid("json")

    // Get Alpaca client for validation (uses user's credentials)
    const alpacaClient = await getAlpacaClientForUser(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)

    const result = await watchlistQueries.addToWatchlist(c.get("db"), userId, symbol, alpacaClient)

    if ("error" in result) {
      return c.json({ error: result.error }, 400)
    }

    return c.json({ data: result }, 201)
  }
)

signalsRouter.delete(
  "/watchlist/:symbol",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const symbol = c.req.param("symbol")
    const deleted = await watchlistQueries.removeFromWatchlist(c.get("db"), userId, symbol)

    if (!deleted) {
      return c.json({ error: "Symbol not in watchlist" }, 404)
    }

    return c.json({ success: true })
  }
)

// Signal endpoints
signalsRouter.get(
  "/signals",
  apiTokenMiddleware,
  zValidator("query", GetSignalsRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const params = c.req.valid("query")
    const signals = await signalQueries.getSignals(c.get("db"), { userId, ...params })
    return c.json({ data: signals })
  }
)

signalsRouter.get("/signals/feed", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const signals = await signalQueries.getSignalsForWatchlist(c.get("db"), userId, 50)
  return c.json({ data: signals })
})

signalsRouter.get("/signals/stats/:symbol", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const symbol = c.req.param("symbol").toUpperCase()
  const stats = await signalQueries.getSignalStats(c.get("db"), userId, symbol)
  return c.json({ data: stats })
})

// Manual trigger for testing
signalsRouter.post("/signals/ingest", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const result = await ingestStockTwitsForUser({ db: c.get("db"), userId })
  return c.json({ data: result })
})

export { signalsRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/watchlist` | Bearer | Get user's watchlist |
| POST | `/api/watchlist` | Bearer | Add symbol to watchlist |
| DELETE | `/api/watchlist/:symbol` | Bearer | Remove symbol |
| GET | `/api/signals` | Bearer | Get signals (optional filters) |
| GET | `/api/signals/feed` | Bearer | Get signals for watchlist symbols |
| GET | `/api/signals/stats/:symbol` | Bearer | Get sentiment stats for symbol |
| POST | `/api/signals/ingest` | Bearer | Manual ingest trigger (testing) |

## Server Functions

```ts
// apps/user-application/src/core/functions/signals/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import * as watchlistQueries from "@repo/data-ops/queries/watchlist"
import * as signalQueries from "@repo/data-ops/queries/signals"
import { AddToWatchlistRequestSchema, GetSignalsRequestSchema } from "@repo/data-ops/zod-schema/signals"

export const getWatchlist = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return watchlistQueries.getUserWatchlist(context.db, context.session.user.id)
  })

export const addToWatchlist = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(AddToWatchlistRequestSchema)
  .handler(async ({ data, context }) => {
    return watchlistQueries.addToWatchlist(context.db, context.session.user.id, data.symbol)
  })

export const removeFromWatchlist = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ symbol: z.string() }))
  .handler(async ({ data, context }) => {
    return watchlistQueries.removeFromWatchlist(context.db, context.session.user.id, data.symbol)
  })

export const getSignalFeed = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return signalQueries.getSignalsForWatchlist(context.db, context.session.user.id, 50)
  })

export const getSignals = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(GetSignalsRequestSchema)
  .handler(async ({ data, context }) => {
    return signalQueries.getSignals(context.db, {
      userId: context.session.user.id,
      ...data,
    })
  })
```
