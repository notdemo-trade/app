### Watchlist Manager

```tsx
// apps/user-application/src/components/signals/watchlist-manager.tsx

interface WatchlistManagerProps {
  watchlist: WatchlistItem[]
  onAdd: (symbol: string) => void
  onRemove: (symbol: string) => void
}

export function WatchlistManager({ watchlist, onAdd, onRemove }: WatchlistManagerProps) {
  const [input, setInput] = useState("")

  const handleSubmit = (e: FormEvent) => {
    e.preventDefault()
    if (input.trim()) {
      onAdd(input.trim().toUpperCase())
      setInput("")
    }
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Watchlist ({watchlist.length})</CardTitle>
      </CardHeader>
      <CardContent>
        <form onSubmit={handleSubmit} className="flex gap-2 mb-4">
          <Input
            value={input}
            onChange={e => setInput(e.target.value)}
            placeholder="Add symbol..."
            maxLength={10}
          />
          <Button type="submit" size="sm">Add</Button>
        </form>

        {watchlist.length === 0 ? (
          <p className="text-muted-foreground text-sm">
            No symbols yet. Add stocks to track signals.
          </p>
        ) : (
          <div className="flex flex-wrap gap-2">
            {watchlist.map(item => (
              <Badge key={item.id} variant="secondary" className="gap-1">
                {item.symbol}
                <button
                  onClick={() => onRemove(item.symbol)}
                  className="ml-1 hover:text-destructive"
                >
                  ×
                </button>
              </Badge>
            ))}
          </div>
        )}
      </CardContent>
    </Card>
  )
}
```

### Signal Feed

```tsx
// apps/user-application/src/components/signals/signal-feed.tsx

interface SignalFeedProps {
  signals: Signal[]
  isLoading?: boolean
}

export function SignalFeed({ signals, isLoading }: SignalFeedProps) {
  if (isLoading) {
    return <div className="animate-pulse">Loading signals...</div>
  }

  if (signals.length === 0) {
    return (
      <Card>
        <CardContent className="py-8 text-center">
          <p className="text-muted-foreground">
            No signals yet. Add symbols to your watchlist to start tracking.
          </p>
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Signal Feed</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-3">
          {signals.map(signal => (
            <SignalCard key={signal.id} signal={signal} />
          ))}
        </div>
      </CardContent>
    </Card>
  )
}

function SignalCard({ signal }: { signal: Signal }) {
  const directionColors = {
    bullish: "text-green-600 bg-green-50",
    bearish: "text-red-600 bg-red-50",
    neutral: "text-gray-600 bg-gray-50",
  }

  return (
    <div className="flex items-start gap-3 p-3 rounded-lg border">
      <Badge variant="outline" className="font-mono">
        {signal.symbol}
      </Badge>

      <div className="flex-1 min-w-0">
        <div className="flex items-center gap-2">
          <span className={cn("text-xs px-2 py-0.5 rounded", directionColors[signal.direction])}>
            {signal.direction}
          </span>
          <span className="text-xs text-muted-foreground">
            {formatRelativeTime(signal.createdAt)}
          </span>
        </div>

        {signal.summary && (
          <p className="text-sm mt-1 text-muted-foreground line-clamp-2">
            {signal.summary}
          </p>
        )}
      </div>

      <div className="text-right">
        <span className="text-xs text-muted-foreground">Strength</span>
        <div className="font-mono text-sm">
          {(signal.strength * 100).toFixed(0)}%
        </div>
      </div>
    </div>
  )
}
```

### Signals Page

```tsx
// apps/user-application/src/routes/_auth/signals/index.tsx

export const Route = createFileRoute("/_auth/signals/")({
  component: SignalsPage,
  loader: async ({ context }) => {
    const [watchlist, signals] = await Promise.all([
      getWatchlist(),
      getSignalFeed(),
    ])
    return { watchlist, signals }
  },
})

function SignalsPage() {
  const { watchlist: initialWatchlist, signals: initialSignals } = Route.useLoaderData()

  const { data: watchlist, refetch: refetchWatchlist } = useWatchlist(initialWatchlist)
  const { data: signals, refetch: refetchSignals } = useSignalFeed(initialSignals)

  const addMutation = useAddToWatchlist()
  const removeMutation = useRemoveFromWatchlist()

  const handleAdd = async (symbol: string) => {
    await addMutation.mutateAsync({ symbol })
    refetchWatchlist()
  }

  const handleRemove = async (symbol: string) => {
    await removeMutation.mutateAsync({ symbol })
    refetchWatchlist()
  }

  return (
    <div className="space-y-6">
      <h1 className="text-2xl font-bold">Signals</h1>

      <div className="grid md:grid-cols-3 gap-6">
        <div className="md:col-span-1">
          <WatchlistManager
            watchlist={watchlist}
            onAdd={handleAdd}
            onRemove={handleRemove}
          />
        </div>

        <div className="md:col-span-2">
          <SignalFeed signals={signals} />
        </div>
      </div>
    </div>
  )
}
```

## Query Hooks

```ts
// apps/user-application/src/lib/signal-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  getWatchlist,
  addToWatchlist,
  removeFromWatchlist,
  getSignalFeed,
  getSignals,
} from "@/core/functions/signals/direct"

export const watchlistQueryOptions = () =>
  queryOptions({
    queryKey: ["watchlist"],
    queryFn: () => getWatchlist(),
  })

export const signalFeedQueryOptions = () =>
  queryOptions({
    queryKey: ["signals", "feed"],
    queryFn: () => getSignalFeed(),
    staleTime: 60_000, // 1 min
    refetchInterval: 60_000,
  })

export const signalsQueryOptions = (params?: { symbol?: string; limit?: number }) =>
  queryOptions({
    queryKey: ["signals", params],
    queryFn: () => getSignals(params ?? {}),
    staleTime: 60_000,
  })

export const useWatchlist = (initial?: WatchlistItem[]) =>
  useQuery({ ...watchlistQueryOptions(), initialData: initial })

export const useSignalFeed = (initial?: Signal[]) =>
  useQuery({ ...signalFeedQueryOptions(), initialData: initial })

export const useSignals = (params?: { symbol?: string; limit?: number }) =>
  useQuery(signalsQueryOptions(params))

export const useAddToWatchlist = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: (data: { symbol: string }) => addToWatchlist(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["watchlist"] })
    },
  })
}

export const useRemoveFromWatchlist = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: (data: { symbol: string }) => removeFromWatchlist(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["watchlist"] })
    },
  })
}
```
