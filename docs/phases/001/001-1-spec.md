## Environment Setup

### Required Environment Variables

All environments (dev/staging/production) require these secrets. Store in environment-specific files and sync to Cloudflare Workers.

**File locations**:
- `apps/data-service/.{env}.vars` (e.g., `.staging.vars`, `.production.vars`)
- `apps/user-application/.env.{env}` (e.g., `.env.staging`, `.env.production`)

**Required secrets**:

```bash
# Database (from Neon dashboard connection details)
DATABASE_HOST="ep-xxx-xxx.us-east-2.aws.neon.tech"
DATABASE_USERNAME="your_username"
DATABASE_PASSWORD="your_password"

# Auth
BETTER_AUTH_SECRET="<generate-with: openssl rand -base64 32>"
BETTER_AUTH_URL="https://staging.notdemo.trade"  # or production URL

# Environment identifier
CLOUDFLARE_ENV="staging"  # or "production"
```

**Setup workflow**:

1. Create environment files from examples:
   ```bash
   cp apps/data-service/.staging.vars.example apps/data-service/.staging.vars
   cp apps/user-application/.env.staging.example apps/user-application/.env.staging
   ```

2. Generate secrets:
   ```bash
   openssl rand -base64 32  # Use for BETTER_AUTH_SECRET
   ```

3. Get database credentials from Neon dashboard

4. Edit files with real values

5. Sync to Cloudflare Workers:
   ```bash
   cd apps/data-service
   ./sync-secrets.sh staging

   cd ../user-application
   ./sync-secrets.sh staging
   ```

6. Verify secrets in Cloudflare dashboard:
   - Workers & Pages → notdemo-ds-staging → Settings → Variables
   - Workers & Pages → notdemo-ua-staging → Settings → Variables

**Additional secrets (added in later phases)**:
- `CREDENTIALS_ENCRYPTION_KEY` (Phase 2)
- `APPROVAL_SECRET` (Phase 10)
- `TELEGRAM_WEBHOOK_BASE_URL` (Phase 11)
- `SENTRY_DSN` (Phase 18)

### 1.5 Per-User Bearer Token

#### Database Schema

Add to `packages/data-ops/src/drizzle/schema.ts`:

```ts
import { pgTable, text, timestamp, boolean, uuid } from "drizzle-orm/pg-core"
import { auth_user } from "./auth-schema"

export const tokenTypeEnum = pgEnum("token_type", ["access", "kill_switch"])

export const api_tokens = pgTable("api_tokens", {
  id: uuid("id").defaultRandom().primaryKey(),
  userId: text("user_id")
    .notNull()
    .references(() => auth_user.id, { onDelete: "cascade" }),
  type: tokenTypeEnum("type").notNull(), // "access" or "kill_switch"
  tokenHash: text("token_hash").notNull(), // SHA-256 hash
  tokenPrefix: text("token_prefix").notNull(), // first 8 chars for display
  lastUsedAt: timestamp("last_used_at"),
  expiresAt: timestamp("expires_at").notNull(), // 1 month default
  createdAt: timestamp("created_at").defaultNow().notNull(),
  revokedAt: timestamp("revoked_at"),
}, (table) => ({
  // One active token per type per user
  uniqueActiveToken: unique().on(table.userId, table.type).where(isNull(table.revokedAt)),
}))
```

**Why hash tokens**: Store SHA-256 hash, not plaintext. User sees token once on creation. Like GitHub personal access tokens.

#### Token Generation

Add `packages/data-ops/src/queries/api-token.ts`:

```ts
import { eq, and, isNull } from "drizzle-orm"
import { api_tokens, TokenType } from "@/drizzle/schema"
import { getDb } from "@/database/setup"
import crypto from "crypto"

const TOKEN_EXPIRATION_DAYS = 30 // 1 month

interface CreateTokenResult {
  id: string
  token: string // plaintext, shown once
  tokenPrefix: string
  type: TokenType
  expiresAt: Date
  createdAt: Date
}

export async function createApiToken(
  userId: string,
  type: TokenType // "access" | "kill_switch"
): Promise<CreateTokenResult> {
  const db = getDb()

  // Revoke existing token of same type (only one active per type)
  await db
    .update(api_tokens)
    .set({ revokedAt: new Date() })
    .where(
      and(
        eq(api_tokens.userId, userId),
        eq(api_tokens.type, type),
        isNull(api_tokens.revokedAt)
      )
    )

  // Generate 32-byte random token, base64url encoded
  const tokenBytes = crypto.getRandomValues(new Uint8Array(32))
  const token = btoa(String.fromCharCode(...tokenBytes))
    .replace(/\+/g, "-")
    .replace(/\//g, "_")
    .replace(/=+$/, "")

  const tokenHash = await hashToken(token)
  const tokenPrefix = token.slice(0, 8)
  const expiresAt = new Date(Date.now() + TOKEN_EXPIRATION_DAYS * 24 * 60 * 60 * 1000)

  const [result] = await db
    .insert(api_tokens)
    .values({
      userId,
      type,
      tokenHash,
      tokenPrefix,
      expiresAt,
    })
    .returning()

  return {
    id: result.id,
    token, // only time plaintext is available
    tokenPrefix,
    type: result.type,
    expiresAt: result.expiresAt,
    createdAt: result.createdAt,
  }
}

interface ValidateTokenResult {
  userId: string
  type: TokenType
}

export async function validateApiToken(token: string): Promise<ValidateTokenResult | null> {
  const db = getDb()
  const tokenHash = await hashToken(token)

  const [result] = await db
    .select({
      userId: api_tokens.userId,
      id: api_tokens.id,
      type: api_tokens.type,
      expiresAt: api_tokens.expiresAt,
    })
    .from(api_tokens)
    .where(
      and(
        eq(api_tokens.tokenHash, tokenHash),
        isNull(api_tokens.revokedAt)
      )
    )
    .limit(1)

  if (!result) return null

  // Check expiration
  if (result.expiresAt < new Date()) return null

  // Update lastUsedAt (fire and forget)
  db.update(api_tokens)
    .set({ lastUsedAt: new Date() })
    .where(eq(api_tokens.id, result.id))
    .execute()

  return { userId: result.userId, type: result.type }
}

export async function revokeApiToken(id: string, userId: string): Promise<boolean> {
  const db = getDb()

  const result = await db
    .update(api_tokens)
    .set({ revokedAt: new Date() })
    .where(
      and(
        eq(api_tokens.id, id),
        eq(api_tokens.userId, userId),
        isNull(api_tokens.revokedAt)
      )
    )
    .returning()

  return result.length > 0
}

export async function listUserTokens(userId: string) {
  const db = getDb()

  return db
    .select({
      id: api_tokens.id,
      type: api_tokens.type,
      tokenPrefix: api_tokens.tokenPrefix,
      expiresAt: api_tokens.expiresAt,
      lastUsedAt: api_tokens.lastUsedAt,
      createdAt: api_tokens.createdAt,
    })
    .from(api_tokens)
    .where(
      and(
        eq(api_tokens.userId, userId),
        isNull(api_tokens.revokedAt)
      )
    )
    .orderBy(api_tokens.type)
}

async function hashToken(token: string): Promise<string> {
  const encoder = new TextEncoder()
  const data = encoder.encode(token)
  const hashBuffer = await crypto.subtle.digest("SHA-256", data)
  const hashArray = Array.from(new Uint8Array(hashBuffer))
  return hashArray.map(b => b.toString(16).padStart(2, "0")).join("")
}
```

  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ type: TokenTypeSchema }))
  .handler(async ({ data, context }) => {

  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ type: TokenTypeSchema }))
  .handler(async ({ data, context }) => {
    return tokenQueries.revokeApiTokenByType(context.session.user.id, data.type)
  })

### 1.7 Auth Middleware for API

Replace static token auth with per-user token validation.

Update `apps/data-service/src/hono/middleware/auth.ts`:

```ts
import { createMiddleware } from "hono/factory"
import { HTTPException } from "hono/http-exception"
import { validateApiToken } from "@repo/data-ops/queries/api-token"
import type { TokenType } from "@repo/data-ops/zod-schema/api-token"

// General API token middleware (accepts any valid token)
export const apiTokenMiddleware = createMiddleware<{ Bindings: Env }>(
  async (c, next) => {
    const authHeader = c.req.header("Authorization")

    if (!authHeader?.startsWith("Bearer ")) {
      throw new HTTPException(401, {
        message: "Authorization header required",
      })
    }

    const token = authHeader.slice(7)
    const result = await validateApiToken(token)

    if (!result) {
      throw new HTTPException(401, {
        message: "Invalid, expired, or revoked token",
      })
    }

    c.set("userId", result.userId)
    c.set("tokenType", result.type)
    await next()
  }
)

// Kill switch middleware - only accepts kill_switch tokens
export const killSwitchTokenMiddleware = createMiddleware<{ Bindings: Env }>(
  async (c, next) => {
    const authHeader = c.req.header("Authorization")

    if (!authHeader?.startsWith("Bearer ")) {
      throw new HTTPException(401, { message: "Authorization required" })
    }

    const token = authHeader.slice(7)
    const result = await validateApiToken(token)

    if (!result || result.type !== "kill_switch") {
      throw new HTTPException(403, {
        message: "Kill switch token required",
      })
    }

    c.set("userId", result.userId)
    c.set("tokenType", result.type)
    await next()
  }
)
```

**Backwards compatibility**: Keep `API_TOKEN` env var for internal service-to-service calls. Check both:

```ts
export const apiTokenMiddleware = createMiddleware<{ Bindings: Env }>(
  async (c, next) => {
    const authHeader = c.req.header("Authorization")

    if (!authHeader?.startsWith("Bearer ")) {
      throw new HTTPException(401, { message: "Authorization required" })
    }

    const token = authHeader.slice(7)

    // Check internal service token first
    if (c.env.INTERNAL_API_TOKEN && token === c.env.INTERNAL_API_TOKEN) {
      c.set("userId", "internal")
      return next()
    }

    // Then check user API token
    const userId = await validateApiToken(token)
    if (!userId) {
      throw new HTTPException(401, { message: "Invalid token" })
    }

    c.set("userId", userId)
    await next()
  }
)
```
