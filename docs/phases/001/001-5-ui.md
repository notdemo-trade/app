### Email/Password Login

Replace `components/auth/google-login.tsx` with `components/auth/login-form.tsx`:

```tsx
interface LoginFormProps {
  onSuccess?: () => void
}

export function LoginForm({ onSuccess }: LoginFormProps) {
  const [isLogin, setIsLogin] = useState(true)
  const [error, setError] = useState<string>()

  const form = useForm({
    defaultValues: { email: "", password: "", name: "" },
    onSubmit: async ({ value }) => {
      setError(undefined)

      if (isLogin) {
        const result = await authClient.signIn.email({
          email: value.email,
          password: value.password,
        })
        if (result.error) {
          setError(result.error.message)
          return
        }
      } else {
        const result = await authClient.signUp.email({
          email: value.email,
          password: value.password,
          name: value.name,
        })
        if (result.error) {
          setError(result.error.message)
          return
        }
      }

      onSuccess?.()
    },
  })

  // ... form JSX
}
```

### Token Management Page

`routes/_auth/settings/tokens.tsx`:

```tsx
export const Route = createFileRoute("/_auth/settings/tokens")({
  component: TokensPage,
})

function TokensPage() {
  const { data: tokens, refetch } = useQuery(tokenListQueryOptions())
  const [newToken, setNewToken] = useState<{ token: string; type: string }>()

  const accessToken = tokens?.find(t => t.type === "access")
  const killSwitchToken = tokens?.find(t => t.type === "kill_switch")

  return (
    <div className="space-y-6">
      <h1 className="text-2xl font-bold">API Tokens</h1>

      {newToken && (
        <Alert>
          <p>Copy your {newToken.type} token now. You won't see it again:</p>
          <code className="block mt-2 p-2 bg-muted rounded">{newToken.token}</code>
        </Alert>
      )}

      {/* Access Token */}
      <TokenCard
        title="Access Token"
        description="For API/agentic access (trading, signals, etc.)"
        token={accessToken}
        type="access"
        onGenerate={(result) => {
          setNewToken({ token: result.token, type: "access" })
          refetch()
        }}
        onRevoke={refetch}
      />

      {/* Kill Switch Token */}
      <TokenCard
        title="Kill Switch Token"
        description="Emergency halt for all trading (keep separate, secure)"
        token={killSwitchToken}
        type="kill_switch"
        onGenerate={(result) => {
          setNewToken({ token: result.token, type: "kill_switch" })
          refetch()
        }}
        onRevoke={refetch}
      />
    </div>
  )
}
```
