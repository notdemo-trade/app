### Alpaca Form Component

```tsx
// components/settings/alpaca-credential-form.tsx

interface AlpacaCredentialFormProps {
  existing?: CredentialInfo
}

export function AlpacaCredentialForm({ existing }: AlpacaCredentialFormProps) {
  const [showSecret, setShowSecret] = useState(false)
  const saveMutation = useSaveCredential()
  const deleteMutation = useDeleteCredential()

  const form = useForm({
    defaultValues: {
      apiKey: "",
      apiSecret: "",
      paper: true,
    },
    onSubmit: async ({ value }) => {
      await saveMutation.mutateAsync({
        provider: "alpaca",
        data: value,
        validate: true,
      })
    },
  })

  return (
    <Card>
      <CardContent>
        {existing && (
          <div className="flex items-center gap-2 mb-4">
            <Badge variant={existing.validationError ? "destructive" : "success"}>
              {existing.validationError ? "Invalid" : "Connected"}
            </Badge>
            {existing.lastValidatedAt && (
              <span className="text-xs text-muted-foreground">
                Last verified {formatRelative(existing.lastValidatedAt)}
              </span>
            )}
          </div>
        )}

        <form onSubmit={form.handleSubmit}>
          <div className="space-y-4">
            <form.Field name="apiKey">
              {(field) => (
                <div>
                  <Label>API Key</Label>
                  <Input
                    type="text"
                    placeholder="PK..."
                    value={field.state.value}
                    onChange={(e) => field.handleChange(e.target.value)}
                  />
                </div>
              )}
            </form.Field>

            <form.Field name="apiSecret">
              {(field) => (
                <div>
                  <Label>API Secret</Label>
                  <div className="relative">
                    <Input
                      type={showSecret ? "text" : "password"}
                      placeholder="Enter secret..."
                      value={field.state.value}
                      onChange={(e) => field.handleChange(e.target.value)}
                    />
                    <Button
                      type="button"
                      variant="ghost"
                      size="sm"
                      className="absolute right-2 top-1/2 -translate-y-1/2"
                      onClick={() => setShowSecret(!showSecret)}
                    >
                      {showSecret ? <EyeOff /> : <Eye />}
                    </Button>
                  </div>
                </div>
              )}
            </form.Field>

            <form.Field name="paper">
              {(field) => (
                <div className="flex items-center gap-2">
                  <Switch
                    checked={field.state.value}
                    onCheckedChange={field.handleChange}
                  />
                  <Label>Paper Trading (recommended)</Label>
                </div>
              )}
            </form.Field>
          </div>

          <div className="flex gap-2 mt-6">
            <Button type="submit" disabled={saveMutation.isPending}>
              {saveMutation.isPending ? "Validating..." : "Save & Validate"}
            </Button>
            {existing && (
              <Button
                type="button"
                variant="destructive"
                onClick={() => deleteMutation.mutate({ provider: "alpaca" })}
              >
                Remove
              </Button>
            )}
          </div>
        </form>
      </CardContent>
    </Card>
  )
}
```

### Trading Config Page

```tsx
// routes/_auth/settings/trading.tsx

export const Route = createFileRoute("/_auth/settings/trading")({
  component: TradingConfigPage,
})

function TradingConfigPage() {
  const { data: config } = useQuery(tradingConfigQueryOptions())
  const { data: credentials } = useQuery(credentialsQueryOptions())
  const updateMutation = useUpdateTradingConfig()

  // Get available models based on configured providers
  const availableModels = useMemo(() => {
    const models: string[] = []
    credentials?.forEach(cred => {
      if (LLM_PROVIDERS.includes(cred.provider)) {
        PROVIDER_MODELS[cred.provider]?.forEach(model => {
          models.push(`${cred.provider}/${model}`)
        })
      }
    })
    return models
  }, [credentials])

  const form = useForm({
    defaultValues: config ?? DEFAULT_TRADING_CONFIG,
    onSubmit: async ({ value }) => {
      await updateMutation.mutateAsync(value)
    },
  })

  return (
    <div className="space-y-8">
      <h1 className="text-2xl font-bold">Trading Configuration</h1>

      <form onSubmit={form.handleSubmit}>
        {/* Position Limits */}
        <Card>
          <CardHeader>
            <CardTitle>Position Limits</CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            <NumberField
              form={form}
              name="maxPositions"
              label="Max Concurrent Positions"
              min={1}
              max={50}
            />
            <NumberField
              form={form}
              name="maxPositionValue"
              label="Max Position Value ($)"
              min={100}
              max={100000}
            />
            <NumberField
              form={form}
              name="maxNotionalPerTrade"
              label="Max per Trade ($)"
              min={100}
              max={100000}
            />
          </CardContent>
        </Card>

        {/* Risk Management */}
        <Card>
          <CardHeader>
            <CardTitle>Risk Management</CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            <PercentField
              form={form}
              name="stopLossPct"
              label="Stop Loss %"
              min={1}
              max={50}
            />
            <PercentField
              form={form}
              name="takeProfitPct"
              label="Take Profit %"
              min={1}
              max={100}
            />
            <PercentField
              form={form}
              name="maxDailyLossPct"
              label="Max Daily Loss %"
              min={0.1}
              max={10}
            />
          </CardContent>
        </Card>

        {/* Model Selection */}
        <Card>
          <CardHeader>
            <CardTitle>AI Models</CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            <ModelSelect
              form={form}
              name="researchModel"
              label="Research Model"
              models={availableModels}
              description="Used for market research and sentiment analysis"
            />
            <ModelSelect
              form={form}
              name="analystModel"
              label="Analyst Model"
              models={availableModels}
              description="Used for trade decisions"
            />
          </CardContent>
        </Card>

        <Button type="submit" className="mt-6">
          Save Configuration
        </Button>
      </form>
    </div>
  )
}
```
