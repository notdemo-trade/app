## API Handlers

```ts
// apps/data-service/src/hono/handlers/approval-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import { ApproveOrderSchema, RejectApprovalSchema, ApprovalConfigSchema } from "@repo/data-ops/zod-schema/approvals"
import {
  validateApprovalToken,
  consumeApprovalToken,
  rejectApproval,
} from "@repo/data-ops/policy/approval"
import {
  getPendingApprovals,
  getApprovalHistory,
  getApprovalById,
} from "@repo/data-ops/db/queries/approvals"
import { getRiskState } from "@repo/data-ops/db/queries/risk-state"
import { getAlpacaTradingClient } from "../utils/alpaca"

const approvalRouter = new Hono<{ Bindings: Env }>()

// Get pending approvals
approvalRouter.get("/pending", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const pendingList = await getPendingApprovals(c.get("db"), userId)

  const now = Date.now()
  const pending = pendingList.map((a) => ({
    id: a.id,
    orderParams: a.orderParams,
    source: a.source,
    expiresAt: a.expiresAt.toISOString(),
    createdAt: a.createdAt.toISOString(),
    timeRemaining: Math.max(0, Math.floor((a.expiresAt.getTime() - now) / 1000)),
  }))

  return c.json({ data: pending })
})

// Get approval history
approvalRouter.get("/history", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const limit = parseInt(c.req.query("limit") ?? "50", 10)

  const history = await getApprovalHistory(c.get("db"), userId, limit)

  return c.json({
    data: history.map((a) => ({
      id: a.id,
      orderParams: a.orderParams,
      source: a.source,
      status: a.usedAt ? "executed" : a.rejectedAt ? "rejected" : "expired",
      createdAt: a.createdAt.toISOString(),
      usedAt: a.usedAt?.toISOString(),
      rejectedAt: a.rejectedAt?.toISOString(),
      rejectionReason: a.rejectionReason,
    })),
  })
})

// Approve and execute order
approvalRouter.post(
  "/approve",
  apiTokenMiddleware,
  zValidator("json", ApproveOrderSchema),
  async (c) => {
    const userId = c.get("userId")
    const { approvalId } = c.req.valid("json")

    // Get approval
    const approval = await getApprovalById(c.get("db"), userId, approvalId)
    if (!approval) {
      return c.json({ error: "Approval not found" }, 404)
    }

    // Check not already used/rejected
    if (approval.usedAt) {
      return c.json({ error: "Approval already used" }, 400)
    }
    if (approval.rejectedAt) {
      return c.json({ error: "Approval was rejected" }, 400)
    }

    // Check not expired
    if (new Date() > approval.expiresAt) {
      return c.json({ error: "Approval expired" }, 400)
    }

    // Check kill switch
    const riskState = await getRiskState(c.get("db"), userId)
    if (riskState.killSwitchActive) {
      return c.json({
        error: `Trading halted: ${riskState.killSwitchReason ?? "Kill switch active"}`,
      }, 400)
    }

    // Get trading client
    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      // Execute order
      const orderParams = approval.orderParams
      const order = await trading.createOrder({
        symbol: orderParams.symbol,
        side: orderParams.side,
        type: orderParams.order_type as any,
        time_in_force: orderParams.time_in_force as any,
        qty: orderParams.qty,
        notional: orderParams.notional,
        limit_price: orderParams.limit_price,
        stop_price: orderParams.stop_price,
        client_order_id: approvalId,
      })

      // Mark as used
      await consumeApprovalToken(c.get("db"), approvalId)

      return c.json({
        data: {
          message: "Order submitted",
          order: {
            id: order.id,
            symbol: order.symbol,
            status: order.status,
            side: order.side,
          },
        },
      })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Reject approval
approvalRouter.post(
  "/reject",
  apiTokenMiddleware,
  zValidator("json", RejectApprovalSchema),
  async (c) => {
    const userId = c.get("userId")
    const { approvalId, reason } = c.req.valid("json")

    const approval = await getApprovalById(c.get("db"), userId, approvalId)
    if (!approval) {
      return c.json({ error: "Approval not found" }, 404)
    }

    if (approval.usedAt || approval.rejectedAt) {
      return c.json({ error: "Approval already processed" }, 400)
    }

    await rejectApproval(c.get("db"), approvalId, reason)

    return c.json({ data: { message: "Approval rejected" } })
  }
)

export { approvalRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/approvals/pending` | Bearer | List pending approvals |
| GET | `/api/approvals/history` | Bearer | List approval history |
| POST | `/api/approvals/approve` | Bearer | Approve and execute order |
| POST | `/api/approvals/reject` | Bearer | Reject approval |
