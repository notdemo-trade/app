```ts
// apps/user-application/src/lib/order-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  previewOrder,
  submitOrder,
  listOrders,
  cancelOrder,
  closePosition,
  getOptionsExpirations,
  getOptionsChain,
  getOptionsByDelta,
  previewOptionsOrder,
  submitOptionsOrder,
} from "@/core/functions/orders/direct"
import type { CreateOrderInput, CreateOptionsOrderInput } from "@repo/data-ops/zod-schema/orders"

export const ordersQueryOptions = (status: "open" | "closed" | "all" = "open", limit = 50) =>
  queryOptions({
    queryKey: ["orders", status, limit],
    queryFn: () => listOrders({ status, limit }),
    staleTime: 10_000,
  })

export const optionsExpirationsQueryOptions = (underlying: string) =>
  queryOptions({
    queryKey: ["options", "expirations", underlying],
    queryFn: () => getOptionsExpirations(underlying),
    enabled: !!underlying,
    staleTime: 60_000,
  })

export const optionsChainQueryOptions = (underlying: string, expiration: string) =>
  queryOptions({
    queryKey: ["options", "chain", underlying, expiration],
    queryFn: () => getOptionsChain({ underlying, expiration }),
    enabled: !!underlying && !!expiration,
    staleTime: 30_000,
  })

export const useOrders = (status?: "open" | "closed" | "all", limit?: number) =>
  useQuery(ordersQueryOptions(status, limit))

export const useOptionsExpirations = (underlying: string) =>
  useQuery(optionsExpirationsQueryOptions(underlying))

export const useOptionsChain = (underlying: string, expiration: string) =>
  useQuery(optionsChainQueryOptions(underlying, expiration))

export const usePreviewOrder = () =>
  useMutation({ mutationFn: (data: CreateOrderInput) => previewOrder(data) })

export const useSubmitOrder = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: CreateOrderInput) => submitOrder(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["orders"] })
      queryClient.invalidateQueries({ queryKey: ["positions"] })
      queryClient.invalidateQueries({ queryKey: ["account"] })
    },
  })
}

export const useCancelOrder = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (orderId: string) => cancelOrder(orderId),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["orders"] })
    },
  })
}

export const useClosePosition = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: { symbol: string; qty?: number; percentage?: number }) => closePosition(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["positions"] })
      queryClient.invalidateQueries({ queryKey: ["orders"] })
      queryClient.invalidateQueries({ queryKey: ["account"] })
    },
  })
}

export const useSubmitOptionsOrder = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: CreateOptionsOrderInput) => submitOptionsOrder(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["orders"] })
      queryClient.invalidateQueries({ queryKey: ["positions"] })
    },
  })
}
```

## UI Components

### Order Entry Form

```tsx
// apps/user-application/src/components/orders/order-entry-form.tsx

import { useState } from "react"
import { usePreviewOrder, useSubmitOrder } from "@/lib/order-queries"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { Label } from "@/components/ui/label"
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group"
import { OrderConfirmDialog } from "./order-confirm-dialog"

export function OrderEntryForm() {
  const [symbol, setSymbol] = useState("")
  const [side, setSide] = useState<"buy" | "sell">("buy")
  const [orderType, setOrderType] = useState<"market" | "limit" | "stop" | "stop_limit">("market")
  const [qty, setQty] = useState("")
  const [limitPrice, setLimitPrice] = useState("")
  const [stopPrice, setStopPrice] = useState("")
  const [showConfirm, setShowConfirm] = useState(false)

  const preview = usePreviewOrder()
  const submit = useSubmitOrder()

  const handlePreview = () => {
    const data = {
      symbol: symbol.toUpperCase(),
      side,
      type: orderType,
      time_in_force: "day" as const,
      qty: qty ? parseFloat(qty) : undefined,
      limit_price: limitPrice ? parseFloat(limitPrice) : undefined,
      stop_price: stopPrice ? parseFloat(stopPrice) : undefined,
    }
    preview.mutate(data, {
      onSuccess: () => setShowConfirm(true),
    })
  }

  const handleSubmit = () => {
    const data = {
      symbol: symbol.toUpperCase(),
      side,
      type: orderType,
      time_in_force: "day" as const,
      qty: qty ? parseFloat(qty) : undefined,
      limit_price: limitPrice ? parseFloat(limitPrice) : undefined,
      stop_price: stopPrice ? parseFloat(stopPrice) : undefined,
    }
    submit.mutate(data, {
      onSuccess: () => {
        setShowConfirm(false)
        setSymbol("")
        setQty("")
        setLimitPrice("")
        setStopPrice("")
      },
    })
  }

  const needsLimitPrice = orderType === "limit" || orderType === "stop_limit"
  const needsStopPrice = orderType === "stop" || orderType === "stop_limit"

  return (
    <div className="space-y-4 p-4 border rounded-lg">
      <div className="grid grid-cols-2 gap-4">
        <div>
          <Label>Symbol</Label>
          <Input
            value={symbol}
            onChange={(e) => setSymbol(e.target.value.toUpperCase())}
            placeholder="AAPL"
          />
        </div>
        <div>
          <Label>Quantity</Label>
          <Input
            type="number"
            value={qty}
            onChange={(e) => setQty(e.target.value)}
            placeholder="10"
          />
        </div>
      </div>

      <div>
        <Label>Side</Label>
        <RadioGroup value={side} onValueChange={(v) => setSide(v as "buy" | "sell")} className="flex gap-4">
          <div className="flex items-center space-x-2">
            <RadioGroupItem value="buy" id="buy" />
            <Label htmlFor="buy" className="text-green-600">Buy</Label>
          </div>
          <div className="flex items-center space-x-2">
            <RadioGroupItem value="sell" id="sell" />
            <Label htmlFor="sell" className="text-red-600">Sell</Label>
          </div>
        </RadioGroup>
      </div>

      <div>
        <Label>Order Type</Label>
        <Select value={orderType} onValueChange={(v) => setOrderType(v as typeof orderType)}>
          <SelectTrigger>
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="market">Market</SelectItem>
            <SelectItem value="limit">Limit</SelectItem>
            <SelectItem value="stop">Stop</SelectItem>
            <SelectItem value="stop_limit">Stop Limit</SelectItem>
          </SelectContent>
        </Select>
      </div>

      {needsLimitPrice && (
        <div>
          <Label>Limit Price</Label>
          <Input
            type="number"
            step="0.01"
            value={limitPrice}
            onChange={(e) => setLimitPrice(e.target.value)}
            placeholder="150.00"
          />
        </div>
      )}

      {needsStopPrice && (
        <div>
          <Label>Stop Price</Label>
          <Input
            type="number"
            step="0.01"
            value={stopPrice}
            onChange={(e) => setStopPrice(e.target.value)}
            placeholder="145.00"
          />
        </div>
      )}

      <Button
        onClick={handlePreview}
        disabled={!symbol || !qty || preview.isPending}
        className="w-full"
      >
        {preview.isPending ? "Loading..." : "Review Order"}
      </Button>

      {preview.error && (
        <p className="text-sm text-destructive">{String(preview.error)}</p>
      )}

      <OrderConfirmDialog
        open={showConfirm}
        onOpenChange={setShowConfirm}
        preview={preview.data?.preview}
        clock={preview.data?.clock}
        onConfirm={handleSubmit}
        isSubmitting={submit.isPending}
      />
    </div>
  )
}
```

### Order Confirm Dialog

```tsx
// apps/user-application/src/components/orders/order-confirm-dialog.tsx

import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from "@/components/ui/alert-dialog"
import { Badge } from "@/components/ui/badge"
import type { OrderPreview, MarketClock } from "@repo/data-ops/providers/alpaca/types"

interface OrderConfirmDialogProps {
  open: boolean
  onOpenChange: (open: boolean) => void
  preview?: OrderPreview
  clock?: MarketClock
  onConfirm: () => void
  isSubmitting: boolean
}

export function OrderConfirmDialog({
  open,
  onOpenChange,
  preview,
  clock,
  onConfirm,
  isSubmitting,
}: OrderConfirmDialogProps) {
  if (!preview) return null

  const isCrypto = preview.asset_class === "crypto"
  const marketClosed = !isCrypto && !clock?.is_open

  return (
    <AlertDialog open={open} onOpenChange={onOpenChange}>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>Confirm Order</AlertDialogTitle>
          <AlertDialogDescription>Review your order before submitting.</AlertDialogDescription>
        </AlertDialogHeader>

        <div className="space-y-3 py-4">
          <div className="flex justify-between">
            <span className="text-muted-foreground">Symbol</span>
            <span className="font-mono font-medium">{preview.symbol}</span>
          </div>
          <div className="flex justify-between">
            <span className="text-muted-foreground">Side</span>
            <Badge variant={preview.side === "buy" ? "default" : "destructive"}>
              {preview.side.toUpperCase()}
            </Badge>
          </div>
          <div className="flex justify-between">
            <span className="text-muted-foreground">Quantity</span>
            <span>{preview.qty ?? `$${preview.notional}`}</span>
          </div>
          <div className="flex justify-between">
            <span className="text-muted-foreground">Type</span>
            <span className="capitalize">{preview.order_type.replace("_", " ")}</span>
          </div>
          {preview.limit_price && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">Limit Price</span>
              <span>${preview.limit_price.toFixed(2)}</span>
            </div>
          )}
          {preview.stop_price && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">Stop Price</span>
              <span>${preview.stop_price.toFixed(2)}</span>
            </div>
          )}
          <div className="flex justify-between">
            <span className="text-muted-foreground">Est. Cost</span>
            <span className="font-medium">${preview.estimated_cost.toFixed(2)}</span>
          </div>

          {marketClosed && (
            <p className="text-sm text-yellow-600 bg-yellow-50 dark:bg-yellow-900/20 p-2 rounded">
              Market is closed. Order will be queued for next open.
            </p>
          )}
        </div>

        <AlertDialogFooter>
          <AlertDialogCancel>Cancel</AlertDialogCancel>
          <AlertDialogAction onClick={onConfirm} disabled={isSubmitting}>
            {isSubmitting ? "Submitting..." : "Submit Order"}
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  )
}
```

### Options Chain Selector

```tsx
// apps/user-application/src/components/orders/options-chain-selector.tsx

import { useState } from "react"
import { useOptionsExpirations, useOptionsChain } from "@/lib/order-queries"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table"
import { Badge } from "@/components/ui/badge"
import { getDTE } from "@repo/data-ops/providers/alpaca/options"

interface OptionsChainSelectorProps {
  underlying: string
  onSelect: (contractSymbol: string, type: "call" | "put") => void
}

export function OptionsChainSelector({ underlying, onSelect }: OptionsChainSelectorProps) {
  const [expiration, setExpiration] = useState("")
  const [optionType, setOptionType] = useState<"call" | "put">("call")

  const { data: expirations, isLoading: loadingExp } = useOptionsExpirations(underlying)
  const { data: chain, isLoading: loadingChain } = useOptionsChain(underlying, expiration)

  const contracts = chain?.[optionType === "call" ? "calls" : "puts"] ?? []

  return (
    <div className="space-y-4">
      <div className="flex gap-4">
        <Select value={expiration} onValueChange={setExpiration}>
          <SelectTrigger className="w-48">
            <SelectValue placeholder="Select expiration" />
          </SelectTrigger>
          <SelectContent>
            {expirations?.map((exp) => (
              <SelectItem key={exp} value={exp}>
                {exp} ({getDTE(exp)} DTE)
              </SelectItem>
            ))}
          </SelectContent>
        </Select>

        <Select value={optionType} onValueChange={(v) => setOptionType(v as "call" | "put")}>
          <SelectTrigger className="w-32">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="call">Calls</SelectItem>
            <SelectItem value="put">Puts</SelectItem>
          </SelectContent>
        </Select>
      </div>

      {loadingChain ? (
        <div className="animate-pulse">Loading chain...</div>
      ) : contracts.length > 0 ? (
        <div className="max-h-80 overflow-auto">
          <Table>
            <TableHeader>
              <TableRow>
                <TableHead>Strike</TableHead>
                <TableHead>OI</TableHead>
                <TableHead></TableHead>
              </TableRow>
            </TableHeader>
            <TableBody>
              {contracts.map((contract) => (
                <TableRow
                  key={contract.symbol}
                  className="cursor-pointer hover:bg-accent"
                  onClick={() => onSelect(contract.symbol, optionType)}
                >
                  <TableCell className="font-mono">${contract.strike.toFixed(2)}</TableCell>
                  <TableCell>{contract.open_interest.toLocaleString()}</TableCell>
                  <TableCell>
                    <Badge variant="outline" className="text-xs">
                      Select
                    </Badge>
                  </TableCell>
                </TableRow>
              ))}
            </TableBody>
          </Table>
        </div>
      ) : (
        expiration && <p className="text-sm text-muted-foreground">No contracts found</p>
      )}
    </div>
  )
}
```

### Position Close Button

```tsx
// apps/user-application/src/components/positions/close-position-button.tsx

import { useState } from "react"
import { useClosePosition } from "@/lib/order-queries"
import { Button } from "@/components/ui/button"
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogFooter,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group"

interface ClosePositionButtonProps {
  symbol: string
  qty: number
}

export function ClosePositionButton({ symbol, qty }: ClosePositionButtonProps) {
  const [open, setOpen] = useState(false)
  const [closeType, setCloseType] = useState<"full" | "partial">("full")
  const [closeQty, setCloseQty] = useState("")

  const closeMutation = useClosePosition()

  const handleClose = () => {
    closeMutation.mutate(
      {
        symbol,
        qty: closeType === "partial" ? parseFloat(closeQty) : undefined,
      },
      {
        onSuccess: () => setOpen(false),
      }
    )
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="outline" size="sm">
          Close
        </Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Close {symbol} Position</DialogTitle>
          <DialogDescription>Current position: {qty} shares</DialogDescription>
        </DialogHeader>

        <RadioGroup value={closeType} onValueChange={(v) => setCloseType(v as "full" | "partial")}>
          <div className="flex items-center space-x-2">
            <RadioGroupItem value="full" id="full" />
            <Label htmlFor="full">Close entire position</Label>
          </div>
          <div className="flex items-center space-x-2">
            <RadioGroupItem value="partial" id="partial" />
            <Label htmlFor="partial">Close partial</Label>
          </div>
        </RadioGroup>

        {closeType === "partial" && (
          <div>
            <Label>Quantity to close</Label>
            <Input
              type="number"
              value={closeQty}
              onChange={(e) => setCloseQty(e.target.value)}
              max={qty}
            />
          </div>
        )}

        <DialogFooter>
          <Button variant="outline" onClick={() => setOpen(false)}>
            Cancel
          </Button>
          <Button
            variant="destructive"
            onClick={handleClose}
            disabled={closeMutation.isPending || (closeType === "partial" && !closeQty)}
          >
            {closeMutation.isPending ? "Closing..." : "Close Position"}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```
