```ts
// packages/data-ops/src/db/schema/orders.ts

import { pgTable, text, timestamp, uuid, real, integer, jsonb, index } from "drizzle-orm/pg-core"
import { users } from "./users"

export const orders = pgTable(
  "orders",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    userId: uuid("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" }),
    alpacaOrderId: text("alpaca_order_id").notNull(),
    clientOrderId: text("client_order_id"),
    symbol: text("symbol").notNull(),
    assetClass: text("asset_class").notNull(), // us_equity, crypto
    side: text("side").notNull(), // buy, sell
    qty: real("qty"),
    notional: real("notional"),
    orderType: text("order_type").notNull(), // market, limit, stop, stop_limit
    limitPrice: real("limit_price"),
    stopPrice: real("stop_price"),
    timeInForce: text("time_in_force").notNull(),
    status: text("status").notNull(),
    filledQty: real("filled_qty"),
    filledAvgPrice: real("filled_avg_price"),
    // Options fields
    underlying: text("underlying"),
    expiration: text("expiration"),
    strike: real("strike"),
    optionType: text("option_type"), // call, put
    createdAt: timestamp("created_at").defaultNow().notNull(),
    updatedAt: timestamp("updated_at").defaultNow().notNull(),
    filledAt: timestamp("filled_at"),
    canceledAt: timestamp("canceled_at"),
  },
  (table) => ({
    userIdIdx: index("orders_user_id_idx").on(table.userId),
    symbolIdx: index("orders_symbol_idx").on(table.symbol),
    statusIdx: index("orders_status_idx").on(table.status),
    createdAtIdx: index("orders_created_at_idx").on(table.createdAt),
  })
)

export type OrderRecord = typeof orders.$inferSelect
export type NewOrderRecord = typeof orders.$inferInsert
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/orders.ts

import { z } from "zod"

export const OrderSideSchema = z.enum(["buy", "sell"])
export const OrderTypeSchema = z.enum(["market", "limit", "stop", "stop_limit"])
export const TimeInForceSchema = z.enum(["day", "gtc", "ioc", "fok"])

export const CreateOrderSchema = z
  .object({
    symbol: z.string().min(1).max(20),
    side: OrderSideSchema,
    type: OrderTypeSchema,
    time_in_force: TimeInForceSchema.default("day"),
    qty: z.number().positive().optional(),
    notional: z.number().positive().optional(),
    limit_price: z.number().positive().optional(),
    stop_price: z.number().positive().optional(),
  })
  .refine((data) => data.qty !== undefined || data.notional !== undefined, {
    message: "Either qty or notional required",
  })
  .refine(
    (data) => {
      if (data.type === "limit" || data.type === "stop_limit") {
        return data.limit_price !== undefined
      }
      return true
    },
    { message: "limit_price required for limit orders" }
  )
  .refine(
    (data) => {
      if (data.type === "stop" || data.type === "stop_limit") {
        return data.stop_price !== undefined
      }
      return true
    },
    { message: "stop_price required for stop orders" }
  )

export const CreateOptionsOrderSchema = z.object({
  contract_symbol: z.string().min(10).max(30),
  side: OrderSideSchema,
  qty: z.number().int().positive(),
  order_type: z.enum(["market", "limit"]),
  limit_price: z.number().positive().optional(),
  time_in_force: z.enum(["day", "gtc"]).default("day"),
})

export const ListOrdersSchema = z.object({
  status: z.enum(["open", "closed", "all"]).optional().default("open"),
  limit: z.coerce.number().min(1).max(500).optional().default(50),
  symbol: z.string().optional(),
})

export const ClosePositionSchema = z.object({
  symbol: z.string().min(1),
  qty: z.number().positive().optional(),
  percentage: z.number().min(0).max(100).optional(),
})

export const GetOptionsChainSchema = z.object({
  underlying: z.string().min(1).max(10),
  expiration: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
})

export const GetOptionsByDeltaSchema = z.object({
  underlying: z.string().min(1).max(10),
  expiration: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
  type: z.enum(["call", "put"]),
  min_delta: z.coerce.number().min(0).max(1).default(0.3),
  max_delta: z.coerce.number().min(0).max(1).default(0.7),
})

export type CreateOrderInput = z.infer<typeof CreateOrderSchema>
export type CreateOptionsOrderInput = z.infer<typeof CreateOptionsOrderSchema>
```
