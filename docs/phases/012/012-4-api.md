## API Handlers (simplified via getAgentByName)

```ts
// apps/data-service/src/hono/handlers/agent-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { getAgentByName } from "agents"
import { apiTokenMiddleware } from "../middleware/auth"
import { AgentConfigSchema } from "@repo/data-ops/zod-schema/agent"
import type { TradingAgent } from "../../agents/trading-agent"

const agentRouter = new Hono<{ Bindings: Env }>()

// Direct RPC — no internal HTTP routing
function getAgent(env: Env, userId: string) {
  return getAgentByName<TradingAgent>(env.TradingAgent, userId)
}

agentRouter.post("/enable", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  const result = await agent.enable()
  return c.json(result)
})

agentRouter.post("/disable", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  const result = await agent.disable()
  return c.json(result)
})

agentRouter.get("/status", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  return c.json(agent.getStatus())
})

agentRouter.get("/config", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  return c.json(agent.getAgentConfig())
})

agentRouter.post(
  "/config",
  apiTokenMiddleware,
  zValidator("json", AgentConfigSchema),
  async (c) => {
    const userId = c.get("userId")
    const config = c.req.valid("json")
    const agent = await getAgent(c.env, userId)
    const result = await agent.updateConfig(config)
    return c.json(result)
  }
)

agentRouter.post("/trigger", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  const result = await agent.trigger()
  return c.json(result)
})

agentRouter.get("/activity", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const agent = await getAgent(c.env, userId)
  return c.json(agent.getActivity())
})

export { agentRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| POST | `/api/agent/enable` | Bearer | Enable trading agent |
| POST | `/api/agent/disable` | Bearer | Disable trading agent |
| GET | `/api/agent/status` | Bearer | Get agent status |
| GET | `/api/agent/config` | Bearer | Get agent config |
| POST | `/api/agent/config` | Bearer | Update agent config |
| POST | `/api/agent/trigger` | Bearer | Manual trigger cycle |
| GET | `/api/agent/activity` | Bearer | Get activity log |

## Client: useAgent Hook (replaces React Query polling)

```tsx
// apps/user-application/src/lib/agent-connection.ts

import { useAgent } from "agents/react"
import type { TradingAgent } from "@repo/data-ops/agent/types"

export function useTradingAgent(userId: string) {
  const agent = useAgent<TradingAgent>({
    agent: "TradingAgent",
    name: userId,
    onStateUpdate: (state, source) => {
      // state updates arrive automatically via WebSocket
      // source: "server" | "client"
    },
  })

  return {
    // Auto-synced state (no polling)
    state: agent.state,

    // Type-safe RPC
    enable: () => agent.stub.enable(),
    disable: () => agent.stub.disable(),
    getStatus: () => agent.stub.getStatus(),
    getConfig: () => agent.stub.getAgentConfig(),
    updateConfig: (updates: Partial<AgentConfig>) => agent.stub.updateConfig(updates),
    trigger: () => agent.stub.trigger(),
    getActivity: (limit?: number) => agent.stub.getActivity(limit),

    // Streaming
    streamAnalysis: (symbols: string[], handlers: {
      onChunk: (chunk: string) => void
      onDone: () => void
    }) => agent.call("streamAnalysis", [symbols], {
      stream: handlers,
    }),

    // Connection state
    ready: agent.ready,
  }
}
```
