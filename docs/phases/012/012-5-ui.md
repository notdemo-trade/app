### Agent Status Card (real-time)

```tsx
// apps/user-application/src/components/agent/agent-status-card.tsx

import { useTradingAgent } from "@/lib/agent-connection"
import { Button } from "@/components/ui/button"
import { Badge } from "@/components/ui/badge"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { formatDistanceToNow } from "date-fns"
import { Play, Pause, RefreshCw, Bot } from "lucide-react"
import { useState } from "react"

export function AgentStatusCard({ userId }: { userId: string }) {
  const agent = useTradingAgent(userId)
  const [loading, setLoading] = useState<string | null>(null)

  // state is auto-synced — no useQuery, no polling
  const isEnabled = agent.state?.enabled ?? false

  const handleAction = async (action: "enable" | "disable" | "trigger") => {
    setLoading(action)
    try {
      await agent[action]()
    } finally {
      setLoading(null)
    }
  }

  return (
    <Card>
      <CardHeader className="pb-2">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <Bot className="h-5 w-5" />
            <CardTitle className="text-lg">Trading Agent</CardTitle>
          </div>
          <Badge variant={isEnabled ? "default" : "secondary"}>
            {isEnabled ? "Running" : "Stopped"}
          </Badge>
        </div>
      </CardHeader>

      <CardContent className="space-y-4">
        {/* Stats — update in real-time */}
        <div className="grid grid-cols-3 gap-4 text-sm">
          <div>
            <div className="text-muted-foreground">Cycles</div>
            <div className="font-mono text-lg">{agent.state?.cycleCount ?? 0}</div>
          </div>
          <div>
            <div className="text-muted-foreground">Errors</div>
            <div className="font-mono text-lg">{agent.state?.errorCount ?? 0}</div>
          </div>
          <div>
            <div className="text-muted-foreground">Last Analysis</div>
            <div className="text-xs">
              {agent.state?.lastAnalysisAt
                ? formatDistanceToNow(new Date(agent.state.lastAnalysisAt), { addSuffix: true })
                : "Never"}
            </div>
          </div>
        </div>

        {agent.state?.lastError && (
          <p className="text-red-500 text-xs">Error: {agent.state.lastError}</p>
        )}

        {/* Controls */}
        <div className="flex gap-2">
          {isEnabled ? (
            <>
              <Button
                variant="outline"
                onClick={() => handleAction("disable")}
                disabled={loading === "disable"}
              >
                <Pause className="h-4 w-4 mr-1" />
                {loading === "disable" ? "Stopping..." : "Stop"}
              </Button>
              <Button
                variant="outline"
                onClick={() => handleAction("trigger")}
                disabled={loading === "trigger"}
              >
                <RefreshCw className="h-4 w-4 mr-1" />
                {loading === "trigger" ? "Running..." : "Trigger"}
              </Button>
            </>
          ) : (
            <Button
              onClick={() => handleAction("enable")}
              disabled={loading === "enable"}
            >
              <Play className="h-4 w-4 mr-1" />
              {loading === "enable" ? "Starting..." : "Start Agent"}
            </Button>
          )}
        </div>
      </CardContent>
    </Card>
  )
}
```

### Agent Activity Log

```tsx
// apps/user-application/src/components/agent/agent-activity-log.tsx

import { useTradingAgent } from "@/lib/agent-connection"
import { Badge } from "@/components/ui/badge"
import { formatDistanceToNow } from "date-fns"
import { cn } from "@/lib/utils"
import { useEffect, useState } from "react"
import type { AgentActivity } from "@repo/data-ops/agent/types"

const actionColors: Record<string, string> = {
  started: "bg-green-500",
  stopped: "bg-gray-500",
  signal_gathered: "bg-blue-500",
  analysis_started: "bg-yellow-500",
  analysis_completed: "bg-yellow-600",
  trade_proposed: "bg-purple-500",
  trade_executed: "bg-green-600",
  trade_rejected: "bg-red-500",
  error: "bg-red-600",
}

export function AgentActivityLog({ userId }: { userId: string }) {
  const agent = useTradingAgent(userId)
  const [activity, setActivity] = useState<AgentActivity[]>([])

  // Refetch activity when state changes (new cycle completed etc)
  useEffect(() => {
    agent.getActivity(50).then(setActivity)
  }, [agent.state?.cycleCount, agent.state?.enabled])

  if (!activity.length) {
    return <p className="text-sm text-muted-foreground">No activity yet</p>
  }

  return (
    <div className="space-y-2">
      {activity.map((item) => (
        <div
          key={item.id}
          className="flex items-center gap-3 p-2 rounded border text-sm"
        >
          <div
            className={cn(
              "w-2 h-2 rounded-full",
              actionColors[item.action] ?? "bg-gray-400"
            )}
          />
          <div className="flex-1">
            <span className="font-medium capitalize">
              {item.action.replace(/_/g, " ")}
            </span>
            {item.symbol && (
              <span className="ml-2 font-mono text-muted-foreground">
                {item.symbol}
              </span>
            )}
          </div>
          <span className="text-xs text-muted-foreground">
            {formatDistanceToNow(new Date(item.timestamp), { addSuffix: true })}
          </span>
        </div>
      ))}
    </div>
  )
}
```

### Agent Config Form

```tsx
// apps/user-application/src/components/agent/agent-config-form.tsx

import { useTradingAgent } from "@/lib/agent-connection"
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { AgentConfigSchema, type AgentConfigInput } from "@repo/data-ops/zod-schema/agent"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Switch } from "@/components/ui/switch"
import { useEffect, useState } from "react"

export function AgentConfigForm({ userId }: { userId: string }) {
  const agent = useTradingAgent(userId)
  const [saving, setSaving] = useState(false)

  const form = useForm<AgentConfigInput>({
    resolver: zodResolver(AgentConfigSchema),
  })

  useEffect(() => {
    agent.getConfig().then((config) => form.reset(config))
  }, [])

  const onSubmit = async (values: AgentConfigInput) => {
    setSaving(true)
    try {
      await agent.updateConfig(values)
    } finally {
      setSaving(false)
    }
  }

  return (
    <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
      <div className="grid gap-4 md:grid-cols-2">
        <div>
          <Label>Signal Gather Interval (sec)</Label>
          <Input
            type="number"
            {...form.register("dataPollIntervalSec", { valueAsNumber: true })}
          />
        </div>
        <div>
          <Label>Analysis Interval (sec)</Label>
          <Input
            type="number"
            {...form.register("analystIntervalSec", { valueAsNumber: true })}
          />
        </div>
        <div>
          <Label>Min Sentiment Score (0-1)</Label>
          <Input
            type="number"
            step="0.1"
            {...form.register("minSentimentScore", { valueAsNumber: true })}
          />
        </div>
        <div>
          <Label>Min Confidence (0-1)</Label>
          <Input
            type="number"
            step="0.1"
            {...form.register("minAnalystConfidence", { valueAsNumber: true })}
          />
        </div>
        <div>
          <Label>Position Size (% of cash)</Label>
          <Input
            type="number"
            step="0.01"
            {...form.register("positionSizePctOfCash", { valueAsNumber: true })}
          />
        </div>
        <div>
          <Label>Max Position Value ($)</Label>
          <Input
            type="number"
            {...form.register("maxPositionValue", { valueAsNumber: true })}
          />
        </div>
      </div>

      <div className="flex items-center justify-between">
        <Label>Auto-Approve Trades</Label>
        <Switch
          checked={form.watch("autoApproveEnabled")}
          onCheckedChange={(v) => form.setValue("autoApproveEnabled", v)}
        />
      </div>

      {form.watch("autoApproveEnabled") && (
        <div>
          <Label>Auto-Approve Max Notional ($)</Label>
          <Input
            type="number"
            {...form.register("autoApproveMaxNotional", { valueAsNumber: true })}
          />
        </div>
      )}

      <Button type="submit" disabled={saving}>
        {saving ? "Saving..." : "Save Configuration"}
      </Button>
    </form>
  )
}
```
