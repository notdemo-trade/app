## Database Schema

```ts
// packages/data-ops/src/db/schema/risk.ts

import { pgTable, text, timestamp, uuid, real, boolean, integer, jsonb, index } from "drizzle-orm/pg-core"
import { users } from "./users"

export const policyConfigs = pgTable(
  "policy_configs",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    userId: uuid("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" })
      .unique(),
    maxPositionPctEquity: real("max_position_pct_equity").notNull().default(0.1),
    maxOpenPositions: integer("max_open_positions").notNull().default(10),
    maxNotionalPerTrade: real("max_notional_per_trade").notNull().default(5000),
    maxDailyLossPct: real("max_daily_loss_pct").notNull().default(0.02),
    cooldownMinutesAfterLoss: integer("cooldown_minutes_after_loss").notNull().default(30),
    allowedOrderTypes: jsonb("allowed_order_types").$type<string[]>().notNull().default(["market", "limit", "stop", "stop_limit"]),
    tradingHoursOnly: boolean("trading_hours_only").notNull().default(true),
    extendedHoursAllowed: boolean("extended_hours_allowed").notNull().default(false),
    allowShortSelling: boolean("allow_short_selling").notNull().default(false),
    allowedSymbols: jsonb("allowed_symbols").$type<string[] | null>(),
    denySymbols: jsonb("deny_symbols").$type<string[]>().notNull().default([]),
    useCashOnly: boolean("use_cash_only").notNull().default(true),
    optionsConfig: jsonb("options_config").$type<OptionsPolicyConfig>(),
    updatedAt: timestamp("updated_at").defaultNow().notNull(),
  }
)

export const riskStates = pgTable(
  "risk_states",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    userId: uuid("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" })
      .unique(),
    killSwitchActive: boolean("kill_switch_active").notNull().default(false),
    killSwitchReason: text("kill_switch_reason"),
    killSwitchAt: timestamp("kill_switch_at"),
    dailyLossUsd: real("daily_loss_usd").notNull().default(0),
    dailyLossResetAt: timestamp("daily_loss_reset_at"),
    lastLossAt: timestamp("last_loss_at"),
    cooldownUntil: timestamp("cooldown_until"),
    updatedAt: timestamp("updated_at").defaultNow().notNull(),
  },
  (table) => ({
    userIdIdx: index("risk_states_user_id_idx").on(table.userId),
  })
)

export type PolicyConfigRecord = typeof policyConfigs.$inferSelect
export type RiskStateRecord = typeof riskStates.$inferSelect
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/risk.ts

import { z } from "zod"

export const PolicyConfigSchema = z.object({
  maxPositionPctEquity: z.number().min(0.01).max(1).optional(),
  maxOpenPositions: z.number().min(1).max(100).optional(),
  maxNotionalPerTrade: z.number().min(100).max(1000000).optional(),
  maxDailyLossPct: z.number().min(0.001).max(0.5).optional(),
  cooldownMinutesAfterLoss: z.number().min(0).max(1440).optional(),
  allowedOrderTypes: z.array(z.enum(["market", "limit", "stop", "stop_limit"])).optional(),
  tradingHoursOnly: z.boolean().optional(),
  extendedHoursAllowed: z.boolean().optional(),
  allowShortSelling: z.boolean().optional(),
  allowedSymbols: z.array(z.string()).nullable().optional(),
  denySymbols: z.array(z.string()).optional(),
  useCashOnly: z.boolean().optional(),
})

export const OptionsPolicyConfigSchema = z.object({
  optionsEnabled: z.boolean().optional(),
  maxPctPerOptionTrade: z.number().min(0.005).max(0.1).optional(),
  maxTotalOptionsExposurePct: z.number().min(0.01).max(0.25).optional(),
  minDte: z.number().min(7).max(365).optional(),
  maxDte: z.number().min(7).max(365).optional(),
  minDelta: z.number().min(0.1).max(0.5).optional(),
  maxDelta: z.number().min(0.3).max(0.9).optional(),
  noAveragingDown: z.boolean().optional(),
  maxOptionPositions: z.number().min(1).max(10).optional(),
})

export const RiskStateSchema = z.object({
  killSwitchActive: z.boolean(),
  killSwitchReason: z.string().nullable(),
  dailyLossUsd: z.number(),
  cooldownUntil: z.string().nullable(),
})

export const EnableKillSwitchSchema = z.object({
  reason: z.string().min(1).max(500),
})

export const DisableKillSwitchSchema = z.object({
  confirmation: z.literal("CONFIRM_RESUME_TRADING"),
})

export type PolicyConfigInput = z.infer<typeof PolicyConfigSchema>
export type OptionsPolicyConfigInput = z.infer<typeof OptionsPolicyConfigSchema>
```
