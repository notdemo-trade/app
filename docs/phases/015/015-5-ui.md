## Query Hooks

```ts
// apps/user-application/src/lib/leaderboard-queries.ts

import { queryOptions, useQuery, useMutation, useQueryClient } from "@tanstack/react-query"
import {
  getMyLeaderboardSettings,
  updateMyLeaderboardSettings,
  getLeaderboardData,
  getMyRanks,
  getTraderProfile,
  checkAliasAvailability,
} from "@/core/functions/leaderboard/direct"
import type { LeaderboardPeriod, TradingMode } from "@repo/data-ops/zod-schema/leaderboard"

export const leaderboardSettingsQueryOptions = () =>
  queryOptions({
    queryKey: ["leaderboard", "settings"],
    queryFn: () => getMyLeaderboardSettings(),
    staleTime: 60_000,
  })

export const leaderboardQueryOptions = (params: {
  period: LeaderboardPeriod
  mode: TradingMode
  page?: number
  search?: string
}) =>
  queryOptions({
    queryKey: ["leaderboard", "list", params],
    queryFn: () => getLeaderboardData(params),
    staleTime: 30_000,
  })

export const myRanksQueryOptions = (mode: TradingMode) =>
  queryOptions({
    queryKey: ["leaderboard", "my-ranks", mode],
    queryFn: () => getMyRanks({ mode }),
    staleTime: 60_000,
  })

export const traderProfileQueryOptions = (alias: string, mode: TradingMode) =>
  queryOptions({
    queryKey: ["leaderboard", "profile", alias, mode],
    queryFn: () => getTraderProfile({ alias, mode }),
    enabled: !!alias,
  })

export const useLeaderboardSettings = () => useQuery(leaderboardSettingsQueryOptions())

export const useLeaderboard = (params: {
  period: LeaderboardPeriod
  mode: TradingMode
  page?: number
  search?: string
}) => useQuery(leaderboardQueryOptions(params))

export const useMyRanks = (mode: TradingMode) => useQuery(myRanksQueryOptions(mode))

export const useTraderProfile = (alias: string, mode: TradingMode) =>
  useQuery(traderProfileQueryOptions(alias, mode))

export const useUpdateLeaderboardSettings = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: updateMyLeaderboardSettings,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["leaderboard", "settings"] })
      queryClient.invalidateQueries({ queryKey: ["leaderboard", "my-ranks"] })
    },
  })
}

export const useCheckAlias = (alias: string) =>
  useQuery({
    queryKey: ["leaderboard", "alias-check", alias],
    queryFn: () => checkAliasAvailability({ alias }),
    enabled: alias.length >= 3,
    staleTime: 10_000,
  })
```

## UI Components

### Leaderboard Settings

```tsx
// apps/user-application/src/components/leaderboard/leaderboard-settings.tsx

import { useState, useEffect } from "react"
import { useLeaderboardSettings, useUpdateLeaderboardSettings, useCheckAlias } from "@/lib/leaderboard-queries"
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card"
import { Switch } from "@/components/ui/switch"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Button } from "@/components/ui/button"
import { useDebounce } from "@/hooks/use-debounce"

export function LeaderboardSettings() {
  const { data: settings, isLoading } = useLeaderboardSettings()
  const updateMutation = useUpdateLeaderboardSettings()
  const [customAlias, setCustomAlias] = useState("")
  const debouncedAlias = useDebounce(customAlias, 500)
  const { data: aliasCheck } = useCheckAlias(debouncedAlias)

  useEffect(() => {
    if (settings?.customAlias) {
      setCustomAlias(settings.customAlias)
    }
  }, [settings?.customAlias])

  if (isLoading) {
    return <div className="animate-pulse h-48" />
  }

  const handleToggle = (isEnabled: boolean) => {
    updateMutation.mutate({ isEnabled })
  }

  const handleSaveAlias = () => {
    if (!aliasCheck?.available) return
    updateMutation.mutate({ customAlias: customAlias || null })
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Anonymous Leaderboard</CardTitle>
        <CardDescription>Compete with other traders using an anonymous alias</CardDescription>
      </CardHeader>
      <CardContent className="space-y-6">
        <div className="flex items-center justify-between">
          <div>
            <Label>Join Leaderboard</Label>
            <p className="text-sm text-muted-foreground">Your performance will be publicly ranked</p>
          </div>
          <Switch
            checked={settings?.isEnabled ?? false}
            onCheckedChange={handleToggle}
            disabled={updateMutation.isPending}
          />
        </div>

        {settings?.isEnabled && (
          <div className="space-y-4">
            <div>
              <Label>Custom Alias</Label>
              <div className="flex gap-2 mt-1">
                <Input
                  value={customAlias}
                  onChange={(e) => setCustomAlias(e.target.value)}
                  placeholder={settings.generatedAlias}
                  maxLength={20}
                />
                <Button
                  onClick={handleSaveAlias}
                  disabled={!aliasCheck?.available || updateMutation.isPending}
                >
                  Save
                </Button>
              </div>
              {debouncedAlias.length >= 3 && (
                <p className={`text-xs mt-1 ${aliasCheck?.available ? "text-green-600" : "text-red-600"}`}>
                  {aliasCheck?.available ? "Available" : "Already taken"}
                </p>
              )}
              <p className="text-xs text-muted-foreground mt-1">
                Or use auto-generated: {settings.generatedAlias}
              </p>
            </div>

            <div className="flex items-center justify-between">
              <div>
                <Label>Visible on Leaderboard</Label>
                <p className="text-sm text-muted-foreground">Hide your ranking temporarily</p>
              </div>
              <Switch
                checked={settings.isVisible}
                onCheckedChange={(isVisible) => updateMutation.mutate({ isVisible })}
                disabled={updateMutation.isPending}
              />
            </div>
          </div>
        )}
      </CardContent>
    </Card>
  )
}
```

### Leaderboard Table

```tsx
// apps/user-application/src/components/leaderboard/leaderboard-table.tsx

import { useState } from "react"
import { useLeaderboard, useMyRanks } from "@/lib/leaderboard-queries"
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table"
import { Tabs, TabsList, TabsTrigger } from "@/components/ui/tabs"
import { Input } from "@/components/ui/input"
import { Button } from "@/components/ui/button"
import { cn } from "@/lib/utils"
import { Link } from "@tanstack/react-router"
import type { LeaderboardPeriod, TradingMode } from "@repo/data-ops/zod-schema/leaderboard"

interface LeaderboardTableProps {
  initialPeriod?: LeaderboardPeriod
  mode?: TradingMode
}

export function LeaderboardTable({ initialPeriod = "30d", mode = "paper" }: LeaderboardTableProps) {
  const [period, setPeriod] = useState<LeaderboardPeriod>(initialPeriod)
  const [page, setPage] = useState(1)
  const [search, setSearch] = useState("")

  const { data, isLoading } = useLeaderboard({ period, mode, page, search: search || undefined })
  const { data: myRanks } = useMyRanks(mode)

  const currentUserId = myRanks ? Object.values(myRanks).find((r) => r)?.userId : null

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <Tabs value={period} onValueChange={(v) => setPeriod(v as LeaderboardPeriod)}>
          <TabsList>
            <TabsTrigger value="24h">24h</TabsTrigger>
            <TabsTrigger value="7d">7d</TabsTrigger>
            <TabsTrigger value="30d">30d</TabsTrigger>
            <TabsTrigger value="90d">90d</TabsTrigger>
            <TabsTrigger value="all">All</TabsTrigger>
          </TabsList>
        </Tabs>

        <Input
          placeholder="Search by alias..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          className="w-64"
        />
      </div>

      {isLoading ? (
        <div className="animate-pulse h-96" />
      ) : (
        <>
          <Table>
            <TableHeader>
              <TableRow>
                <TableHead className="w-16">Rank</TableHead>
                <TableHead>Trader</TableHead>
                <TableHead className="text-right">Score</TableHead>
                <TableHead className="text-right">ROI</TableHead>
                <TableHead className="text-right">Win Rate</TableHead>
                <TableHead className="text-right">Sharpe</TableHead>
                <TableHead className="text-right">Trades</TableHead>
              </TableRow>
            </TableHeader>
            <TableBody>
              {data?.entries.map((entry) => (
                <TableRow
                  key={entry.userId}
                  className={cn(entry.userId === currentUserId && "bg-accent")}
                >
                  <TableCell className="font-bold">{entry.rank}</TableCell>
                  <TableCell>
                    <Link
                      to="/leaderboard/user/$alias"
                      params={{ alias: entry.alias }}
                      className="hover:underline font-medium"
                    >
                      {entry.alias}
                    </Link>
                  </TableCell>
                  <TableCell className="text-right font-mono">
                    {entry.compositeScore.toFixed(2)}
                  </TableCell>
                  <TableCell
                    className={cn(
                      "text-right font-mono",
                      entry.roi >= 0 ? "text-green-600" : "text-red-600"
                    )}
                  >
                    {entry.roi >= 0 ? "+" : ""}
                    {(entry.roi * 100).toFixed(2)}%
                  </TableCell>
                  <TableCell className="text-right font-mono">
                    {(entry.winRate * 100).toFixed(0)}%
                  </TableCell>
                  <TableCell className="text-right font-mono">
                    {entry.sharpe?.toFixed(2) ?? "-"}
                  </TableCell>
                  <TableCell className="text-right">{entry.tradesCount}</TableCell>
                </TableRow>
              ))}
            </TableBody>
          </Table>

          {data && data.meta.totalPages > 1 && (
            <div className="flex items-center justify-center gap-2">
              <Button
                variant="outline"
                size="sm"
                onClick={() => setPage((p) => Math.max(1, p - 1))}
                disabled={page === 1}
              >
                Previous
              </Button>
              <span className="text-sm text-muted-foreground">
                Page {page} of {data.meta.totalPages}
              </span>
              <Button
                variant="outline"
                size="sm"
                onClick={() => setPage((p) => Math.min(data.meta.totalPages, p + 1))}
                disabled={page === data.meta.totalPages}
              >
                Next
              </Button>
            </div>
          )}
        </>
      )}
    </div>
  )
}
```

### User Profile Page

```tsx
// apps/user-application/src/components/leaderboard/trader-profile.tsx

import { useTraderProfile } from "@/lib/leaderboard-queries"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Avatar, AvatarFallback } from "@/components/ui/avatar"
import { cn } from "@/lib/utils"
import type { TradingMode, LeaderboardPeriod } from "@repo/data-ops/zod-schema/leaderboard"

interface TraderProfileProps {
  alias: string
  mode?: TradingMode
}

const PERIODS: LeaderboardPeriod[] = ["24h", "7d", "30d", "90d", "all"]

export function TraderProfile({ alias, mode = "paper" }: TraderProfileProps) {
  const { data: profile, isLoading } = useTraderProfile(alias, mode)

  if (isLoading) {
    return <div className="animate-pulse h-96" />
  }

  if (!profile) {
    return <p className="text-destructive">Trader not found</p>
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center gap-4">
        <Avatar className="h-16 w-16">
          <AvatarFallback className="text-2xl">{alias[0].toUpperCase()}</AvatarFallback>
        </Avatar>
        <div>
          <h1 className="text-2xl font-bold">{alias}</h1>
          <p className="text-muted-foreground">Anonymous Trader</p>
        </div>
      </div>

      <div className="grid grid-cols-5 gap-4">
        {PERIODS.map((period) => {
          const rankData = profile.ranks[period]
          return (
            <Card key={period}>
              <CardHeader className="pb-2">
                <CardTitle className="text-sm text-muted-foreground uppercase">{period}</CardTitle>
              </CardHeader>
              <CardContent>
                {rankData ? (
                  <>
                    <div className="text-3xl font-bold">#{rankData.rank}</div>
                    <div className="text-sm text-muted-foreground">
                      of {rankData.totalParticipants}
                    </div>
                    <div className="mt-2 space-y-1 text-xs">
                      <div className="flex justify-between">
                        <span>Score</span>
                        <span className="font-mono">{rankData.compositeScore.toFixed(2)}</span>
                      </div>
                      <div className="flex justify-between">
                        <span>ROI</span>
                        <span
                          className={cn(
                            "font-mono",
                            rankData.roi >= 0 ? "text-green-600" : "text-red-600"
                          )}
                        >
                          {rankData.roi >= 0 ? "+" : ""}
                          {(rankData.roi * 100).toFixed(1)}%
                        </span>
                      </div>
                      <div className="flex justify-between">
                        <span>Win Rate</span>
                        <span className="font-mono">{(rankData.winRate * 100).toFixed(0)}%</span>
                      </div>
                    </div>
                  </>
                ) : (
                  <div className="text-muted-foreground">-</div>
                )}
              </CardContent>
            </Card>
          )
        })}
      </div>
    </div>
  )
}
```
