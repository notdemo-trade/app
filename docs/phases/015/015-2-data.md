## Database Schema

```sql
-- packages/data-ops/src/db/migrations/0012_leaderboard.sql

CREATE TABLE leaderboard_settings (
  user_id TEXT PRIMARY KEY REFERENCES "user"(id) ON DELETE CASCADE,
  is_enabled BOOLEAN NOT NULL DEFAULT false,
  custom_alias TEXT UNIQUE,
  generated_alias TEXT NOT NULL UNIQUE,
  alpaca_account_id TEXT UNIQUE,
  alias_last_changed_at TIMESTAMPTZ,
  is_visible BOOLEAN NOT NULL DEFAULT true,
  trading_mode TEXT NOT NULL DEFAULT 'paper' CHECK (trading_mode IN ('paper', 'live')),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_leaderboard_settings_alias ON leaderboard_settings(custom_alias) WHERE custom_alias IS NOT NULL;
CREATE INDEX idx_leaderboard_settings_enabled ON leaderboard_settings(is_enabled, is_visible);

CREATE TABLE leaderboard_snapshots (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES "user"(id) ON DELETE CASCADE,
  period TEXT NOT NULL CHECK (period IN ('24h', '7d', '30d', '90d', 'all')),
  trading_mode TEXT NOT NULL CHECK (trading_mode IN ('paper', 'live')),
  period_start TIMESTAMPTZ NOT NULL,
  period_end TIMESTAMPTZ NOT NULL,
  roi REAL NOT NULL,
  sharpe REAL,
  win_rate REAL NOT NULL,
  max_drawdown REAL NOT NULL,
  trades_count INTEGER NOT NULL,
  win_count INTEGER NOT NULL,
  loss_count INTEGER NOT NULL,
  total_volume REAL NOT NULL,
  avg_hold_time_hours REAL NOT NULL,
  starting_equity REAL NOT NULL,
  ending_equity REAL NOT NULL,
  composite_score REAL,
  normalized_roi REAL,
  normalized_sharpe REAL,
  normalized_win_rate REAL,
  normalized_max_drawdown REAL,
  rank INTEGER,
  total_participants INTEGER,
  meets_minimum_requirements BOOLEAN NOT NULL DEFAULT false,
  snapshot_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_snapshots_user_period ON leaderboard_snapshots(user_id, period, trading_mode);
CREATE INDEX idx_snapshots_period_rank ON leaderboard_snapshots(period, trading_mode, rank) WHERE rank IS NOT NULL;
CREATE INDEX idx_snapshots_qualified ON leaderboard_snapshots(period, trading_mode, meets_minimum_requirements);

-- Denormalized for fast leaderboard queries
CREATE TABLE leaderboard_ranks (
  user_id TEXT NOT NULL,
  period TEXT NOT NULL CHECK (period IN ('24h', '7d', '30d', '90d', 'all')),
  trading_mode TEXT NOT NULL CHECK (trading_mode IN ('paper', 'live')),
  rank INTEGER NOT NULL,
  composite_score REAL NOT NULL,
  alias TEXT NOT NULL,
  roi REAL NOT NULL,
  win_rate REAL NOT NULL,
  sharpe REAL,
  trades_count INTEGER NOT NULL,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (period, trading_mode, rank)
);

CREATE INDEX idx_ranks_user ON leaderboard_ranks(user_id, period, trading_mode);
CREATE INDEX idx_ranks_alias ON leaderboard_ranks(alias);
```

## Drizzle Schema

```ts
// packages/data-ops/src/db/schema/leaderboard.ts

import { pgTable, text, real, integer, boolean, timestamp, index, primaryKey, uniqueIndex } from "drizzle-orm/pg-core"
import { user } from "./auth"
import type { LeaderboardPeriod, TradingMode } from "../../types/leaderboard"

export const leaderboardSettings = pgTable(
  "leaderboard_settings",
  {
    userId: text("user_id").primaryKey().references(() => user.id, { onDelete: "cascade" }),
    isEnabled: boolean("is_enabled").notNull().default(false),
    customAlias: text("custom_alias").unique(),
    generatedAlias: text("generated_alias").notNull().unique(),
    alpacaAccountId: text("alpaca_account_id").unique(),
    aliasLastChangedAt: timestamp("alias_last_changed_at", { withTimezone: true }),
    isVisible: boolean("is_visible").notNull().default(true),
    tradingMode: text("trading_mode").notNull().default("paper").$type<TradingMode>(),
    createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    aliasIdx: index("idx_leaderboard_settings_alias").on(table.customAlias),
    enabledIdx: index("idx_leaderboard_settings_enabled").on(table.isEnabled, table.isVisible),
  })
)

export const leaderboardSnapshots = pgTable(
  "leaderboard_snapshots",
  {
    id: text("id").primaryKey(),
    userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
    period: text("period").notNull().$type<LeaderboardPeriod>(),
    tradingMode: text("trading_mode").notNull().$type<TradingMode>(),
    periodStart: timestamp("period_start", { withTimezone: true }).notNull(),
    periodEnd: timestamp("period_end", { withTimezone: true }).notNull(),
    roi: real("roi").notNull(),
    sharpe: real("sharpe"),
    winRate: real("win_rate").notNull(),
    maxDrawdown: real("max_drawdown").notNull(),
    tradesCount: integer("trades_count").notNull(),
    winCount: integer("win_count").notNull(),
    lossCount: integer("loss_count").notNull(),
    totalVolume: real("total_volume").notNull(),
    avgHoldTimeHours: real("avg_hold_time_hours").notNull(),
    startingEquity: real("starting_equity").notNull(),
    endingEquity: real("ending_equity").notNull(),
    compositeScore: real("composite_score"),
    normalizedRoi: real("normalized_roi"),
    normalizedSharpe: real("normalized_sharpe"),
    normalizedWinRate: real("normalized_win_rate"),
    normalizedMaxDrawdown: real("normalized_max_drawdown"),
    rank: integer("rank"),
    totalParticipants: integer("total_participants"),
    meetsMinimumRequirements: boolean("meets_minimum_requirements").notNull().default(false),
    snapshotAt: timestamp("snapshot_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    userPeriodIdx: index("idx_snapshots_user_period").on(table.userId, table.period, table.tradingMode),
    periodRankIdx: index("idx_snapshots_period_rank").on(table.period, table.tradingMode, table.rank),
    qualifiedIdx: index("idx_snapshots_qualified").on(table.period, table.tradingMode, table.meetsMinimumRequirements),
  })
)

export const leaderboardRanks = pgTable(
  "leaderboard_ranks",
  {
    userId: text("user_id").notNull(),
    period: text("period").notNull().$type<LeaderboardPeriod>(),
    tradingMode: text("trading_mode").notNull().$type<TradingMode>(),
    rank: integer("rank").notNull(),
    compositeScore: real("composite_score").notNull(),
    alias: text("alias").notNull(),
    roi: real("roi").notNull(),
    winRate: real("win_rate").notNull(),
    sharpe: real("sharpe"),
    tradesCount: integer("trades_count").notNull(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    pk: primaryKey({ columns: [table.period, table.tradingMode, table.rank] }),
    userIdx: index("idx_ranks_user").on(table.userId, table.period, table.tradingMode),
    aliasIdx: index("idx_ranks_alias").on(table.alias),
  })
)

export type LeaderboardSettingsRow = typeof leaderboardSettings.$inferSelect
export type LeaderboardSnapshotRow = typeof leaderboardSnapshots.$inferSelect
export type LeaderboardRankRow = typeof leaderboardRanks.$inferSelect
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/leaderboard.ts

import { z } from "zod"

export const LeaderboardPeriodSchema = z.enum(["24h", "7d", "30d", "90d", "all"])
export const TradingModeSchema = z.enum(["paper", "live"])

export const LeaderboardSettingsSchema = z.object({
  userId: z.string(),
  isEnabled: z.boolean(),
  customAlias: z.string().nullable(),
  generatedAlias: z.string(),
  alpacaAccountId: z.string(),
  aliasLastChangedAt: z.string().nullable(),
  isVisible: z.boolean(),
  tradingMode: TradingModeSchema,
  createdAt: z.string(),
  updatedAt: z.string(),
})

export const UpdateSettingsRequestSchema = z.object({
  isEnabled: z.boolean().optional(),
  customAlias: z
    .string()
    .min(3)
    .max(20)
    .regex(/^[a-zA-Z0-9_]+$/, "Only alphanumeric and underscore allowed")
    .nullable()
    .optional(),
  isVisible: z.boolean().optional(),
  tradingMode: TradingModeSchema.optional(),
})

export const LeaderboardEntrySchema = z.object({
  rank: z.number(),
  userId: z.string(),
  alias: z.string(),
  compositeScore: z.number(),
  roi: z.number(),
  winRate: z.number(),
  sharpe: z.number().nullable(),
  tradesCount: z.number(),
  isCurrentUser: z.boolean().optional(),
})

export const LeaderboardQuerySchema = z.object({
  period: LeaderboardPeriodSchema.default("30d"),
  mode: TradingModeSchema.default("paper"),
  page: z.coerce.number().min(1).default(1),
  limit: z.coerce.number().min(1).max(100).default(50),
  search: z.string().optional(),
})

export const LeaderboardProfileSchema = z.object({
  alias: z.string(),
  ranks: z.record(
    LeaderboardPeriodSchema,
    z
      .object({
        rank: z.number(),
        totalParticipants: z.number(),
        compositeScore: z.number(),
        roi: z.number(),
        winRate: z.number(),
        sharpe: z.number().nullable(),
        tradesCount: z.number(),
      })
      .nullable()
  ),
  rankHistory: z.array(
    z.object({
      date: z.string(),
      period: LeaderboardPeriodSchema,
      rank: z.number(),
    })
  ),
})

export type LeaderboardPeriod = z.infer<typeof LeaderboardPeriodSchema>
export type TradingMode = z.infer<typeof TradingModeSchema>
export type LeaderboardSettings = z.infer<typeof LeaderboardSettingsSchema>
export type LeaderboardEntry = z.infer<typeof LeaderboardEntrySchema>
export type LeaderboardProfile = z.infer<typeof LeaderboardProfileSchema>
```
