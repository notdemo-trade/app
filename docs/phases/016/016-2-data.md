## Database Schema

Database schema defined in Phase 11 (`notification_settings` table). Phase 12 adds notification logging:

```sql
-- packages/data-ops/src/db/migrations/0012_notification_log.sql
-- notification_settings table already created in Phase 11

CREATE TABLE notification_log (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES "user"(id) ON DELETE CASCADE,
  channel TEXT NOT NULL DEFAULT 'telegram',
  event_type TEXT NOT NULL,
  payload_json JSONB NOT NULL,
  status TEXT NOT NULL,
  error_message TEXT,
  sent_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_notif_log_user ON notification_log(user_id);
CREATE INDEX idx_notif_log_sent ON notification_log(user_id, sent_at DESC);
```

## Drizzle Schema

```ts
// packages/data-ops/src/db/schema/notifications.ts
// notification_settings defined in Phase 11
// This schema adds notification logging

import { pgTable, text, timestamp, jsonb, index } from "drizzle-orm/pg-core"
import { user } from "./auth"
import type { NotificationEventType, NotificationPayload } from "../../types/notifications"

export const notificationLog = pgTable(
  "notification_log",
  {
    id: text("id").primaryKey(),
    userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
    channel: text("channel").notNull().default("telegram"),
    eventType: text("event_type").notNull().$type<NotificationEventType>(),
    payloadJson: jsonb("payload_json").$type<NotificationPayload>().notNull(),
    status: text("status").notNull().$type<"sent" | "failed" | "cooldown">(),
    errorMessage: text("error_message"),
    sentAt: timestamp("sent_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    userIdx: index("idx_notif_log_user").on(table.userId),
    sentIdx: index("idx_notif_log_sent").on(table.userId, table.sentAt),
  })
)

export type NotificationConfigRow = typeof notificationConfig.$inferSelect
export type NotificationLogRow = typeof notificationLog.$inferSelect
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/notifications.ts

import { z } from "zod"

export const NotificationEventTypeSchema = z.enum([
  "trade_executed",
  "stop_loss_triggered",
  "take_profit_triggered",
  "kill_switch_activated",
  "daily_loss_limit_hit",
  "agent_error",
  "daily_summary",
])

export const NotificationChannelSchema = z.enum(["telegram"])

export const NotificationConfigSchema = z.object({
  id: z.string(),
  userId: z.string(),
  channel: NotificationChannelSchema,
  webhookUrl: z.string().nullable(),
  phoneNumber: z.string().nullable(),
  enabled: z.boolean(),
  enabledEvents: z.array(NotificationEventTypeSchema),
  cooldownMins: z.number(),
  createdAt: z.string(),
  updatedAt: z.string(),
})

export const UpdateNotificationConfigSchema = z.object({
  webhookUrl: z.string().url().nullable().optional(),
  enabled: z.boolean().optional(),
  enabledEvents: z.array(NotificationEventTypeSchema).optional(),
  cooldownMins: z.number().min(0).max(60).optional(),
})

export const TestWebhookSchema = z.object({
  webhookUrl: z.string().url(),
})

export const NotificationLogSchema = z.object({
  id: z.string(),
  userId: z.string(),
  channel: NotificationChannelSchema,
  eventType: NotificationEventTypeSchema,
  payload: z.object({
    title: z.string(),
    description: z.string(),
    color: z.enum(["success", "warning", "error", "info"]),
    fields: z.array(z.object({
      name: z.string(),
      value: z.string(),
      inline: z.boolean().optional(),
    })),
    symbol: z.string().optional(),
    timestamp: z.string(),
  }),
  status: z.enum(["sent", "failed", "cooldown"]),
  errorMessage: z.string().nullable(),
  sentAt: z.string(),
})

export type NotificationEventType = z.infer<typeof NotificationEventTypeSchema>
export type NotificationConfig = z.infer<typeof NotificationConfigSchema>
export type NotificationLog = z.infer<typeof NotificationLogSchema>
```
