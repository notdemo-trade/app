## Rate Limiting (P0 Blocker)

Prevent notification test spam.

### Middleware Implementation

```ts
// apps/data-service/src/hono/middleware/rate-limit.ts

import { rateLimiter } from 'hono-rate-limiter'

export const notificationTestLimit = rateLimiter({
  windowMs: 60_000,
  limit: 3,
  keyGenerator: (c) => c.get('userId') || c.req.header('cf-connecting-ip') || 'anonymous'
})
```

### Apply to Test Notification Endpoint

```ts
// apps/data-service/src/hono/handlers/notification-handlers.ts

import { notificationTestLimit } from '../middleware/rate-limit'

notifications.post('/test', sessionMiddleware, notificationTestLimit, testNotificationHandler)
```

**Limits**:
- Notification test: 3 req/min per user

**Why**: Test sends real Telegram messages. Low limit prevents user/admin notification spam while allowing setup verification.

## API Handlers

```ts
// apps/data-service/src/hono/handlers/notification-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import { notificationTestLimit } from "../middleware/rate-limit"
import { UpdateNotificationConfigSchema, TestWebhookSchema } from "@repo/data-ops/zod-schema/notifications"
import { getNotificationConfig, upsertNotificationConfig, getNotificationHistory } from "@repo/data-ops/db/queries/notifications"

const notificationRouter = new Hono<{ Bindings: Env }>()

// Get config
notificationRouter.get(
  "/config",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const config = await getNotificationConfig(c.get("db"), userId)
    return c.json({ data: config })
  }
)

// Update config
notificationRouter.patch(
  "/config",
  apiTokenMiddleware,
  zValidator("json", UpdateNotificationConfigSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    await upsertNotificationConfig(c.get("db"), userId, data)
    return c.json({ success: true })
  }
)

// Test notification - uses Telegram via Phase 11 service
// See telegram-webhook-handlers.ts POST /api/telegram/test

// Get history
notificationRouter.get(
  "/history",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const limit = Number(c.req.query("limit") || 50)

    const history = await getNotificationHistory(c.get("db"), userId, Math.min(limit, 100))
    return c.json({ data: history })
  }
)

export { notificationRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/notifications/config` | Bearer | Get notification config |
| PATCH | `/api/notifications/config` | Bearer | Update config |
| POST | `/api/notifications/test` | Bearer | Test webhook |
| GET | `/api/notifications/history` | Bearer | Get notification history |

## Server Functions

```ts
// apps/user-application/src/core/functions/notifications/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { UpdateNotificationConfigSchema, TestWebhookSchema } from "@repo/data-ops/zod-schema/notifications"
import { getNotificationConfig, upsertNotificationConfig, getNotificationHistory } from "@repo/data-ops/db/queries/notifications"
import { z } from "zod"

export const getNotifConfig = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return getNotificationConfig(context.db, context.session.user.id)
  })

export const updateNotifConfig = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(UpdateNotificationConfigSchema)
  .handler(async ({ data, context }) => {
    await upsertNotificationConfig(context.db, context.session.user.id, data)
    return { success: true }
  })

// testWebhook defined in Phase 11 as sendTestMessage
// Uses TelegramService to send test notification

export const getNotifHistory = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ limit: z.number().optional() }))
  .handler(async ({ data, context }) => {
    return getNotificationHistory(context.db, context.session.user.id, data.limit)
  })
```
