## API Handlers

```ts
// apps/data-service/src/hono/handlers/llm-analysis-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import { AnalyzeRequestSchema, GetAnalysesRequestSchema } from "@repo/data-ops/zod-schema/llm-analysis"
import { createLLMProviderForUser } from "@repo/data-ops/providers/llm/factory"
import { runFullAnalysis } from "@repo/data-ops/services/llm-analysis-service"
import { analyzeSymbol } from "@repo/data-ops/services/analysis-service"
import { getAlpacaClientForUser, getLLMCredentialsForUser } from "../utils/user-credentials"
import { getAnalyses, getAnalysisById, insertAnalysis, updateUsage, getUsageSummary } from "@repo/data-ops/db/queries/llm-analysis"
import { z } from "zod"

const llmRouter = new Hono<{ Bindings: Env }>()

llmRouter.post(
  "/analyze",
  apiTokenMiddleware,
  zValidator("json", AnalyzeRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { symbol, includeResearch } = c.req.valid("json")

    // Get user's LLM credentials
    const llmCreds = await getLLMCredentialsForUser(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!llmCreds) {
      return c.json({ error: "LLM credentials not configured" }, 400)
    }

    // Get user's Alpaca client for market data
    const alpacaClient = await getAlpacaClientForUser(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!alpacaClient) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      // Get technicals and signals
      const taResult = await analyzeSymbol({ alpacaClient }, symbol)

      // Build context
      const context = {
        symbol: symbol.toUpperCase(),
        technicals: taResult,
        signals: taResult.signals,
        price: { current: taResult.indicators.price, change_pct: 0 },
      }

      // Create LLM provider
      const llm = createLLMProviderForUser(llmCreds)

      // Run analysis
      const result = await runFullAnalysis(llm, context, {
        includeResearch,
        model: llmCreds.model,
        provider: llmCreds.provider,
        userId,
      })

      // Store analysis
      await insertAnalysis(c.get("db"), {
        userId,
        symbol: result.symbol,
        action: result.recommendation.action,
        confidence: result.recommendation.confidence,
        rationale: result.recommendation.rationale,
        entryPrice: result.recommendation.entry_price,
        targetPrice: result.recommendation.target_price,
        stopLoss: result.recommendation.stop_loss,
        positionSizePct: result.recommendation.position_size_pct,
        timeframe: result.recommendation.timeframe,
        risks: result.recommendation.risks,
        research: result.research,
        technicals: result.technicals,
        signals: result.signals,
        model: result.model,
        provider: result.provider,
        promptTokens: result.usage.prompt_tokens,
        completionTokens: result.usage.completion_tokens,
        totalTokens: result.usage.total_tokens,
        estimatedCostUsd: result.usage.estimated_cost_usd,
      })

      // Update usage tracking
      await updateUsage(c.get("db"), userId, result.provider, result.model, {
        promptTokens: result.usage.prompt_tokens,
        completionTokens: result.usage.completion_tokens,
        totalTokens: result.usage.total_tokens,
        estimatedCostUsd: result.usage.estimated_cost_usd,
      })

      return c.json({ data: result })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

llmRouter.get(
  "/analyses",
  apiTokenMiddleware,
  zValidator("query", GetAnalysesRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { symbol, limit } = c.req.valid("query")

    const analyses = await getAnalyses(c.get("db"), userId, { symbol, limit })
    return c.json({ data: analyses })
  }
)

llmRouter.get(
  "/analyses/:id",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")

    const analysis = await getAnalysisById(c.get("db"), userId, id)
    if (!analysis) {
      return c.json({ error: "Analysis not found" }, 404)
    }

    return c.json({ data: analysis })
  }
)

llmRouter.get(
  "/usage",
  apiTokenMiddleware,
  zValidator("query", z.object({ days: z.coerce.number().min(1).max(90).optional().default(30) })),
  async (c) => {
    const userId = c.get("userId")
    const { days } = c.req.valid("query")

    const summary = await getUsageSummary(c.get("db"), userId, days)
    return c.json({ data: summary })
  }
)

export { llmRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| POST | `/api/llm/analyze` | Bearer | Run AI analysis on symbol |
| GET | `/api/llm/analyses` | Bearer | List analysis history |
| GET | `/api/llm/analyses/:id` | Bearer | Get single analysis |
| GET | `/api/llm/usage` | Bearer | Get usage summary |

## Server Functions

```ts
// apps/user-application/src/core/functions/llm/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { AnalyzeRequestSchema, GetAnalysesRequestSchema } from "@repo/data-ops/zod-schema/llm-analysis"
import { createLLMProviderForUser } from "@repo/data-ops/providers/llm/factory"
import { runFullAnalysis } from "@repo/data-ops/services/llm-analysis-service"
import { analyzeSymbol } from "@repo/data-ops/services/analysis-service"
import { getAlpacaClientForUser, getLLMCredentialsForUser } from "@repo/data-ops/utils/user-credentials"
import { getAnalyses, insertAnalysis, updateUsage, getUsageSummary } from "@repo/data-ops/db/queries/llm-analysis"
import { z } from "zod"

export const runAnalysis = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(AnalyzeRequestSchema)
  .handler(async ({ data, context }) => {
    const llmCreds = await getLLMCredentialsForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )
    if (!llmCreds) {
      throw new Error("LLM credentials not configured")
    }

    const alpacaClient = await getAlpacaClientForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )
    if (!alpacaClient) {
      throw new Error("Alpaca credentials not configured")
    }

    const taResult = await analyzeSymbol({ alpacaClient }, data.symbol)

    const llm = createLLMProviderForUser(llmCreds)
    const result = await runFullAnalysis(
      llm,
      {
        symbol: data.symbol.toUpperCase(),
        technicals: taResult,
        signals: taResult.signals,
        price: { current: taResult.indicators.price, change_pct: 0 },
      },
      {
        includeResearch: data.includeResearch,
        model: llmCreds.model,
        provider: llmCreds.provider,
        userId: context.session.user.id,
      }
    )

    await insertAnalysis(context.db, {
      userId: context.session.user.id,
      symbol: result.symbol,
      action: result.recommendation.action,
      confidence: result.recommendation.confidence,
      rationale: result.recommendation.rationale,
      entryPrice: result.recommendation.entry_price,
      targetPrice: result.recommendation.target_price,
      stopLoss: result.recommendation.stop_loss,
      positionSizePct: result.recommendation.position_size_pct,
      timeframe: result.recommendation.timeframe,
      risks: result.recommendation.risks,
      research: result.research,
      technicals: result.technicals,
      signals: result.signals,
      model: result.model,
      provider: result.provider,
      promptTokens: result.usage.prompt_tokens,
      completionTokens: result.usage.completion_tokens,
      totalTokens: result.usage.total_tokens,
      estimatedCostUsd: result.usage.estimated_cost_usd,
    })

    await updateUsage(context.db, context.session.user.id, result.provider, result.model, {
      promptTokens: result.usage.prompt_tokens,
      completionTokens: result.usage.completion_tokens,
      totalTokens: result.usage.total_tokens,
      estimatedCostUsd: result.usage.estimated_cost_usd,
    })

    return result
  })

export const getAnalysisHistory = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(GetAnalysesRequestSchema)
  .handler(async ({ data, context }) => {
    return getAnalyses(context.db, context.session.user.id, {
      symbol: data.symbol,
      limit: data.limit,
    })
  })

export const getLLMUsage = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ days: z.number().min(1).max(90).optional().default(30) }))
  .handler(async ({ data, context }) => {
    return getUsageSummary(context.db, context.session.user.id, data.days)
  })
```

## Rate Limiting (P0 Blocker)

Control LLM API costs and prevent abuse.

### Middleware Implementation

```ts
// apps/data-service/src/hono/middleware/rate-limit.ts

import { rateLimiter } from 'hono-rate-limiter'

export const llmAnalysisLimit = rateLimiter({
  windowMs: 60_000,
  limit: 10,
  keyGenerator: (c) => c.get('userId') || c.req.header('cf-connecting-ip') || 'anonymous'
})
```

### Apply to LLM Endpoints

```ts
// apps/data-service/src/hono/handlers/llm-analysis-handlers.ts

import { llmAnalysisLimit } from '../middleware/rate-limit'

llmRouter.post('/analyze', apiTokenMiddleware, llmAnalysisLimit, zValidator('json', AnalyzeRequestSchema), analyzeHandler)
```

**Limits**:
- LLM analysis: 10 req/min per user

**Why**: Each analysis costs money (LLM API tokens). 10/min allows ~600 analyses/hour while preventing runaway costs from bugs or abuse.
