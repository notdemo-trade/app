## Query Hooks

```ts
// apps/user-application/src/lib/llm-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import { runAnalysis, getAnalysisHistory, getLLMUsage } from "@/core/functions/llm/direct"
import type { AnalyzeRequest } from "@repo/data-ops/zod-schema/llm-analysis"

export const analysisHistoryQueryOptions = (symbol?: string, limit?: number) =>
  queryOptions({
    queryKey: ["llm", "analyses", { symbol, limit }],
    queryFn: () => getAnalysisHistory({ symbol, limit }),
    staleTime: 30_000,
  })

export const llmUsageQueryOptions = (days: number = 30) =>
  queryOptions({
    queryKey: ["llm", "usage", days],
    queryFn: () => getLLMUsage({ days }),
    staleTime: 60_000,
  })

export const useAnalysisHistory = (symbol?: string, limit?: number) =>
  useQuery(analysisHistoryQueryOptions(symbol, limit))

export const useLLMUsage = (days?: number) => useQuery(llmUsageQueryOptions(days))

export const useRunAnalysis = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: (data: AnalyzeRequest) => runAnalysis(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["llm", "analyses"] })
      queryClient.invalidateQueries({ queryKey: ["llm", "usage"] })
    },
  })
}
```

## UI Components

### Recommendation Card

```tsx
// apps/user-application/src/components/llm/recommendation-card.tsx

import type { TradeRecommendation } from "@repo/data-ops/zod-schema/llm-analysis"
import { Badge } from "@/components/ui/badge"
import { cn } from "@/lib/utils"

interface RecommendationCardProps {
  recommendation: TradeRecommendation
  symbol: string
}

export function RecommendationCard({ recommendation, symbol }: RecommendationCardProps) {
  const actionColors = {
    buy: "bg-green-500/10 text-green-600 border-green-500/30",
    sell: "bg-red-500/10 text-red-600 border-red-500/30",
    hold: "bg-yellow-500/10 text-yellow-600 border-yellow-500/30",
  }

  return (
    <div className="rounded-lg border p-4 space-y-4">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-3">
          <span className="text-2xl font-bold">{symbol}</span>
          <Badge className={cn("text-lg px-3 py-1", actionColors[recommendation.action])}>
            {recommendation.action.toUpperCase()}
          </Badge>
        </div>
        <div className="text-right">
          <div className="text-sm text-muted-foreground">Confidence</div>
          <div className="text-xl font-mono">{(recommendation.confidence * 100).toFixed(0)}%</div>
        </div>
      </div>

      <p className="text-sm">{recommendation.rationale}</p>

      {(recommendation.entry_price || recommendation.target_price || recommendation.stop_loss) && (
        <div className="grid grid-cols-3 gap-4 text-sm">
          {recommendation.entry_price && (
            <div>
              <div className="text-muted-foreground">Entry</div>
              <div className="font-mono">${recommendation.entry_price.toFixed(2)}</div>
            </div>
          )}
          {recommendation.target_price && (
            <div>
              <div className="text-muted-foreground">Target</div>
              <div className="font-mono text-green-600">${recommendation.target_price.toFixed(2)}</div>
            </div>
          )}
          {recommendation.stop_loss && (
            <div>
              <div className="text-muted-foreground">Stop</div>
              <div className="font-mono text-red-600">${recommendation.stop_loss.toFixed(2)}</div>
            </div>
          )}
        </div>
      )}

      {recommendation.risks.length > 0 && (
        <div>
          <div className="text-sm text-muted-foreground mb-1">Risks</div>
          <ul className="text-sm list-disc list-inside space-y-1">
            {recommendation.risks.map((risk, i) => (
              <li key={i}>{risk}</li>
            ))}
          </ul>
        </div>
      )}

      <div className="flex gap-2 text-xs text-muted-foreground">
        {recommendation.timeframe && <Badge variant="outline">{recommendation.timeframe}</Badge>}
        {recommendation.position_size_pct && (
          <Badge variant="outline">{recommendation.position_size_pct}% position</Badge>
        )}
      </div>
    </div>
  )
}
```

### Usage Dashboard

```tsx
// apps/user-application/src/components/llm/usage-dashboard.tsx

import { useLLMUsage } from "@/lib/llm-queries"
```

### Analysis History

```tsx
// apps/user-application/src/components/llm/analysis-history.tsx

import { useAnalysisHistory } from "@/lib/llm-queries"
import { Badge } from "@/components/ui/badge"
import { formatDistanceToNow } from "date-fns"
import { cn } from "@/lib/utils"
import Link from "next/link"

interface AnalysisHistoryProps {
  symbol?: string
}

export function AnalysisHistory({ symbol }: AnalysisHistoryProps) {
  const { data, isLoading } = useAnalysisHistory(symbol, 20)

  if (isLoading) {
    return <div className="animate-pulse">Loading history...</div>
  }

  if (!data?.length) {
    return <p className="text-sm text-muted-foreground">No analyses yet</p>
  }

  return (
    <div className="space-y-2">
      {data.map((analysis) => (
        <Link
          key={analysis.id}
          href={`/analysis/${analysis.id}`}
          className="block p-3 rounded border hover:bg-accent transition-colors"
        >
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-2">
              <span className="font-mono font-medium">{analysis.symbol}</span>
              <Badge
                variant="outline"
                className={cn(
                  analysis.action === "buy" && "border-green-500 text-green-600",
                  analysis.action === "sell" && "border-red-500 text-red-600",
                  analysis.action === "hold" && "border-yellow-500 text-yellow-600"
                )}
              >
                {analysis.action}
              </Badge>
              <span className="text-sm text-muted-foreground">
                {(analysis.confidence * 100).toFixed(0)}%
              </span>
            </div>
            <span className="text-xs text-muted-foreground">
              {formatDistanceToNow(new Date(analysis.createdAt), { addSuffix: true })}
            </span>
          </div>
          <p className="text-sm text-muted-foreground mt-1 line-clamp-1">{analysis.rationale}</p>
        </Link>
      ))}
    </div>
  )
}
```

### Analysis Page

```tsx
// apps/user-application/src/routes/_auth/llm/index.tsx

import { createFileRoute } from "@tanstack/react-router"
import { useState } from "react"
import { useRunAnalysis } from "@/lib/llm-queries"
import { RecommendationCard } from "@/components/llm/recommendation-card"
import { AnalysisHistory } from "@/components/llm/analysis-history"
import { UsageDashboard } from "@/components/llm/usage-dashboard"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Checkbox } from "@/components/ui/checkbox"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs"

export const Route = createFileRoute("/_auth/llm/")({
  component: LLMAnalysisPage,
})

function LLMAnalysisPage() {
  const [symbol, setSymbol] = useState("")
  const [includeResearch, setIncludeResearch] = useState(false)
  const { mutate, isPending, data, error } = useRunAnalysis()

  const handleAnalyze = () => {
    if (symbol.trim()) {
      mutate({ symbol: symbol.trim().toUpperCase(), includeResearch })
    }
  }

  return (
    <div className="space-y-6">
      <h1 className="text-2xl font-bold">AI Analysis</h1>

      <div className="flex gap-4">
        <Input
          placeholder="Enter symbol (e.g., AAPL)"
          value={symbol}
          onChange={(e) => setSymbol(e.target.value.toUpperCase())}
          className="max-w-xs"
          onKeyDown={(e) => e.key === "Enter" && handleAnalyze()}
        />
        <div className="flex items-center gap-2">
          <Checkbox
            id="research"
            checked={includeResearch}
            onCheckedChange={(c) => setIncludeResearch(!!c)}
          />
          <label htmlFor="research" className="text-sm">
            Include research report
          </label>
        </div>
        <Button onClick={handleAnalyze} disabled={isPending || !symbol.trim()}>
          {isPending ? "Analyzing..." : "Analyze"}
        </Button>
      </div>

      {error && <p className="text-destructive text-sm">{String(error)}</p>}

      {data && (
        <div className="space-y-4">
          <RecommendationCard recommendation={data.recommendation} symbol={data.symbol} />

          {data.usage && (
            <div className="text-xs text-muted-foreground">
              Tokens: {data.usage.total_tokens.toLocaleString()} | Est. cost: $
              {data.usage.estimated_cost_usd.toFixed(4)} | Model: {data.model}
            </div>
          )}

          {data.research && (
            <div className="prose prose-sm dark:prose-invert max-w-none">
              <div dangerouslySetInnerHTML={{ __html: data.research }} />
            </div>
          )}
        </div>
      )}

      <Tabs defaultValue="history">
        <TabsList>
          <TabsTrigger value="history">History</TabsTrigger>
          <TabsTrigger value="usage">Usage</TabsTrigger>
        </TabsList>
        <TabsContent value="history">
          <AnalysisHistory />
        </TabsContent>
        <TabsContent value="usage">
          <UsageDashboard />
        </TabsContent>
      </Tabs>
    </div>
  )
}
```
