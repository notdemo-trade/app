## Query Hooks

```ts
// apps/user-application/src/lib/analysis-queries.ts

import { queryOptions, useQuery } from "@tanstack/react-query"
import { getAnalysis, getBatchAnalysis } from "@/core/functions/analysis/direct"
import type { Timeframe } from "@repo/data-ops/zod-schema/analysis"

export const analysisQueryOptions = (symbol: string, timeframe: Timeframe = "1Day") =>
  queryOptions({
    queryKey: ["analysis", symbol, timeframe],
    queryFn: () => getAnalysis({ symbol, timeframe }),
    staleTime: 60_000, // 1 min
  })

export const batchAnalysisQueryOptions = (symbols: string[], timeframe: Timeframe = "1Day") =>
  queryOptions({
    queryKey: ["analysis", "batch", symbols, timeframe],
    queryFn: () => getBatchAnalysis({ symbols, timeframe }),
    staleTime: 60_000,
    enabled: symbols.length > 0,
  })

export const useAnalysis = (symbol: string, timeframe?: Timeframe) =>
  useQuery(analysisQueryOptions(symbol, timeframe))

export const useBatchAnalysis = (symbols: string[], timeframe?: Timeframe) =>
  useQuery(batchAnalysisQueryOptions(symbols, timeframe))
```

## UI Components

### Price Chart

Using `lightweight-charts` for performance.

```tsx
// apps/user-application/src/components/analysis/price-chart.tsx

import { useEffect, useRef } from "react"
import { createChart, IChartApi, ISeriesApi } from "lightweight-charts"
import type { Bar, TechnicalIndicators } from "@repo/data-ops/zod-schema/analysis"

interface PriceChartProps {
  bars: Bar[]
  indicators: TechnicalIndicators
  height?: number
}

export function PriceChart({ bars, indicators, height = 400 }: PriceChartProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const chartRef = useRef<IChartApi | null>(null)

  useEffect(() => {
    if (!containerRef.current) return

    const chart = createChart(containerRef.current, {
      height,
      layout: {
        background: { color: "transparent" },
        textColor: "#9ca3af",
      },
      grid: {
        vertLines: { color: "#374151" },
        horzLines: { color: "#374151" },
      },
      timeScale: { timeVisible: true },
    })

    chartRef.current = chart

    // Candlestick series
    const candleSeries = chart.addCandlestickSeries({
      upColor: "#22c55e",
      downColor: "#ef4444",
      borderVisible: false,
      wickUpColor: "#22c55e",
      wickDownColor: "#ef4444",
    })

    candleSeries.setData(
      bars.map(b => ({
        time: new Date(b.t).getTime() / 1000,
        open: b.o,
        high: b.h,
        low: b.l,
        close: b.c,
      }))
    )

    // SMA lines
    if (indicators.sma_20 !== null) {
      addSMALine(chart, bars, 20, "#3b82f6")
    }
    if (indicators.sma_50 !== null) {
      addSMALine(chart, bars, 50, "#f59e0b")
    }

    // Bollinger Bands
    if (indicators.bollinger) {
      addBollingerBands(chart, bars, indicators.bollinger)
    }

    chart.timeScale().fitContent()

    return () => {
      chart.remove()
    }
  }, [bars, indicators, height])

  return <div ref={containerRef} className="w-full" />
}

function addSMALine(chart: IChartApi, bars: Bar[], period: number, color: string) {
  const closes = bars.map(b => b.c)
  const smaData: { time: number; value: number }[] = []

  for (let i = period - 1; i < bars.length; i++) {
    const slice = closes.slice(i - period + 1, i + 1)
    const sma = slice.reduce((a, b) => a + b, 0) / period
    smaData.push({
      time: new Date(bars[i].t).getTime() / 1000,
      value: sma,
    })
  }

  const series = chart.addLineSeries({ color, lineWidth: 1 })
  series.setData(smaData)
}

function addBollingerBands(
  chart: IChartApi,
  bars: Bar[],
  bb: { upper: number; middle: number; lower: number }
) {
  // Simplified: just show current bands as horizontal lines
  // Full implementation would calculate bands for each bar
  const upperSeries = chart.addLineSeries({ color: "#6366f1", lineWidth: 1, lineStyle: 2 })
  const lowerSeries = chart.addLineSeries({ color: "#6366f1", lineWidth: 1, lineStyle: 2 })

  // For MVP, show static lines at current values
  const lastTime = new Date(bars[bars.length - 1].t).getTime() / 1000
  const firstTime = new Date(bars[0].t).getTime() / 1000

  upperSeries.setData([
    { time: firstTime, value: bb.upper },
    { time: lastTime, value: bb.upper },
  ])
  lowerSeries.setData([
    { time: firstTime, value: bb.lower },
    { time: lastTime, value: bb.lower },
  ])
}
```

### Indicator Panels

```tsx
// apps/user-application/src/components/analysis/indicator-panels.tsx

import type { TechnicalIndicators } from "@repo/data-ops/zod-schema/analysis"
import { cn } from "@/lib/utils"

interface IndicatorPanelsProps {
  indicators: TechnicalIndicators
}

export function IndicatorPanels({ indicators }: IndicatorPanelsProps) {
  return (
    <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
      <IndicatorCard
        label="RSI (14)"
        value={indicators.rsi_14?.toFixed(1) ?? "—"}
        status={getRSIStatus(indicators.rsi_14)}
      />
      <IndicatorCard
        label="MACD"
        value={indicators.macd?.histogram.toFixed(3) ?? "—"}
        status={getMACDStatus(indicators.macd)}
      />
      <IndicatorCard
        label="ATR (14)"
        value={indicators.atr_14?.toFixed(2) ?? "—"}
      />
      <IndicatorCard
        label="Rel. Volume"
        value={indicators.relative_volume?.toFixed(1) + "x" ?? "—"}
        status={getVolumeStatus(indicators.relative_volume)}
      />
    </div>
  )
}

function IndicatorCard({
  label,
  value,
  status,
}: {
  label: string
  value: string
  status?: "bullish" | "bearish" | "neutral"
}) {
  return (
    <div className="p-3 rounded-lg border bg-card">
      <p className="text-xs text-muted-foreground">{label}</p>
      <p
        className={cn(
          "text-lg font-mono font-semibold",
          status === "bullish" && "text-green-500",
          status === "bearish" && "text-red-500"
        )}
      >
        {value}
      </p>
    </div>
  )
}

function getRSIStatus(rsi: number | null): "bullish" | "bearish" | "neutral" | undefined {
  if (rsi === null) return undefined
  if (rsi < 30) return "bullish"
  if (rsi > 70) return "bearish"
  return "neutral"
}

function getMACDStatus(
  macd: { histogram: number } | null
): "bullish" | "bearish" | "neutral" | undefined {
  if (!macd) return undefined
  if (macd.histogram > 0) return "bullish"
  if (macd.histogram < 0) return "bearish"
  return "neutral"
}

function getVolumeStatus(relVol: number | null): "bullish" | "bearish" | "neutral" | undefined {
  if (relVol === null) return undefined
  if (relVol > 2) return "bullish" // High volume = notable
  return "neutral"
}
```

### Signal List

```tsx
// apps/user-application/src/components/analysis/signal-list.tsx

import type { TechnicalSignal } from "@repo/data-ops/zod-schema/analysis"
import { Badge } from "@/components/ui/badge"
import { cn } from "@/lib/utils"

interface SignalListProps {
  signals: TechnicalSignal[]
}

export function SignalList({ signals }: SignalListProps) {
  if (signals.length === 0) {
    return (
      <p className="text-sm text-muted-foreground">No signals detected</p>
    )
  }

  return (
    <div className="space-y-2">
      {signals.map((signal, i) => (
        <div
          key={i}
          className={cn(
            "flex items-center justify-between p-2 rounded border",
            signal.direction === "bullish" && "border-green-500/30 bg-green-500/5",
            signal.direction === "bearish" && "border-red-500/30 bg-red-500/5",
            signal.direction === "neutral" && "border-gray-500/30"
          )}
        >
          <div className="flex items-center gap-2">
            <Badge
              variant={
                signal.direction === "bullish"
                  ? "default"
                  : signal.direction === "bearish"
                    ? "destructive"
                    : "secondary"
              }
            >
              {signal.direction}
            </Badge>
            <span className="text-sm">{signal.description}</span>
          </div>
          <span className="text-xs text-muted-foreground">
            {(signal.strength * 100).toFixed(0)}%
          </span>
        </div>
      ))}
    </div>
  )
}
```

### Analysis Page

```tsx
// apps/user-application/src/routes/_auth/analysis/[symbol].tsx

import { createFileRoute } from "@tanstack/react-router"
import { useAnalysis } from "@/lib/analysis-queries"
import { PriceChart } from "@/components/analysis/price-chart"
import { IndicatorPanels } from "@/components/analysis/indicator-panels"
import { SignalList } from "@/components/analysis/signal-list"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { useState } from "react"
import type { Timeframe } from "@repo/data-ops/zod-schema/analysis"

export const Route = createFileRoute("/_auth/analysis/$symbol")({
  component: AnalysisPage,
})

function AnalysisPage() {
  const { symbol } = Route.useParams()
  const [timeframe, setTimeframe] = useState<Timeframe>("1Day")

  const { data, isLoading, error } = useAnalysis(symbol, timeframe)

  if (isLoading) {
    return <div className="animate-pulse">Loading analysis...</div>
  }

  if (error) {
    return (
      <div className="text-destructive">
        Failed to load analysis: {error.message}
      </div>
    )
  }

  if (!data) return null

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">{symbol}</h1>
          <p className="text-muted-foreground">
            ${data.indicators.price.toFixed(2)}
          </p>
        </div>

        <Select value={timeframe} onValueChange={v => setTimeframe(v as Timeframe)}>
          <SelectTrigger className="w-32">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="1Min">1 Min</SelectItem>
            <SelectItem value="5Min">5 Min</SelectItem>
            <SelectItem value="15Min">15 Min</SelectItem>
            <SelectItem value="1Hour">1 Hour</SelectItem>
            <SelectItem value="1Day">1 Day</SelectItem>
          </SelectContent>
        </Select>
      </div>

      <PriceChart bars={data.bars} indicators={data.indicators} />

      <IndicatorPanels indicators={data.indicators} />

      <div>
        <h2 className="text-lg font-semibold mb-3">Signals</h2>
        <SignalList signals={data.signals} />
      </div>
    </div>
  )
}
```
