## API Handlers

```ts
// apps/data-service/src/hono/handlers/analysis-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import { GetAnalysisRequestSchema, BatchAnalysisRequestSchema } from "@repo/data-ops/zod-schema/analysis"
import { analyzeSymbol, analyzeMultiple, InsufficientDataError } from "@repo/data-ops/services/analysis-service"
import { getAlpacaClientForUser } from "../utils/alpaca"

const analysisRouter = new Hono<{ Bindings: Env }>()

analysisRouter.get(
  "/:symbol",
  apiTokenMiddleware,
  zValidator("param", z.object({ symbol: z.string() })),
  zValidator("query", GetAnalysisRequestSchema.omit({ symbol: true })),
  async (c) => {
    const userId = c.get("userId")
    const symbol = c.req.param("symbol")
    const { timeframe, includeBars } = c.req.valid("query")

    const alpacaClient = await getAlpacaClientForUser(
      c.get("db"),
      userId,
      c.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!alpacaClient) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const result = await analyzeSymbol({ alpacaClient }, symbol, timeframe)

      // Optionally omit bars to reduce payload
      if (!includeBars) {
        return c.json({
          data: {
            ...result,
            bars: undefined,
          },
        })
      }

      return c.json({ data: result })
    } catch (err) {
      if (err instanceof InsufficientDataError) {
        return c.json({ error: err.message }, 400)
      }
      throw err
    }
  }
)

analysisRouter.post(
  "/batch",
  apiTokenMiddleware,
  zValidator("json", BatchAnalysisRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { symbols, timeframe } = c.req.valid("json")

    const alpacaClient = await getAlpacaClientForUser(
      c.get("db"),
      userId,
      c.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!alpacaClient) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    const results = await analyzeMultiple({ alpacaClient }, symbols, timeframe)

    const data: Record<string, unknown> = {}
    for (const [sym, result] of results) {
      if (result instanceof Error) {
        data[sym] = { error: result.message }
      } else {
        data[sym] = {
          indicators: result.indicators,
          signals: result.signals,
        }
      }
    }

    return c.json({ data })
  }
)

export { analysisRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/analysis/:symbol` | Bearer | Get indicators + signals for symbol |
| POST | `/api/analysis/batch` | Bearer | Batch analysis (max 20 symbols) |

Query params for GET:
- `timeframe`: 1Min, 5Min, 15Min, 1Hour, 1Day (default: 1Day)
- `includeBars`: true/false (default: true)

## Server Functions

```ts
// apps/user-application/src/core/functions/analysis/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { GetAnalysisRequestSchema, BatchAnalysisRequestSchema } from "@repo/data-ops/zod-schema/analysis"
import { analyzeSymbol, analyzeMultiple } from "@repo/data-ops/services/analysis-service"
import { getAlpacaClientForUser } from "@repo/data-ops/utils/alpaca"

export const getAnalysis = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(GetAnalysisRequestSchema)
  .handler(async ({ data, context }) => {
    const alpacaClient = await getAlpacaClientForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!alpacaClient) {
      throw new Error("Alpaca credentials not configured")
    }

    return analyzeSymbol({ alpacaClient }, data.symbol, data.timeframe)
  })

export const getBatchAnalysis = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(BatchAnalysisRequestSchema)
  .handler(async ({ data, context }) => {
    const alpacaClient = await getAlpacaClientForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!alpacaClient) {
      throw new Error("Alpaca credentials not configured")
    }

    const results = await analyzeMultiple({ alpacaClient }, data.symbols, data.timeframe)

    // Convert Map to plain object
    const output: Record<string, unknown> = {}
    for (const [sym, result] of results) {
      if (result instanceof Error) {
        output[sym] = { error: result.message }
      } else {
        output[sym] = result
      }
    }

    return output
  })
```
