# Phase 6: LLM Analysis Agent — Part 4: API & Server Functions

## API Handlers

Proxy to LLMAnalysisAgent via `getAgentByName()`.

```ts
// apps/data-service/src/hono/handlers/llm-analysis-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { getAgentByName } from "agents"
import { sessionAuthMiddleware } from "../middleware/session-auth"
import { AnalyzeRequestSchema, GetAnalysesRequestSchema } from "@repo/data-ops/zod-schema/llm-analysis"
import type { LLMAgentRPC } from "@repo/data-ops/agents/llm/types"
import { getAnalyses, getAnalysisById, getUsageSummary } from "@repo/data-ops/queries/llm-analysis"

const llmRouter = new Hono<{ Bindings: Env; Variables: { userId: string } }>()

llmRouter.post("/analyze", sessionAuthMiddleware, llmAnalysisLimit, zValidator("json", AnalyzeRequestSchema), async (c) => {
  const userId = c.get("userId")
  const request = c.req.valid("json")

  const agent = await getAgentByName<LLMAgentRPC>(c.env.LLMAnalysisAgent, userId)
  const result = await agent.analyze(request)
  return c.json({ data: result })
})

llmRouter.get("/analyses", sessionAuthMiddleware, zValidator("query", GetAnalysesRequestSchema), async (c) => {
  const analyses = await getAnalyses(c.get("db"), c.get("userId"), c.req.valid("query"))
  return c.json({ data: analyses })
})

llmRouter.get("/analyses/:id", sessionAuthMiddleware, async (c) => {
  const analysis = await getAnalysisById(c.get("db"), c.get("userId"), c.req.param("id"))
  if (!analysis) return c.json({ error: "Not found" }, 404)
  return c.json({ data: analysis })
})

llmRouter.get("/usage", sessionAuthMiddleware, async (c) => {
  const days = Number(c.req.query("days") ?? 30)
  const summary = await getUsageSummary(c.get("db"), c.get("userId"), days)
  return c.json({ data: summary })
})

// Classify raw event via agent
llmRouter.post("/classify", sessionAuthMiddleware, async (c) => {
  const { content } = await c.req.json()
  const agent = await getAgentByName<LLMAgentRPC>(c.env.LLMAnalysisAgent, c.get("userId"))
  const result = await agent.classifyEvent(content)
  return c.json({ data: result })
})

export { llmRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| POST | `/api/llm/analyze` | Session | Run AI analysis (via agent RPC) |
| GET | `/api/llm/analyses` | Session | Analysis history |
| GET | `/api/llm/analyses/:id` | Session | Single analysis |
| GET | `/api/llm/usage` | Session | Usage summary |
| POST | `/api/llm/classify` | Session | Classify raw event |
