## PriceChart

Candlestick chart with SMA overlays using `lightweight-charts`.

```tsx
// apps/user-application/src/components/analysis/price-chart.tsx

import type { Bar, TechnicalIndicators } from "@repo/data-ops/agents/ta/types"

interface PriceChartProps {
  bars: Bar[]
  indicators: TechnicalIndicators
  height?: number       // default 400
}

// Layout:
// ┌──────────────────────────────────────┐
// │  Candlestick chart area              │
// │  - Green/red candles                 │
// │  - SMA 20 (blue line)               │
// │  - SMA 50 (orange line)             │
// │  - SMA 200 (red line)               │
// │  - Bollinger bands (gray fill)       │
// │  - Volume bars at bottom             │
// ├──────────────────────────────────────┤
// │  Time axis                           │
// └──────────────────────────────────────┘
//
// Uses: createChart(), addCandlestickSeries(), addLineSeries()
// Ref: lightweight-charts docs
// Responsive: useRef + ResizeObserver
```

## IndicatorPanels

Grid of indicator value cards.

```tsx
// apps/user-application/src/components/analysis/indicator-panels.tsx

import type { TechnicalIndicators } from "@repo/data-ops/agents/ta/types"

interface IndicatorPanelsProps {
  indicators: TechnicalIndicators
}

// Layout:
// ┌─────────────┬─────────────┬─────────────┬─────────────┐
// │ RSI (14)    │ MACD        │ ATR (14)    │ Rel Volume  │
// │ ■■■■■░░░░░ │ MACD: 1.23  │ $2.45       │ 1.8x        │
// │ 62.3        │ Signal: 0.8 │             │             │
// │ [neutral]   │ Hist: 0.43  │             │ [■■■■░░]    │
// └─────────────┴─────────────┴─────────────┴─────────────┘
//
// Color coding:
// - RSI < 30: green (oversold), > 70: red (overbought), else: neutral
// - MACD histogram > 0: green, < 0: red
// - Rel Volume > 2x: highlighted

interface RSIPanelProps {
  value: number | null
}

interface MACDPanelProps {
  macd: { macd: number; signal: number; histogram: number } | null
}

interface ATRPanelProps {
  value: number | null
  price: number
}

interface VolumePanelProps {
  relativeVolume: number | null
  volumeSma: number | null
}
```

## Query Hooks

```ts
// apps/user-application/src/lib/analysis-queries.ts

import { queryOptions } from "@tanstack/react-query"
import { getAnalysis, getBatchAnalysis } from "@/core/functions/analysis/direct"
import type { Timeframe } from "@repo/data-ops/agents/ta/types"

export const analysisQueryOptions = (symbol: string, timeframe: Timeframe = "1Day") =>
  queryOptions({
    queryKey: ["analysis", symbol, timeframe],
    queryFn: () => getAnalysis({ data: { symbol, timeframe } }),
    staleTime: 60_000,       // 1 min
    refetchInterval: 60_000, // auto-refresh
  })

export const batchAnalysisQueryOptions = (symbols: string[], timeframe: Timeframe = "1Day") =>
  queryOptions({
    queryKey: ["analysis", "batch", symbols, timeframe],
    queryFn: () => getBatchAnalysis({ data: { symbols, timeframe } }),
    staleTime: 60_000,
    enabled: symbols.length > 0,
  })
```
