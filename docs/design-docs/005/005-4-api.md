# Phase 5: Technical Analysis Agent — Part 4: API & Server Functions

## Auth Middleware

Better Auth session validation. Extracts userId from session, sets on Hono context.

```ts
// apps/data-service/src/hono/middleware/session-auth.ts

import { createMiddleware } from "hono/factory"
import { AppError } from "../types/result"
import { auth } from "@repo/data-ops/auth"

type SessionVariables = { userId: string }

export const sessionAuthMiddleware = createMiddleware<{
  Bindings: Env
  Variables: SessionVariables
}>(async (c, next) => {
  const session = await auth.api.getSession({
    headers: c.req.raw.headers,
  })

  if (!session?.user?.id) {
    throw new AppError("Unauthorized", 401)
  }

  c.set("userId", session.user.id)
  await next()
})
```

## API Handlers

Proxy to TechnicalAnalysisAgent via `getAgentByName()`.

```ts
// apps/data-service/src/hono/handlers/analysis-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { getAgentByName } from "agents"
import { sessionAuthMiddleware } from "../middleware/session-auth"
import { GetAnalysisRequestSchema, BatchAnalysisRequestSchema } from "@repo/data-ops/analysis"
import type { TechnicalAnalysisAgent } from "../../agents/technical-analysis-agent"

const analysisRouter = new Hono<{ Bindings: Env; Variables: { userId: string } }>()

analysisRouter.use("*", sessionAuthMiddleware)

analysisRouter.get("/:symbol", zValidator("query", GetAnalysisRequestSchema.omit({ symbol: true })), async (c) => {
  const userId = c.get("userId")
  const symbol = c.req.param("symbol").toUpperCase()
  const { timeframe } = c.req.valid("query")

  const agent = getAgentByName<TechnicalAnalysisAgent>(c.env.TechnicalAnalysisAgent, `${userId}:${symbol}`)
  const result = await agent.analyze(timeframe)
  return c.json({ data: result })
})

analysisRouter.post("/batch", zValidator("json", BatchAnalysisRequestSchema), async (c) => {
  const userId = c.get("userId")
  const { symbols, timeframe } = c.req.valid("json")

  const results: Record<string, unknown> = {}
  const settled = await Promise.allSettled(
    symbols.map(async (sym) => {
      const agent = getAgentByName<TechnicalAnalysisAgent>(c.env.TechnicalAnalysisAgent, `${userId}:${sym}`)
      const result = await agent.analyze(timeframe)
      return { sym, result }
    })
  )

  for (let i = 0; i < settled.length; i++) {
    const s = settled[i]
    const sym = symbols[i]  // use original array index — rejected reason has no sym
    if (s.status === "fulfilled") {
      results[s.value.sym] = { indicators: s.value.result.indicators, signals: s.value.result.signals }
    } else {
      results[sym] = { error: String(s.reason) }
    }
  }

  return c.json({ data: results })
})

// Get cached indicators without recomputing
analysisRouter.get("/:symbol/indicators", async (c) => {
  const userId = c.get("userId")
  const symbol = c.req.param("symbol").toUpperCase()
  const agent = getAgentByName<TechnicalAnalysisAgent>(c.env.TechnicalAnalysisAgent, `${userId}:${symbol}`)
  const indicators = await agent.getIndicators()
  return c.json({ data: indicators })
})

export { analysisRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/analysis/:symbol` | Session | Analyze symbol (triggers agent computation) |
| POST | `/api/analysis/batch` | Session | Batch analysis (max 20 symbols) |
| GET | `/api/analysis/:symbol/indicators` | Session | Get cached indicators |

## Server Functions

```ts
// apps/user-application/src/core/functions/analysis/direct.ts

import { createServerFn } from "@tanstack/start"
import { z } from "zod"

const API_BASE = import.meta.env.VITE_API_URL

export const getAnalysis = createServerFn({ method: "GET" })
  .validator(z.object({ symbol: z.string(), timeframe: z.string().optional() }))
  .handler(async ({ data, context }) => {
    const params = new URLSearchParams()
    if (data.timeframe) params.set("timeframe", data.timeframe)
    const res = await fetch(`${API_BASE}/api/analysis/${data.symbol}?${params}`, {
      headers: { cookie: context.request.headers.get("cookie") ?? "" },
    })
    if (!res.ok) throw new Error(`Analysis failed: ${res.status}`)
    return res.json()
  })

export const getBatchAnalysis = createServerFn({ method: "GET" })
  .validator(z.object({ symbols: z.array(z.string()), timeframe: z.string().optional() }))
  .handler(async ({ data, context }) => {
    const res = await fetch(`${API_BASE}/api/analysis/batch`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        cookie: context.request.headers.get("cookie") ?? "",
      },
      body: JSON.stringify(data),
    })
    if (!res.ok) throw new Error(`Batch analysis failed: ${res.status}`)
    return res.json()
  })
```
