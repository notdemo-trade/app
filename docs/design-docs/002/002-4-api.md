## API Endpoints

### data-service handlers

```ts
// apps/data-service/src/hono/handlers/credential-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { sessionMiddleware } from "../middleware/auth"
import {
  SaveCredentialRequestSchema,
  DeleteCredentialRequestSchema,
} from "@repo/data-ops/zod-schema/credentials"
import * as credentialService from "../services/credential-service"

const credentials = new Hono<{ Bindings: Env }>()

// List user's credentials (metadata only, no secrets)
credentials.get("/", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const list = await credentialService.listCredentials(c.env, userId)
  return c.json({ data: list })
})

// Save credential (create or update)
credentials.put(
  "/",
  sessionMiddleware,
  zValidator("json", SaveCredentialRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const body = c.req.valid("json")

    const result = await credentialService.saveCredential(c.env, {
      userId,
      provider: body.provider,
      data: body.data,
      validate: body.validate,
    })

    return c.json({ data: result })
  }
)

// Delete credential
credentials.delete(
  "/",
  sessionMiddleware,
  zValidator("json", DeleteCredentialRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { provider } = c.req.valid("json")

    await credentialService.deleteCredential(c.env, { userId, provider })
    return c.json({ success: true })
  }
)

// Validate existing credential
credentials.post("/:provider/validate", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const provider = c.req.param("provider")

  const result = await credentialService.validateCredential(c.env, { userId, provider })
  return c.json({ data: result })
})

export { credentials }
```

```ts
// apps/data-service/src/hono/handlers/trading-config-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { sessionMiddleware } from "../middleware/auth"
import { UpdateTradingConfigRequestSchema } from "@repo/data-ops/zod-schema/trading-config"
import * as tradingConfigService from "../services/trading-config-service"

const tradingConfig = new Hono<{ Bindings: Env }>()

// Get trading config (returns defaults if none set)
tradingConfig.get("/", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const config = await tradingConfigService.getTradingConfig(c.env, userId)
  return c.json({ data: config })
})

// Update trading config
tradingConfig.patch(
  "/",
  sessionMiddleware,
  zValidator("json", UpdateTradingConfigRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const updates = c.req.valid("json")

    const config = await tradingConfigService.updateTradingConfig(c.env, userId, updates)
    return c.json({ data: config })
  }
)

export { tradingConfig }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/credentials` | Session | List credential metadata |
| PUT | `/api/credentials` | Session | Save/update credential |
| DELETE | `/api/credentials` | Session | Delete credential |
| POST | `/api/credentials/:provider/validate` | Session | Re-validate credential |
| GET | `/api/trading-config` | Session | Get trading config |
| PATCH | `/api/trading-config` | Session | Update trading config |

## Server Functions

```ts
// apps/user-application/src/core/functions/credentials/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import * as credentialQueries from "@repo/data-ops/queries/credentials"
import { validateCredentialByProvider } from "@repo/data-ops/services/credential-validation"
import { SaveCredentialRequestSchema, DeleteCredentialRequestSchema } from "@repo/data-ops/zod-schema/credentials"

export const listCredentials = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return credentialQueries.listCredentials(context.db, context.session.user.id)
  })

export const saveCredential = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(SaveCredentialRequestSchema)
  .handler(async ({ data, context }) => {
    const { provider, data: credData, validate } = data

    await credentialQueries.saveCredential(context.db, {
      userId: context.session.user.id,
      provider,
      data: credData,
      masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
    })

    if (validate) {
      const result = await validateCredentialByProvider(provider, credData)
      await credentialQueries.updateValidationStatus(context.db, {
        userId: context.session.user.id,
        provider,
        success: result.success,
        error: result.error,
      })
      return result
    }

    return { success: true }
  })

export const deleteCredential = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(DeleteCredentialRequestSchema)
  .handler(async ({ data, context }) => {
    return credentialQueries.deleteCredential(context.db, {
      userId: context.session.user.id,
      provider: data.provider,
    })
  })
```

```ts
// apps/user-application/src/core/functions/trading-config/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import * as configQueries from "@repo/data-ops/queries/trading-config"
import { UpdateTradingConfigRequestSchema, TradingConfigSchema } from "@repo/data-ops/zod-schema/trading-config"

export const getTradingConfig = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    const config = await configQueries.getTradingConfig(context.db, context.session.user.id)
    // Return defaults merged with user config
    return TradingConfigSchema.parse(config ?? {})
  })

export const updateTradingConfig = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(UpdateTradingConfigRequestSchema)
  .handler(async ({ data, context }) => {
    return configQueries.upsertTradingConfig(context.db, context.session.user.id, data)
  })
```
