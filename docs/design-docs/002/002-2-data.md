## Database Schema

### user_credentials table

```ts
// packages/data-ops/src/drizzle/schema.ts

import { pgTable, text, timestamp, uuid, pgEnum, integer, real, boolean } from "drizzle-orm/pg-core"
import { unique } from "drizzle-orm/pg-core"
import { auth_user } from "./auth-schema"

// Social providers (twitter, stocktwits, telegram) deferred to signal source phase
export const credentialProviderEnum = pgEnum("credential_provider", [
  "alpaca",
  "openai",
  "anthropic",
  "google",
  "xai",
  "deepseek",
])

// NOTE: user_id is text (not uuid) to match auth_user.id from Better Auth
export const user_credentials = pgTable("user_credentials", {
  id: uuid("id").defaultRandom().primaryKey(),
  userId: text("user_id")
    .notNull()
    .references(() => auth_user.id, { onDelete: "cascade" }),
  provider: credentialProviderEnum("provider").notNull(),

  // Encrypted credential data (JSON string)
  // For Alpaca: { apiKey, apiSecret, paper }
  // For LLM: { apiKey, baseUrl? }
  encryptedData: text("encrypted_data").notNull(),

  // Unencrypted metadata for list queries (avoids decryption)
  // For Alpaca: true = paper trading
  paperMode: boolean("paper_mode"),

  // Encryption metadata
  iv: text("iv").notNull(), // initialization vector (base64)
  salt: text("salt").notNull(), // per-credential salt (base64)

  // Validation state
  lastValidatedAt: timestamp("last_validated_at"),
  validationError: text("validation_error"),

  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at")
    .defaultNow()
    .$onUpdate(() => new Date())
    .notNull(),
}, (table) => ({
  // One credential per provider per user
  uniqueProviderPerUser: unique().on(table.userId, table.provider),
}))
```

export const user_trading_config = pgTable("user_trading_config", {

  extendedHoursAllowed: boolean("extended_hours_allowed").notNull().default(false),

### Migration SQL

```sql
-- 0002_user_credentials.sql

CREATE TYPE credential_provider AS ENUM (
  'alpaca', 'openai', 'anthropic', 'google', 'xai', 'deepseek'
);

CREATE TABLE user_credentials (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  provider credential_provider NOT NULL,
  encrypted_data TEXT NOT NULL,
  iv TEXT NOT NULL,
  salt TEXT NOT NULL,
  paper_mode BOOLEAN,
  last_validated_at TIMESTAMP,
  validation_error TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE(user_id, provider)
);

CREATE INDEX idx_user_credentials_user_id ON user_credentials(user_id);

-- 0003_user_trading_config.sql

CREATE TABLE user_trading_config (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT NOT NULL UNIQUE REFERENCES auth_user(id) ON DELETE CASCADE,

  max_position_value INTEGER NOT NULL DEFAULT 5000,
  max_positions INTEGER NOT NULL DEFAULT 10,
  max_notional_per_trade INTEGER NOT NULL DEFAULT 5000,

  max_daily_loss_pct REAL NOT NULL DEFAULT 0.02,
  take_profit_pct REAL NOT NULL DEFAULT 0.15,
  stop_loss_pct REAL NOT NULL DEFAULT 0.08,
  position_size_pct_of_cash REAL NOT NULL DEFAULT 0.10,

  cooldown_minutes_after_loss INTEGER NOT NULL DEFAULT 30,

  research_model TEXT DEFAULT 'openai/gpt-4o-mini',
  analyst_model TEXT DEFAULT 'openai/gpt-4o',

  trading_hours_only BOOLEAN NOT NULL DEFAULT TRUE,
  extended_hours_allowed BOOLEAN NOT NULL DEFAULT FALSE,
  allow_short_selling BOOLEAN NOT NULL DEFAULT FALSE,

  ticker_blacklist TEXT[] DEFAULT '{}',
  ticker_allowlist TEXT[],

  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_user_trading_config_user_id ON user_trading_config(user_id);
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/credentials.ts

import { z } from "zod"

// Social providers (twitter, stocktwits, telegram) deferred to signal source phase
export const CredentialProviderSchema = z.enum([
  "alpaca", "openai", "anthropic", "google", "xai", "deepseek",
])
export type CredentialProvider = z.infer<typeof CredentialProviderSchema>

export const LLMProviderSchema = z.enum(["openai", "anthropic", "google", "xai", "deepseek"])
export type LLMProvider = z.infer<typeof LLMProviderSchema>

// Alpaca-specific
export const AlpacaCredentialSchema = z.object({
  apiKey: z.string().min(1),
  apiSecret: z.string().min(1),
  paper: z.boolean().default(true),
})
export type AlpacaCredential = z.infer<typeof AlpacaCredentialSchema>

// LLM provider
export const LLMCredentialSchema = z.object({
  apiKey: z.string().min(1),
  baseUrl: z.string().url().optional(),
})
export type LLMCredential = z.infer<typeof LLMCredentialSchema>

// Union — provider field on request determines which shape to use
export const CredentialDataSchema = z.union([
  AlpacaCredentialSchema,
  LLMCredentialSchema,
])

// API request schemas
export const SaveCredentialRequestSchema = z.object({
  provider: CredentialProviderSchema,
  data: CredentialDataSchema,
  validate: z.boolean().default(true), // whether to validate on save
})

export const DeleteCredentialRequestSchema = z.object({
  provider: CredentialProviderSchema,
})

// Response — list query returns metadata only, no decryption needed
export const CredentialInfoSchema = z.object({
  provider: CredentialProviderSchema,
  paperMode: z.boolean().nullable(), // unencrypted column, only set for alpaca
  lastValidatedAt: z.date().nullable(),
  validationError: z.string().nullable(),
  createdAt: z.date(),
  updatedAt: z.date(),
})
```

```ts
// packages/data-ops/src/zod-schema/trading-config.ts

import { z } from "zod"

export const ModelSelectionSchema = z.string()
  .regex(/^(openai|anthropic|google|xai|deepseek)\/[\w-]+$/,
    "Model must be in format provider/model")

export const TradingConfigSchema = z.object({
  // Position limits
  maxPositionValue: z.number().int().min(100).max(100000).default(5000),
  maxPositions: z.number().int().min(1).max(50).default(10),
  maxNotionalPerTrade: z.number().int().min(100).max(100000).default(5000),

  // Risk management
  maxDailyLossPct: z.number().min(0.001).max(0.10).default(0.02),
  takeProfitPct: z.number().min(0.01).max(1.0).default(0.15),
  stopLossPct: z.number().min(0.01).max(0.50).default(0.08),
  positionSizePctOfCash: z.number().min(0.01).max(1.0).default(0.10),

  // Cooldown
  cooldownMinutesAfterLoss: z.number().int().min(0).max(1440).default(30),

  // LLM
  researchModel: ModelSelectionSchema.default("openai/gpt-4o-mini"),
  analystModel: ModelSelectionSchema.default("openai/gpt-4o"),

  // Feature flags
  tradingHoursOnly: z.boolean().default(true),
  extendedHoursAllowed: z.boolean().default(false),
  allowShortSelling: z.boolean().default(false),

  // Symbols
  tickerBlacklist: z.array(z.string().toUpperCase()).default([]),
  tickerAllowlist: z.array(z.string().toUpperCase()).nullable().default(null),
})

export type TradingConfig = z.infer<typeof TradingConfigSchema>

export const UpdateTradingConfigRequestSchema = TradingConfigSchema.partial()
```
