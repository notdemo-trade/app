## API Handlers

```ts
// apps/data-service/src/hono/handlers/portfolio-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import { ListOrdersRequestSchema, PortfolioHistoryRequestSchema } from "@repo/data-ops/portfolio"
import * as portfolioService from "../services/portfolio-service"

const portfolio = new Hono<{ Bindings: Env }>()

function getServiceContext(c: Context) {
  return {
    db: c.get("db"),
    userId: c.get("userId"),
    masterKey: c.env.CREDENTIALS_ENCRYPTION_KEY,
  }
}

portfolio.get("/account", apiTokenMiddleware, async (c) => {
  const account = await portfolioService.getAccount(getServiceContext(c))
  return c.json({ data: account })
})

portfolio.get("/positions", apiTokenMiddleware, async (c) => {
  const positions = await portfolioService.getPositions(getServiceContext(c))
  return c.json({ data: positions })
})

portfolio.get(
  "/orders",
  apiTokenMiddleware,
  zValidator("query", ListOrdersRequestSchema),
  async (c) => {
    const params = c.req.valid("query")
    const orders = await portfolioService.getOrders(getServiceContext(c), params)
    return c.json({ data: orders })
  }
)

portfolio.get("/clock", apiTokenMiddleware, async (c) => {
  const clock = await portfolioService.getClock(getServiceContext(c))
  return c.json({ data: clock })
})

portfolio.get(
  "/portfolio-history",
  apiTokenMiddleware,
  zValidator("query", PortfolioHistoryRequestSchema),
  async (c) => {
    const params = c.req.valid("query")
    const history = await portfolioService.getPortfolioHistory(getServiceContext(c), params)
    return c.json({ data: history })
  }
)

export { portfolio }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/account` | Bearer | Account info (equity, cash, buying power) |
| GET | `/api/positions` | Bearer | Current positions with P&L |
| GET | `/api/orders` | Bearer | Recent orders |
| GET | `/api/clock` | Bearer | Market open/close status |
| GET | `/api/portfolio-history` | Bearer | Equity over time |

## Server Functions

```ts
// apps/user-application/src/core/functions/portfolio/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { ListOrdersRequestSchema, PortfolioHistoryRequestSchema } from "@repo/data-ops/portfolio"
import * as portfolioService from "@repo/data-ops/services/portfolio-service"

export const getAccount = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return portfolioService.getAccount({
      db: context.db,
      userId: context.session.user.id,
      masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
    })
  })

export const getPositions = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return portfolioService.getPositions({
      db: context.db,
      userId: context.session.user.id,
      masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
    })
  })

export const getOrders = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(ListOrdersRequestSchema)
  .handler(async ({ data, context }) => {
    return portfolioService.getOrders(
      {
        db: context.db,
        userId: context.session.user.id,
        masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
      },
      data
    )
  })

export const getClock = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return portfolioService.getClock({
      db: context.db,
      userId: context.session.user.id,
      masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
    })
  })

export const getPortfolioHistory = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(PortfolioHistoryRequestSchema)
  .handler(async ({ data, context }) => {
    return portfolioService.getPortfolioHistory(
      {
        db: context.db,
        userId: context.session.user.id,
        masterKey: context.env.CREDENTIALS_ENCRYPTION_KEY,
      },
      data
    )
  })
```
