# Phase 20: SRP Multi-Agent Architecture Refactor — Part 5: Frontend & UI

## Overview

New UI components for the discussion feed, trade proposal approval, and orchestration settings. Built on `useAgentChat` hook from Agents SDK + custom `useSession()` wrapper.

## Hooks

### `useSession` Hook

Primary hook for all SessionAgent interaction. Wraps `useAgentChat`.

```ts
// apps/user-application/src/lib/session-connection.ts
// See 020-4-api.md for full implementation

const {
  messages,        // Chat messages (includes tool calls)
  sendMessage,     // Send natural language to SessionAgent
  state,           // Real-time WebSocket state (SessionAgentWebSocketState)
  pendingApprovals, // Tool parts with state="approval-required"
  approveToolCall,  // Approve a trade
  rejectToolCall,   // Reject a trade
  start, stop,      // Enable/disable agent
  updateConfig,     // Update session config
  triggerAnalysis,  // Manual analysis trigger
  getThreads,       // Fetch thread history
  getProposals,     // Fetch proposal history
} = useSession(userId)
```

### `useDiscussionFeed` Hook

Derives discussion data from session state.

```ts
// apps/user-application/src/hooks/use-discussion-feed.ts

export function useDiscussionFeed(state: SessionAgentWebSocketState) {
  const activeThread = state.activeThread

  const messages = useMemo(
    () => activeThread?.messages ?? [],
    [activeThread?.messages]
  )

  const groupedByPhase = useMemo(
    () => groupBy(messages, m => m.phase),
    [messages]
  )

  return {
    thread: activeThread,
    messages,
    groupedByPhase,
    isActive: activeThread?.status === 'in_progress',
    hasProposal: activeThread?.proposal !== null,
  }
}
```

### Sender Display Mapping

```ts
function getSenderDisplay(sender: MessageSender): { avatar: string; name: string; color: string } {
  switch (sender.type) {
    case 'system':
      return { avatar: '/agents/system.svg', name: 'System', color: 'ring-gray-400' }
    case 'data_agent':
      return { avatar: '/agents/data.svg', name: sender.name, color: 'ring-blue-400' }
    case 'analysis_agent':
      return { avatar: '/agents/analysis.svg', name: sender.name, color: 'ring-purple-400' }
    case 'persona':
      return getPersonaDisplay(sender.persona)
    case 'moderator':
      return { avatar: '/agents/moderator.svg', name: 'Moderator', color: 'ring-amber-400' }
    case 'broker':
      return { avatar: '/agents/broker.svg', name: sender.name, color: 'ring-green-400' }
    case 'user':
      return { avatar: '/agents/user.svg', name: 'You', color: 'ring-indigo-400' }
  }
}

function getPersonaDisplay(persona: string) {
  const map: Record<string, { avatar: string; name: string; color: string }> = {
    bull_analyst: { avatar: '/agents/bull.svg', name: 'Bull Analyst', color: 'ring-green-500' },
    bear_analyst: { avatar: '/agents/bear.svg', name: 'Bear Analyst', color: 'ring-red-500' },
    risk_manager: { avatar: '/agents/risk.svg', name: 'Risk Manager', color: 'ring-yellow-500' },
  }
  return map[persona] ?? { avatar: '/agents/persona.svg', name: persona, color: 'ring-gray-400' }
}
```

### TradeProposalCard Component

```tsx
// apps/user-application/src/components/agents/trade-proposal-card.tsx

interface TradeProposalCardProps {
  proposal: TradeProposal
  onApprove?: () => void
  onReject?: () => void
}

export function TradeProposalCard({ proposal, onApprove, onReject }: TradeProposalCardProps) {
  const timeLeft = useCountdown(proposal.expiresAt)
  const isPending = proposal.status === 'pending'

  return (
    <Card className={cn(
      'border-2',
      proposal.action === 'buy' ? 'border-green-500/50' : 'border-red-500/50'
    )}>
      <CardHeader>
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <ActionBadge action={proposal.action} />
            <span className="font-bold text-lg">{proposal.symbol}</span>
          </div>
          <ConfidenceBadge value={proposal.confidence} />
        </div>
      </CardHeader>

      <CardContent className="space-y-3">
        <p className="text-sm">{proposal.rationale}</p>

        <div className="grid grid-cols-2 gap-2 text-sm">
          {proposal.entryPrice && <Stat label="Entry" value={formatCurrency(proposal.entryPrice)} />}
          {proposal.targetPrice && <Stat label="Target" value={formatCurrency(proposal.targetPrice)} />}
          {proposal.stopLoss && <Stat label="Stop Loss" value={formatCurrency(proposal.stopLoss)} />}
          <Stat label="Size" value={`${(proposal.positionSizePct * 100).toFixed(1)}%`} />
          {proposal.notional && <Stat label="Notional" value={formatCurrency(proposal.notional)} />}
        </div>

        {proposal.risks.length > 0 && (
          <div className="text-sm">
            <span className="font-medium text-amber-600">Risks:</span>
            <ul className="list-disc list-inside mt-1">
              {proposal.risks.map((risk, i) => <li key={i}>{risk}</li>)}
            </ul>
          </div>
        )}
      </CardContent>

      {isPending && (
        <CardFooter className="flex justify-between items-center">
          <span className="text-xs text-muted-foreground">
            Expires in {formatDuration(timeLeft)}
          </span>
          <div className="flex gap-2">
            <Button variant="outline" size="sm" onClick={onReject}>
              Reject
            </Button>
            <Button
              variant={proposal.action === 'buy' ? 'default' : 'destructive'}
              size="sm"
              onClick={onApprove}
            >
              Approve {proposal.action.toUpperCase()}
            </Button>
          </div>
        </CardFooter>
      )}

      {!isPending && (
        <CardFooter>
          <StatusBadge status={proposal.status} />
        </CardFooter>
      )}
    </Card>
  )
}
```

```tsx
interface LLMProviderSelectorProps {
  provider: LLMProviderName
  model: string
  onProviderChange: (provider: LLMProviderName, model: string) => void
}

export function LLMProviderSelector({ provider, model, onProviderChange }: LLMProviderSelectorProps) {
  const selected = LLM_PROVIDER_OPTIONS.find(p => p.id === provider)

  return (
    <div className="space-y-3">
      {/* Provider cards */}
      <div className="grid grid-cols-2 gap-2">
        {LLM_PROVIDER_OPTIONS.map(option => (
          <ProviderCard
            key={option.id}
            active={provider === option.id}
            name={option.name}
            description={option.description}
            badge={option.badge}
            requiresKey={option.requiresKey}
            onClick={() => onProviderChange(option.id, option.defaultModel)}
          />
        ))}
      </div>

      {/* Model selector for active provider */}
      {selected && (
        <Select value={model} onValueChange={(m) => onProviderChange(provider, m)}>
          {selected.models.map(m => (
            <SelectItem key={m.id} value={m.id}>{m.name}</SelectItem>
          ))}
        </Select>
      )}

      {/* Credential status for key-required providers */}
      {selected?.requiresKey && (
        <CredentialStatus provider={provider} />
      )}
    </div>
  )
}
```

The `CredentialStatus` sub-component checks whether the user has stored a valid API key for the selected provider. If not, it shows a prompt to add credentials via the existing credential management flow.

### Dashboard Page

```tsx
// apps/user-application/src/routes/dashboard.tsx

export function DashboardPage() {
  const { userId } = useAuth()
  const session = useSession(userId)
  const feed = useDiscussionFeed(session.state)

  return (
    <div className="flex h-full">
      {/* Main area: Discussion Feed */}
      <div className="flex-1 overflow-y-auto p-4">
        <AgentStatusBar
          enabled={session.state.enabled}
          onToggle={() => session.state.enabled ? session.stop() : session.start()}
          cycleCount={session.state.cycleCount}
          lastCycleAt={session.state.lastCycleAt}
        />

        {feed.thread ? (
          <DiscussionThread thread={feed.thread} />
        ) : (
          <EmptyState
            title="No active analysis"
            description="Start the agent or trigger manual analysis"
            action={
              <Button onClick={() => session.triggerAnalysis(session.state.watchlistSymbols?.[0] ?? 'AAPL')}>
                Run Analysis
              </Button>
            }
          />
        )}

        {/* Thread history */}
        <ThreadHistory getThreads={session.getThreads} />
      </div>

      {/* Sidebar: Pending Proposals */}
      <div className="w-80 border-l overflow-y-auto p-4 space-y-3">
        <h3 className="font-semibold text-sm text-muted-foreground">Pending Proposals</h3>
        {session.pendingApprovals.map(approval => (
          <TradeProposalCard
            key={approval.toolCallId}
            proposal={approval.args}
            onApprove={() => session.approveToolCall(approval.toolCallId)}
            onReject={() => session.rejectToolCall(approval.toolCallId)}
          />
        ))}
        {session.pendingApprovals.length === 0 && (
          <p className="text-sm text-muted-foreground">No pending proposals</p>
        )}
      </div>
    </div>
  )
}
```

### Settings Page (or modal)

```tsx
// apps/user-application/src/components/agents/session-settings.tsx

export function SessionSettings() {
  const { userId } = useAuth()
  const session = useSession(userId)
  const [config, setConfig] = useState<SessionConfig | null>(null)

  useEffect(() => {
    session.getConfig().then(setConfig)
  }, [])

  if (!config) return <Loading />

  return (
    <div className="space-y-6">
      {/* LLM Provider */}
      <Section title="LLM Provider">
        <LLMProviderSelector
          provider={config.llmProvider}
          model={config.llmModel}
          onProviderChange={(provider, model) => {
            setConfig({ ...config, llmProvider: provider, llmModel: model })
            session.updateConfig({ llmProvider: provider, llmModel: model })
          }}
        />
      </Section>

      {/* Orchestration mode */}
      <Section title="Orchestration Mode">
        <OrchestrationModeSelector
          value={config.orchestrationMode}
          onChange={(mode) => {
            setConfig({ ...config, orchestrationMode: mode })
            session.updateConfig({ orchestrationMode: mode })
          }}
        />
      </Section>

      {/* Watchlist */}
      <Section title="Watchlist">
        <WatchlistEditor
          symbols={config.watchlistSymbols}
          onChange={(symbols) => {
            setConfig({ ...config, watchlistSymbols: symbols })
            session.updateConfig({ watchlistSymbols: symbols })
          }}
        />
      </Section>

      {/* Strategy */}
      <Section title="Strategy">
        <StrategySelector
          value={config.activeStrategyId}
          onChange={(id) => {
            setConfig({ ...config, activeStrategyId: id })
            session.updateConfig({ activeStrategyId: id })
          }}
        />
      </Section>

      {/* Advanced (collapsed) */}
      <Collapsible title="Advanced Settings">
        <NumberInput
          label="Analysis Interval (seconds)"
          value={config.analysisIntervalSec}
          min={30} max={3600}
          onChange={(v) => session.updateConfig({ analysisIntervalSec: v })}
        />
        <NumberInput
          label="Min Confidence Threshold"
          value={config.minConfidenceThreshold}
          min={0} max={1} step={0.05}
          onChange={(v) => session.updateConfig({ minConfidenceThreshold: v })}
        />
        <NumberInput
          label="Position Size (% of cash)"
          value={config.positionSizePctOfCash}
          min={0.01} max={0.5} step={0.01}
          onChange={(v) => session.updateConfig({ positionSizePctOfCash: v })}
        />
        <NumberInput
          label="Debate Rounds"
          value={config.debateRounds}
          min={1} max={5}
          onChange={(v) => session.updateConfig({ debateRounds: v })}
        />
        <NumberInput
          label="Proposal Timeout (seconds)"
          value={config.proposalTimeoutSec}
          min={60} max={3600}
          onChange={(v) => session.updateConfig({ proposalTimeoutSec: v })}
        />
      </Collapsible>
    </div>
  )
}
```
