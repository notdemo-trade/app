# Phase 23: Extended User Settings — Part 5: UI

## Overview

Extends the existing trading settings page (`/settings/trading`) with a new "Advanced" section. Updates `TradeProposalCard` and `WindowSelector` to read thresholds from user config instead of hardcoded constants.

---

## UX Principle: Minimal Config by Default

These 6 settings are for **advanced users only**. The existing trading settings page (Position Limits, Risk Management, Trading Hours) remains the primary configuration surface.

1. **"Advanced" section is collapsed by default** — new users never see it unless they deliberately expand it
2. **All defaults match current hardcoded values** — zero behavior change for users who never open Advanced
3. **No sidebar navigation change needed** — these settings live inside the existing `/settings/trading` page, not a new route
4. **Sliders with labeled endpoints** — temperature and max tokens use range sliders with human-readable labels (Conservative/Balanced/Creative) instead of raw numbers, reducing cognitive load

## New Component: ScoreWindowsEditor

File: `apps/user-application/src/components/settings/score-windows-editor.tsx`

A small editor for the `scoreWindows` array: shows current windows as removable tags, with an input to add new values.

```tsx
interface ScoreWindowsEditorProps {
  value: number[];
  onChange: (windows: number[]) => void;
}

export function ScoreWindowsEditor({ value, onChange }: ScoreWindowsEditorProps) {
  const [input, setInput] = useState('');

  function addWindow() {
    const n = Number(input);
    if (Number.isNaN(n) || n < 7 || n > 365) return;
    if (value.includes(n)) return;
    if (value.length >= 5) return;
    const next = [...value, n].sort((a, b) => a - b);
    onChange(next);
    setInput('');
  }

  function removeWindow(w: number) {
    if (value.length <= 1) return; // must keep at least 1
    onChange(value.filter((v) => v !== w));
  }

  return (
    <div className="space-y-2">
      <div className="flex flex-wrap gap-2">
        {value.map((w) => (
          <Badge key={w} variant="secondary" className="gap-1">
            {w}d
            {value.length > 1 && (
              <button
                type="button"
                onClick={() => removeWindow(w)}
                className="ml-1 text-muted-foreground hover:text-foreground"
              >
                <X className="h-3 w-3" />
              </button>
            )}
          </Badge>
        ))}
      </div>
      {value.length < 5 && (
        <div className="flex gap-2">
          <Input
            type="number"
            min={7}
            max={365}
            placeholder="Add window (days)"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === 'Enter') {
                e.preventDefault();
                addWindow();
              }
            }}
            className="w-40"
          />
          <Button type="button" variant="outline" size="sm" onClick={addWindow}>
            Add
          </Button>
        </div>
      )}
    </div>
  );
}
```

## New Component: ConfidencePreview

File: `apps/user-application/src/components/settings/confidence-preview.tsx`

A visual bar showing the three confidence bands (red/yellow/green) based on current threshold values.

```tsx
interface ConfidencePreviewProps {
  high: number;
  med: number;
}

export function ConfidencePreview({ high, med }: ConfidencePreviewProps) {
  const lowPct = med * 100;
  const medPct = (high - med) * 100;
  const highPct = (1 - high) * 100;

  return (
    <div className="space-y-1">
      <Label className="text-xs text-muted-foreground">Confidence Band Preview</Label>
      <div className="flex h-6 w-full overflow-hidden rounded-md border border-border">
        <div
          className="flex items-center justify-center bg-red-500/20 text-xs text-red-700 dark:text-red-400"
          style={{ width: `${lowPct}%` }}
        >
          {lowPct > 15 && 'Low'}
        </div>
        <div
          className="flex items-center justify-center bg-yellow-500/20 text-xs text-yellow-700 dark:text-yellow-400"
          style={{ width: `${medPct}%` }}
        >
          {medPct > 15 && 'Med'}
        </div>
        <div
          className="flex items-center justify-center bg-green-500/20 text-xs text-green-700 dark:text-green-400"
          style={{ width: `${highPct}%` }}
        >
          {highPct > 15 && 'High'}
        </div>
      </div>
      <div className="flex justify-between text-xs text-muted-foreground">
        <span>0%</span>
        <span>{(med * 100).toFixed(0)}%</span>
        <span>{(high * 100).toFixed(0)}%</span>
        <span>100%</span>
      </div>
    </div>
  );
}
```

}

export function WindowSelector({ value, onChange, windows }: WindowSelectorProps) {
  const displayWindows = windows ?? [30, 90, 180];

  return (
    <div className="flex gap-1 rounded-lg border border-border bg-muted p-1">
      {displayWindows.map((w) => (
        <Button
          key={w}
          variant="ghost"
