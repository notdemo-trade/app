# Phase 20: SRP Multi-Agent Architecture Refactor — Part 2: Data Model

## Overview

Data layer additions for the new agent categories. Each Durable Object agent uses `this.sql` (SQLite) for local persistence. Shared PG tables remain for cross-agent data (signals written by TA, read by orchestrator).

---

## SQLite Schemas (per Durable Object)

### AlpacaMarketDataAgent SQLite

```sql
-- Cached market data bars
CREATE TABLE IF NOT EXISTS bars (
  symbol    TEXT NOT NULL,
  timeframe TEXT NOT NULL,
  t         TEXT NOT NULL,    -- ISO timestamp
  o         REAL NOT NULL,
  h         REAL NOT NULL,
  l         REAL NOT NULL,
  c         REAL NOT NULL,
  v         INTEGER NOT NULL,
  n         INTEGER NOT NULL,
  vw        REAL NOT NULL,
  fetched_at INTEGER NOT NULL, -- epoch ms
  PRIMARY KEY (symbol, timeframe, t)
);

-- Fetch log for dedup/rate-limiting
CREATE TABLE IF NOT EXISTS fetch_log (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  symbol     TEXT NOT NULL,
  timeframe  TEXT NOT NULL,
  bar_count  INTEGER NOT NULL,
  fetched_at INTEGER NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_bars_symbol_tf ON bars(symbol, timeframe, t DESC);
```

### SessionAgent SQLite

> Note: `AIChatAgent` automatically creates a `messages` table for chat persistence. The tables below are additional.

```sql
-- Session configuration
CREATE TABLE IF NOT EXISTS session_config (
  id                       TEXT PRIMARY KEY DEFAULT 'current',
  orchestration_mode       TEXT NOT NULL DEFAULT 'debate',
  broker_type              TEXT NOT NULL DEFAULT 'AlpacaBrokerAgent',
  llm_provider             TEXT NOT NULL DEFAULT 'workers-ai',  -- LLMProviderName
  llm_model                TEXT NOT NULL DEFAULT '@cf/meta/llama-3.3-70b-instruct-fp8-fast',
  watchlist_symbols        TEXT NOT NULL DEFAULT '[]',   -- JSON array
  analysis_interval_sec    INTEGER NOT NULL DEFAULT 120,
  min_confidence_threshold REAL NOT NULL DEFAULT 0.7,
  position_size_pct        REAL NOT NULL DEFAULT 0.05,
  active_strategy_id       TEXT NOT NULL DEFAULT 'moderate',
  debate_rounds            INTEGER NOT NULL DEFAULT 2,
  proposal_timeout_sec     INTEGER NOT NULL DEFAULT 900,
  updated_at               INTEGER NOT NULL
);

-- Discussion threads
CREATE TABLE IF NOT EXISTS discussion_threads (
  id                  TEXT PRIMARY KEY,
  orchestration_mode  TEXT NOT NULL,
  symbol              TEXT NOT NULL,
  status              TEXT NOT NULL DEFAULT 'in_progress',
  started_at          INTEGER NOT NULL,
  completed_at        INTEGER,
  proposal_id         TEXT
);

-- Discussion messages
CREATE TABLE IF NOT EXISTS discussion_messages (
  id          TEXT PRIMARY KEY,
  thread_id   TEXT NOT NULL REFERENCES discussion_threads(id),
  timestamp   INTEGER NOT NULL,
  sender      TEXT NOT NULL,   -- JSON MessageSender
  phase       TEXT NOT NULL,
  content     TEXT NOT NULL,
  metadata    TEXT NOT NULL DEFAULT '{}',  -- JSON
  FOREIGN KEY (thread_id) REFERENCES discussion_threads(id)
);

-- Trade proposals
CREATE TABLE IF NOT EXISTS trade_proposals (
  id                TEXT PRIMARY KEY,
  thread_id         TEXT NOT NULL REFERENCES discussion_threads(id),
  symbol            TEXT NOT NULL,
  action            TEXT NOT NULL,        -- 'buy' | 'sell'
  confidence        REAL NOT NULL,
  rationale         TEXT NOT NULL,
  entry_price       REAL,
  target_price      REAL,
  stop_loss         REAL,
  qty               REAL,
  notional          REAL,
  position_size_pct REAL NOT NULL,
  risks             TEXT NOT NULL DEFAULT '[]',  -- JSON array
  expires_at        INTEGER NOT NULL,
  status            TEXT NOT NULL DEFAULT 'pending',
  created_at        INTEGER NOT NULL,
  decided_at        INTEGER
);

-- Strategy templates (migrated from OrchestratorAgent in M7)
CREATE TABLE IF NOT EXISTS strategy_templates (
  id                    TEXT PRIMARY KEY,
  name                  TEXT NOT NULL,
  risk_tolerance        TEXT NOT NULL,
  position_size_bias    REAL NOT NULL,
  preferred_timeframe   TEXT NOT NULL,
  analysis_focus        TEXT NOT NULL DEFAULT '[]',  -- JSON array
  custom_prompt_suffix  TEXT
);

-- Entitlements (migrated from OrchestratorAgent in M7)
CREATE TABLE IF NOT EXISTS entitlements (
  agent_type TEXT PRIMARY KEY,
  enabled    INTEGER NOT NULL DEFAULT 1
);

-- Persona configurations
CREATE TABLE IF NOT EXISTS persona_configs (
  id            TEXT PRIMARY KEY,
  name          TEXT NOT NULL,
  role          TEXT NOT NULL,
  system_prompt TEXT NOT NULL,
  bias          TEXT NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_threads_symbol ON discussion_threads(symbol, started_at DESC);
CREATE INDEX IF NOT EXISTS idx_threads_status ON discussion_threads(status);
CREATE INDEX IF NOT EXISTS idx_messages_thread ON discussion_messages(thread_id, timestamp ASC);
CREATE INDEX IF NOT EXISTS idx_proposals_status ON trade_proposals(status, created_at DESC);
```

### DebateOrchestratorAgent SQLite

```sql
-- Debate sessions
CREATE TABLE IF NOT EXISTS debate_sessions (
  id              TEXT PRIMARY KEY,
  symbol          TEXT NOT NULL,
  status          TEXT NOT NULL DEFAULT 'analyzing',
  config          TEXT NOT NULL,        -- JSON DebateConfig
  started_at      INTEGER NOT NULL,
  completed_at    INTEGER
);

-- Persona analyses (initial independent analysis)
CREATE TABLE IF NOT EXISTS persona_analyses (
  id          TEXT PRIMARY KEY,
  session_id  TEXT NOT NULL REFERENCES debate_sessions(id),
  persona_id  TEXT NOT NULL,
  action      TEXT NOT NULL,
  confidence  REAL NOT NULL,
  rationale   TEXT NOT NULL,
  key_points  TEXT NOT NULL DEFAULT '[]',  -- JSON array
  created_at  INTEGER NOT NULL
);

-- Debate rounds
CREATE TABLE IF NOT EXISTS debate_rounds (
  id           TEXT PRIMARY KEY,
  session_id   TEXT NOT NULL REFERENCES debate_sessions(id),
  round_number INTEGER NOT NULL,
  created_at   INTEGER NOT NULL
);

-- Persona responses per round
CREATE TABLE IF NOT EXISTS debate_responses (
  id                 TEXT PRIMARY KEY,
  round_id           TEXT NOT NULL REFERENCES debate_rounds(id),
  persona_id         TEXT NOT NULL,
  responding_to      TEXT NOT NULL DEFAULT '[]',  -- JSON PersonaId[]
  content            TEXT NOT NULL,
  revised_confidence REAL NOT NULL,
  revised_action     TEXT NOT NULL,
  created_at         INTEGER NOT NULL
);

-- Consensus results
CREATE TABLE IF NOT EXISTS consensus_results (
  id                TEXT PRIMARY KEY,
  session_id        TEXT NOT NULL REFERENCES debate_sessions(id) UNIQUE,
  action            TEXT NOT NULL,
  confidence        REAL NOT NULL,
  rationale         TEXT NOT NULL,
  dissent           TEXT,
  entry_price       REAL,
  target_price      REAL,
  stop_loss         REAL,
  position_size_pct REAL,
  risks             TEXT NOT NULL DEFAULT '[]',  -- JSON array
  created_at        INTEGER NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_debate_sessions_symbol ON debate_sessions(symbol, started_at DESC);
```

```ts
export const credentialProviderEnum = pgEnum('credential_provider', [
  'alpaca', 'openai', 'anthropic', 'google', 'xai', 'deepseek', 'workers-ai',
]);
```

Requires migration: `pnpm run drizzle:dev:generate && pnpm run drizzle:dev:migrate`.

> Note: Workers AI doesn't store credentials in the table (no API key needed), but the enum value is required for the `LLMProviderSchema` to align with the pgEnum.

## Zod Schemas

### Session Schemas

```ts
// packages/data-ops/src/agents/session/schema.ts

import { z } from 'zod'

export const SessionConfigSchema = z.object({
  orchestrationMode: z.enum(['debate', 'pipeline']),
  brokerType: z.string().default('AlpacaBrokerAgent'),
  llmProvider: z.enum(['openai', 'anthropic', 'google', 'xai', 'deepseek', 'workers-ai']).default('workers-ai'),
  llmModel: z.string().default('@cf/meta/llama-3.3-70b-instruct-fp8-fast'),
  watchlistSymbols: z.array(z.string()).default([]),
  analysisIntervalSec: z.number().int().min(30).max(3600).default(120),
  minConfidenceThreshold: z.number().min(0).max(1).default(0.7),
  positionSizePctOfCash: z.number().min(0.01).max(0.5).default(0.05),
  activeStrategyId: z.string().default('moderate'),
  debateRounds: z.number().int().min(1).max(5).default(2),
  proposalTimeoutSec: z.number().int().min(60).max(3600).default(900),
})

export type SessionConfigInput = z.infer<typeof SessionConfigSchema>

export const TradeProposalSchema = z.object({
  symbol: z.string(),
  action: z.enum(['buy', 'sell']),
  confidence: z.number().min(0).max(1),
  rationale: z.string(),
  entryPrice: z.number().nullable(),
  targetPrice: z.number().nullable(),
  stopLoss: z.number().nullable(),
  qty: z.number().nullable(),
  notional: z.number().nullable(),
  positionSizePct: z.number(),
  risks: z.array(z.string()),
  timeInForce: z.enum(['day', 'gtc', 'ioc', 'foc']).default('day'),
})

export const DiscussionMessageSchema = z.object({
  sender: z.discriminatedUnion('type', [
    z.object({ type: z.literal('system') }),
    z.object({ type: z.literal('data_agent'), name: z.string() }),
    z.object({ type: z.literal('analysis_agent'), name: z.string() }),
    z.object({ type: z.literal('persona'), persona: z.string() }),
    z.object({ type: z.literal('moderator') }),
    z.object({ type: z.literal('broker'), name: z.string() }),
    z.object({ type: z.literal('user') }),
  ]),
  phase: z.enum([
    'data_collection', 'analysis', 'debate_round', 'consensus',
    'proposal', 'human_decision', 'execution', 'completed',
  ]),
  content: z.string(),
  metadata: z.record(z.unknown()).default({}),
})
```

import { z } from 'zod'

export const OrderRequestSchema = z.object({
