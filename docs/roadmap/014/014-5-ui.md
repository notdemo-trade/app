## Query Hooks

```ts
// apps/user-application/src/lib/platform-stats-queries.ts

import { queryOptions, useQuery, useMutation, useQueryClient } from "@tanstack/react-query"
import {
  getPlatformStats,
  getPrivacySettings,
  updatePrivacySettings
} from "@/core/functions/platform-stats/direct"

export const platformStatsQueryOptions = () =>
  queryOptions({
    queryKey: ["platform", "stats"],
    queryFn: () => getPlatformStats(),
    staleTime: 5 * 60 * 1000, // 5 minutes
    refetchOnWindowFocus: false,
  })

export const privacySettingsQueryOptions = () =>
  queryOptions({
    queryKey: ["platform", "privacy"],
    queryFn: () => getPrivacySettings(),
    staleTime: 60_000,
  })

export const usePlatformStats = () =>
  useQuery(platformStatsQueryOptions())

export const usePrivacySettings = () =>
  useQuery(privacySettingsQueryOptions())

export const useUpdatePrivacySettings = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: updatePrivacySettings,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["platform", "privacy"] })
    },
  })
}
```

## UI Components

### Platform Stats Widget (Homepage)

```tsx
// apps/user-application/src/components/platform-stats/stats-widget.tsx

import { usePlatformStats } from "@/lib/platform-stats-queries"
import { Card, CardContent } from "@/components/ui/card"
import { formatCompact } from "@repo/data-ops/lib/fuzzy-round"
import { cn } from "@/lib/utils"

interface StatCardProps {
  label: string
  value: string
  subValue?: string
  highlight?: boolean
}

function StatCard({ label, value, subValue, highlight }: StatCardProps) {
  return (
    <Card>
      <CardContent className="pt-4 text-center">
        <p className="text-xs text-muted-foreground uppercase tracking-wide">{label}</p>
        <p className={cn(
          "text-2xl font-bold mt-1",
          highlight && "text-green-500"
        )}>
          {value}
        </p>
        {subValue && (
          <p className="text-xs text-muted-foreground">{subValue}</p>
        )}
      </CardContent>
    </Card>
  )
}

export function PlatformStatsWidget() {
  const { data: stats, isLoading } = usePlatformStats()

  if (isLoading) {
    return (
      <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-6 gap-4">
        {Array.from({ length: 6 }).map((_, i) => (
          <Card key={i}>
            <CardContent className="pt-4 animate-pulse">
              <div className="h-4 bg-muted rounded w-16 mx-auto" />
              <div className="h-8 bg-muted rounded w-20 mx-auto mt-2" />
            </CardContent>
          </Card>
        ))}
      </div>
    )
  }

  if (!stats) {
    return null
  }

  return (
    <div className="space-y-4">
      <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-6 gap-4">
        <StatCard
          label="Active Users"
          value={formatCompact(stats.totalUsers)}
        />
        <StatCard
          label="Active Agents"
          value={formatCompact(stats.activeAgents)}
        />
        <StatCard
          label="Total Trades"
          value={formatCompact(stats.totalTrades)}
        />
        <StatCard
          label="Total Volume"
          value={`$${formatCompact(stats.totalVolume)}`}
        />
        <StatCard
          label="Avg Win Rate"
          value={`${(stats.avgWinRate * 100).toFixed(0)}%`}
          highlight={stats.avgWinRate > 0.5}
        />
        <StatCard
          label="Avg ROI (30d)"
          value={`${stats.avgRoi30d > 0 ? "+" : ""}${(stats.avgRoi30d * 100).toFixed(1)}%`}
          highlight={stats.avgRoi30d > 0}
        />
      </div>
      <p className="text-xs text-muted-foreground text-center">
        Last updated: {new Date(stats.lastUpdated).toLocaleString()}
      </p>
    </div>
  )
}
```

### Privacy Toggle (User Settings)

```tsx
// apps/user-application/src/components/platform-stats/privacy-toggle.tsx

import { usePrivacySettings, useUpdatePrivacySettings } from "@/lib/platform-stats-queries"
import { Switch } from "@/components/ui/switch"
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card"
import { Label } from "@/components/ui/label"

export function PrivacyToggle() {
  const { data: settings, isLoading } = usePrivacySettings()
  const updateMutation = useUpdatePrivacySettings()

  if (isLoading) {
    return <div className="animate-pulse h-24" />
  }

  const handleToggle = (checked: boolean) => {
    updateMutation.mutate({ includeInPlatformStats: checked })
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Platform Statistics</CardTitle>
        <CardDescription>
          Contribute to anonymous platform metrics
        </CardDescription>
      </CardHeader>
      <CardContent>
        <div className="flex items-center justify-between">
          <div className="space-y-1">
            <Label htmlFor="include-stats">Include my data in platform statistics</Label>
            <p className="text-sm text-muted-foreground">
              Help build trust by contributing to anonymous aggregate metrics.
              Your individual data is never shown.
            </p>
          </div>
          <Switch
            id="include-stats"
            checked={settings?.includeInPlatformStats ?? true}
            onCheckedChange={handleToggle}
            disabled={updateMutation.isPending}
          />
        </div>
      </CardContent>
    </Card>
  )
}
```
