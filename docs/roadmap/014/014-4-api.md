## API Handlers

```ts
// apps/data-service/src/hono/handlers/platform-stats-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { rateLimiter } from "hono-rate-limiter"
import { apiTokenMiddleware } from "../middleware/auth"
import { UpdatePrivacySettingsSchema } from "@repo/data-ops/zod-schema/platform-stats"
import {
  getPublicPlatformStats,
  getUserPrivacySettings,
  updateUserPrivacySettings
} from "@repo/data-ops/db/queries/platform-stats"

const platformStatsRouter = new Hono<{ Bindings: Env }>()

// Public endpoint - no auth required
platformStatsRouter.get(
  "/stats",
  rateLimiter({
    windowMs: 60 * 1000,
    limit: 100,
    keyGenerator: (c) => c.req.header("cf-connecting-ip") ?? "unknown",
  }),
  async (c) => {
    const stats = await getPublicPlatformStats(c.get("db"))

    if (!stats) {
      return c.json({
        data: null,
        message: "Not enough data to display platform statistics"
      })
    }

    return c.json({ data: stats })
  }
)

// Get user privacy settings
platformStatsRouter.get(
  "/privacy",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const settings = await getUserPrivacySettings(c.get("db"), userId)
    return c.json({ data: settings })
  }
)

// Update user privacy settings
platformStatsRouter.patch(
  "/privacy",
  apiTokenMiddleware,
  zValidator("json", UpdatePrivacySettingsSchema),
  async (c) => {
    const userId = c.get("userId")
    const { includeInPlatformStats } = c.req.valid("json")

    await updateUserPrivacySettings(c.get("db"), userId, includeInPlatformStats)
    return c.json({ success: true })
  }
)

export { platformStatsRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/platform/stats` | None | Get public platform stats (rate limited) |
| GET | `/api/platform/privacy` | Bearer | Get user privacy settings |
| PATCH | `/api/platform/privacy` | Bearer | Update privacy settings |

## Server Functions

```ts
// apps/user-application/src/core/functions/platform-stats/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { UpdatePrivacySettingsSchema } from "@repo/data-ops/zod-schema/platform-stats"
import {
  getPublicPlatformStats,
  getUserPrivacySettings,
  updateUserPrivacySettings
} from "@repo/data-ops/db/queries/platform-stats"

export const getPlatformStats = createServerFn({ method: "GET" })
  .handler(async ({ context }) => {
    return getPublicPlatformStats(context.db)
  })

export const getPrivacySettings = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return getUserPrivacySettings(context.db, context.session.user.id)
  })

export const updatePrivacySettings = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(UpdatePrivacySettingsSchema)
  .handler(async ({ data, context }) => {
    await updateUserPrivacySettings(
      context.db,
      context.session.user.id,
      data.includeInPlatformStats
    )
    return { success: true }
  })
```
