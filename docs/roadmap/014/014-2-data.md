## Database Schema

```sql
-- packages/data-ops/src/db/migrations/0013_platform_stats.sql

CREATE TABLE platform_stats (
  id TEXT PRIMARY KEY,

  -- Counts
  total_users INTEGER NOT NULL DEFAULT 0,
  active_agents INTEGER NOT NULL DEFAULT 0,

  -- Trading volume
  total_trades INTEGER NOT NULL DEFAULT 0,
  total_trades_last_30d INTEGER NOT NULL DEFAULT 0,
  total_trades_last_7d INTEGER NOT NULL DEFAULT 0,
  total_trades_last_24h INTEGER NOT NULL DEFAULT 0,
  total_volume REAL NOT NULL DEFAULT 0,
  total_volume_last_30d REAL NOT NULL DEFAULT 0,

  -- Performance averages (30d)
  avg_roi_30d REAL,
  avg_win_rate_30d REAL,
  avg_sharpe_30d REAL,
  avg_max_drawdown_30d REAL,

  -- Asset breakdown (30d)
  stocks_percent REAL DEFAULT 0,
  crypto_percent REAL DEFAULT 0,
  options_percent REAL DEFAULT 0,

  -- Metadata
  users_included INTEGER NOT NULL DEFAULT 0,
  calculated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Only one row ever exists (latest stats)
CREATE UNIQUE INDEX idx_platform_stats_singleton ON platform_stats((1));

CREATE TABLE user_privacy_settings (
  user_id TEXT PRIMARY KEY REFERENCES "user"(id) ON DELETE CASCADE,
  include_in_platform_stats BOOLEAN NOT NULL DEFAULT true,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_privacy_opted_in ON user_privacy_settings(include_in_platform_stats)
  WHERE include_in_platform_stats = true;
```

## Drizzle Schema

```ts
// packages/data-ops/src/db/schema/platform-stats.ts

import { pgTable, text, integer, real, boolean, timestamp, index, unique } from "drizzle-orm/pg-core"
import { user } from "./auth"
import { sql } from "drizzle-orm"

export const platformStats = pgTable(
  "platform_stats",
  {
    id: text("id").primaryKey(),

    // Counts
    totalUsers: integer("total_users").notNull().default(0),
    activeAgents: integer("active_agents").notNull().default(0),

    // Trading volume
    totalTrades: integer("total_trades").notNull().default(0),
    totalTradesLast30d: integer("total_trades_last_30d").notNull().default(0),
    totalTradesLast7d: integer("total_trades_last_7d").notNull().default(0),
    totalTradesLast24h: integer("total_trades_last_24h").notNull().default(0),
    totalVolume: real("total_volume").notNull().default(0),
    totalVolumeLast30d: real("total_volume_last_30d").notNull().default(0),

    // Performance averages (30d)
    avgRoi30d: real("avg_roi_30d"),
    avgWinRate30d: real("avg_win_rate_30d"),
    avgSharpe30d: real("avg_sharpe_30d"),
    avgMaxDrawdown30d: real("avg_max_drawdown_30d"),

    // Asset breakdown (30d)
    stocksPercent: real("stocks_percent").default(0),
    cryptoPercent: real("crypto_percent").default(0),
    optionsPercent: real("options_percent").default(0),

    // Metadata
    usersIncluded: integer("users_included").notNull().default(0),
    calculatedAt: timestamp("calculated_at", { withTimezone: true }).notNull().defaultNow(),
  }
)

export const userPrivacySettings = pgTable(
  "user_privacy_settings",
  {
    userId: text("user_id").primaryKey().references(() => user.id, { onDelete: "cascade" }),
    includeInPlatformStats: boolean("include_in_platform_stats").notNull().default(true),
    updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    optedInIdx: index("idx_privacy_opted_in")
      .on(table.includeInPlatformStats)
      .where(sql`${table.includeInPlatformStats} = true`),
  })
)

export type PlatformStatsRow = typeof platformStats.$inferSelect
export type UserPrivacySettingsRow = typeof userPrivacySettings.$inferSelect
```
