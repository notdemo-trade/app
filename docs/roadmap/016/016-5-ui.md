## Query Hooks

```ts
// apps/user-application/src/lib/notification-queries.ts

import { queryOptions, useQuery, useMutation, useQueryClient } from "@tanstack/react-query"
import { getNotifConfig, updateNotifConfig, testWebhook, getNotifHistory } from "@/core/functions/notifications/direct"

export const notificationConfigQueryOptions = () =>
  queryOptions({
    queryKey: ["notifications", "config"],
    queryFn: () => getNotifConfig(),
    staleTime: 60_000,
  })

export const notificationHistoryQueryOptions = (limit?: number) =>
  queryOptions({
    queryKey: ["notifications", "history", limit],
    queryFn: () => getNotifHistory({ limit }),
    staleTime: 30_000,
  })

export const useNotificationConfig = () =>
  useQuery(notificationConfigQueryOptions())

export const useNotificationHistory = (limit?: number) =>
  useQuery(notificationHistoryQueryOptions(limit))

export const useUpdateNotificationConfig = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: updateNotifConfig,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["notifications", "config"] })
    },
  })
}

export const useTestWebhook = () => {
  return useMutation({
    mutationFn: testWebhook,
  })
}
```

## UI Components

### Notification Settings

```tsx
// apps/user-application/src/components/notifications/notification-settings.tsx

import { useState } from "react"
import { useNotificationConfig, useUpdateNotificationConfig, useTestWebhook } from "@/lib/notification-queries"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Switch } from "@/components/ui/switch"
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card"
import { Checkbox } from "@/components/ui/checkbox"
import { Label } from "@/components/ui/label"
import type { NotificationEventType } from "@repo/data-ops/notifications"

const EVENT_LABELS: Record<NotificationEventType, string> = {
  trade_executed: "Trade Executed",
  stop_loss_triggered: "Stop Loss Triggered",
  take_profit_triggered: "Take Profit Triggered",
  kill_switch_activated: "Kill Switch Activated",
  daily_loss_limit_hit: "Daily Loss Limit",
  agent_error: "Agent Errors",
  daily_summary: "Daily Summary",
}

export function NotificationSettings() {
  const { data: config, isLoading } = useNotificationConfig()
  const updateMutation = useUpdateNotificationConfig()
  const testMutation = useTestWebhook()

  const [webhookUrl, setWebhookUrl] = useState("")
  const [testResult, setTestResult] = useState<{ success: boolean; message: string } | null>(null)

  // Initialize webhook URL from config
  useState(() => {
    if (config?.webhookUrl) {
      setWebhookUrl(config.webhookUrl)
    }
  })

  if (isLoading) {
    return <div className="animate-pulse">Loading...</div>
  }

  const handleSaveWebhook = () => {
    updateMutation.mutate({ webhookUrl: webhookUrl || null })
  }

  const handleTestWebhook = async () => {
    if (!webhookUrl) return

    try {
      await testMutation.mutateAsync({ webhookUrl })
      setTestResult({ success: true, message: "Test notification sent!" })
    } catch (err) {
      setTestResult({ success: false, message: String(err) })
    }
  }

  const handleToggleEnabled = (enabled: boolean) => {
    updateMutation.mutate({ enabled })
  }

  const handleToggleEvent = (event: NotificationEventType, checked: boolean) => {
    const currentEvents = config?.enabledEvents ?? []
    const newEvents = checked
      ? [...currentEvents, event]
      : currentEvents.filter(e => e !== event)

    updateMutation.mutate({ enabledEvents: newEvents })
  }

  return (
    <div className="space-y-6">
      <Card>
        <CardHeader>
          <div className="flex items-center justify-between">
            <div>
              <CardTitle>Telegram Notifications</CardTitle>
              <CardDescription>Receive alerts via Telegram</CardDescription>
            </div>
            <Switch
              checked={config?.enabled ?? false}
              onCheckedChange={handleToggleEnabled}
            />
          </div>
        </CardHeader>
        <CardContent className="space-y-4">
          {/* Telegram setup handled in Phase 11 TelegramSetupWizard */}
          <p className="text-sm text-muted-foreground">
            Telegram bot setup available in the Telegram Setup tab.
          </p>

          <div className="space-y-2">
            <Label>Cooldown (minutes)</Label>
            <Input
              type="number"
              value={config?.cooldownMins ?? 5}
              onChange={(e) => updateMutation.mutate({ cooldownMins: Number(e.target.value) })}
              min={0}
              max={60}
              className="w-24"
            />
            <p className="text-xs text-muted-foreground">
              Minimum time between notifications for same symbol
            </p>
          </div>
        </CardContent>
      </Card>

      <Card>
        <CardHeader>
          <CardTitle>Event Types</CardTitle>
          <CardDescription>Choose which events trigger notifications</CardDescription>
        </CardHeader>
        <CardContent>
          <div className="grid grid-cols-2 gap-4">
            {(Object.entries(EVENT_LABELS) as [NotificationEventType, string][]).map(
              ([event, label]) => (
                <div key={event} className="flex items-center gap-2">
                  <Checkbox
                    id={event}
                    checked={config?.enabledEvents?.includes(event) ?? false}
                    onCheckedChange={(checked) => handleToggleEvent(event, !!checked)}
                  />
                  <Label htmlFor={event}>{label}</Label>
                </div>
              )
            )}
          </div>
        </CardContent>
      </Card>
    </div>
  )
}
```

### Notification History

```tsx
// apps/user-application/src/components/notifications/notification-history.tsx

import { useNotificationHistory } from "@/lib/notification-queries"
import { Badge } from "@/components/ui/badge"
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table"

export function NotificationHistory() {
  const { data: history, isLoading } = useNotificationHistory(50)

  if (isLoading) {
    return <div className="animate-pulse">Loading history...</div>
  }

  if (!history?.length) {
    return <p className="text-muted-foreground">No notifications sent yet</p>
  }

  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead>Time</TableHead>
          <TableHead>Event</TableHead>
          <TableHead>Title</TableHead>
          <TableHead>Status</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {history.map((log) => (
          <TableRow key={log.id}>
            <TableCell className="text-sm text-muted-foreground">
              {new Date(log.sentAt).toLocaleString()}
            </TableCell>
            <TableCell>
              <Badge variant="outline">{log.eventType}</Badge>
            </TableCell>
            <TableCell>{log.payload.title}</TableCell>
            <TableCell>
              <Badge
                variant={
                  log.status === "sent"
                    ? "default"
                    : log.status === "cooldown"
                      ? "secondary"
                      : "destructive"
                }
              >
                {log.status}
              </Badge>
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  )
}
```
