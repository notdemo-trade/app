## Query Hooks

```ts
// apps/user-application/src/lib/journal-queries.ts

import { queryOptions, useQuery, useMutation, useQueryClient } from "@tanstack/react-query"
import {
  getJournalEntries,
  getJournalEntryById,
  getPerformanceStats,
  getMemoryRules,
  createRule,
  toggleRuleActive,
  removeRule,
} from "@/core/functions/journal/direct"
import type { TradeOutcome } from "@repo/data-ops/journal"

export const journalQueryOptions = (params?: {
  symbol?: string
  outcome?: TradeOutcome
  limit?: number
}) =>
  queryOptions({
    queryKey: ["journal", "list", params],
    queryFn: () => getJournalEntries(params ?? {}),
    staleTime: 30_000,
  })

export const journalEntryQueryOptions = (id: string) =>
  queryOptions({
    queryKey: ["journal", "entry", id],
    queryFn: () => getJournalEntryById({ id }),
    enabled: !!id,
  })

export const journalStatsQueryOptions = (days: number = 30) =>
  queryOptions({
    queryKey: ["journal", "stats", days],
    queryFn: () => getPerformanceStats({ days }),
    staleTime: 60_000,
  })

export const memoryRulesQueryOptions = () =>
  queryOptions({
    queryKey: ["journal", "rules"],
    queryFn: () => getMemoryRules(),
    staleTime: 60_000,
  })

export const useJournal = (params?: { symbol?: string; outcome?: TradeOutcome }) =>
  useQuery(journalQueryOptions(params))

export const useJournalEntry = (id: string) =>
  useQuery(journalEntryQueryOptions(id))

export const useJournalStats = (days?: number) =>
  useQuery(journalStatsQueryOptions(days))

export const useMemoryRules = () =>
  useQuery(memoryRulesQueryOptions())

export const useCreateRule = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: createRule,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["journal", "rules"] })
    },
  })
}

export const useToggleRule = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: toggleRuleActive,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["journal", "rules"] })
    },
  })
}

export const useDeleteRule = () => {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: removeRule,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["journal", "rules"] })
    },
  })
}
```

### Journal List

```tsx
// apps/user-application/src/components/journal/journal-list.tsx

import { useJournal } from "@/lib/journal-queries"
import { Badge } from "@/components/ui/badge"
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table"
import { cn } from "@/lib/utils"
import { Link } from "@tanstack/react-router"
import type { TradeOutcome } from "@repo/data-ops/journal"

interface JournalListProps {
  symbol?: string
  outcome?: TradeOutcome
}

export function JournalList({ symbol, outcome }: JournalListProps) {
  const { data: entries, isLoading } = useJournal({ symbol, outcome })

  if (isLoading) {
    return <div className="animate-pulse">Loading journal...</div>
  }

  if (!entries?.length) {
    return <p className="text-muted-foreground">No trades recorded yet</p>
  }

  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead>Date</TableHead>
          <TableHead>Symbol</TableHead>
          <TableHead>Side</TableHead>
          <TableHead className="text-right">P&L</TableHead>
          <TableHead>Outcome</TableHead>
          <TableHead>Hold</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {entries.map((entry) => (
          <TableRow key={entry.id}>
            <TableCell>
              <Link
                to="/journal/$id"
                params={{ id: entry.id }}
                className="hover:underline"
              >
                {new Date(entry.createdAt).toLocaleDateString()}
              </Link>
            </TableCell>
            <TableCell className="font-mono">{entry.symbol}</TableCell>
            <TableCell>
              <Badge variant={entry.side === "buy" ? "default" : "secondary"}>
                {entry.side}
              </Badge>
            </TableCell>
            <TableCell className={cn(
              "text-right font-mono",
              entry.pnlUsd && entry.pnlUsd > 0 && "text-green-500",
              entry.pnlUsd && entry.pnlUsd < 0 && "text-red-500"
            )}>
              {entry.pnlUsd ? `$${entry.pnlUsd.toFixed(2)}` : "—"}
            </TableCell>
            <TableCell>
              {entry.outcome && (
                <Badge variant={getOutcomeVariant(entry.outcome)}>
                  {entry.outcome}
                </Badge>
              )}
            </TableCell>
            <TableCell className="text-muted-foreground">
              {entry.holdDurationMins ? `${entry.holdDurationMins}m` : "—"}
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  )
}

function getOutcomeVariant(outcome: TradeOutcome) {
  switch (outcome) {
    case "win": return "default"
    case "loss": return "destructive"
    case "scratch": return "secondary"
  }
}
```

### Performance Stats Card

```tsx
// apps/user-application/src/components/journal/stats-card.tsx

import { useJournalStats } from "@/lib/journal-queries"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { cn } from "@/lib/utils"

interface StatsCardProps {
  days?: number
}

export function StatsCard({ days = 30 }: StatsCardProps) {
  const { data: stats, isLoading } = useJournalStats(days)

  if (isLoading) {
    return <div className="animate-pulse h-32" />
  }

  if (!stats) return null

  return (
    <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
      <StatItem
        label="Win Rate"
        value={`${(stats.winRate * 100).toFixed(1)}%`}
        highlight={stats.winRate >= 0.5}
      />
      <StatItem
        label="Total P&L"
        value={`$${stats.totalPnl.toFixed(2)}`}
        highlight={stats.totalPnl > 0}
        negative={stats.totalPnl < 0}
      />
      <StatItem
        label="Avg P&L"
        value={`$${stats.avgPnl.toFixed(2)}`}
        highlight={stats.avgPnl > 0}
        negative={stats.avgPnl < 0}
      />
      <StatItem
        label="Total Trades"
        value={String(stats.totalTrades)}
        subtitle={`${stats.wins}W / ${stats.losses}L / ${stats.scratches}S`}
      />
    </div>
  )
}

function StatItem({
  label,
  value,
  subtitle,
  highlight,
  negative,
}: {
  label: string
  value: string
  subtitle?: string
  highlight?: boolean
  negative?: boolean
}) {
  return (
    <Card>
      <CardHeader className="pb-2">
        <CardTitle className="text-sm text-muted-foreground">{label}</CardTitle>
      </CardHeader>
      <CardContent>
        <p className={cn(
          "text-2xl font-bold",
          highlight && "text-green-500",
          negative && "text-red-500"
        )}>
          {value}
        </p>
        {subtitle && (
          <p className="text-xs text-muted-foreground">{subtitle}</p>
        )}
      </CardContent>
    </Card>
  )
}
```

### Trade Detail View

```tsx
// apps/user-application/src/components/journal/trade-detail.tsx

import { useJournalEntry } from "@/lib/journal-queries"
import { Badge } from "@/components/ui/badge"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { cn } from "@/lib/utils"

interface TradeDetailProps {
  id: string
}

export function TradeDetail({ id }: TradeDetailProps) {
  const { data: entry, isLoading } = useJournalEntry(id)

  if (isLoading) {
    return <div className="animate-pulse">Loading trade...</div>
  }

  if (!entry) {
    return <p className="text-destructive">Trade not found</p>
  }

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">{entry.symbol}</h1>
          <p className="text-muted-foreground">
            {new Date(entry.createdAt).toLocaleString()}
          </p>
        </div>
        {entry.outcome && (
          <Badge
            variant={entry.outcome === "win" ? "default" : entry.outcome === "loss" ? "destructive" : "secondary"}
            className="text-lg px-4 py-1"
          >
            {entry.outcome.toUpperCase()}
          </Badge>
        )}
      </div>

      {/* Trade Details */}
      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
        <DetailCard label="Side" value={entry.side.toUpperCase()} />
        <DetailCard label="Quantity" value={String(entry.qty)} />
        <DetailCard label="Entry" value={entry.entryPrice ? `$${entry.entryPrice.toFixed(2)}` : "—"} />
        <DetailCard label="Exit" value={entry.exitPrice ? `$${entry.exitPrice.toFixed(2)}` : "—"} />
      </div>

      {/* P&L */}
      {entry.pnlUsd !== null && (
        <Card>
          <CardHeader>
            <CardTitle>Performance</CardTitle>
          </CardHeader>
          <CardContent className="grid grid-cols-3 gap-4">
            <div>
              <p className="text-sm text-muted-foreground">P&L (USD)</p>
              <p className={cn(
                "text-xl font-bold",
                entry.pnlUsd > 0 && "text-green-500",
                entry.pnlUsd < 0 && "text-red-500"
              )}>
                ${entry.pnlUsd.toFixed(2)}
              </p>
            </div>
            <div>
              <p className="text-sm text-muted-foreground">P&L (%)</p>
              <p className={cn(
                "text-xl font-bold",
                entry.pnlPct && entry.pnlPct > 0 && "text-green-500",
                entry.pnlPct && entry.pnlPct < 0 && "text-red-500"
              )}>
                {entry.pnlPct ? `${(entry.pnlPct * 100).toFixed(2)}%` : "—"}
              </p>
            </div>
            <div>
              <p className="text-sm text-muted-foreground">Hold Time</p>
              <p className="text-xl font-bold">
                {entry.holdDurationMins ? formatDuration(entry.holdDurationMins) : "—"}
              </p>
            </div>
          </CardContent>
        </Card>
      )}

      {/* Signals Snapshot */}
      {entry.signals && (
        <Card>
          <CardHeader>
            <CardTitle>Signals at Entry</CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-sm">Aggregate Score: {entry.signals.aggregateScore.toFixed(2)}</p>
            <div className="mt-2 flex flex-wrap gap-2">
              {entry.signals.topSignals.map((sig, i) => (
                <Badge key={i} variant="outline">{sig}</Badge>
              ))}
            </div>
          </CardContent>
        </Card>
      )}

      {/* Technicals Snapshot */}
      {entry.technicals && (
        <Card>
          <CardHeader>
            <CardTitle>Technicals at Entry</CardTitle>
          </CardHeader>
          <CardContent className="grid grid-cols-3 md:grid-cols-6 gap-4 text-sm">
            {entry.technicals.rsi14 && (
              <div>
                <p className="text-muted-foreground">RSI</p>
                <p className="font-mono">{entry.technicals.rsi14.toFixed(1)}</p>
              </div>
            )}
            {entry.technicals.macdHistogram && (
              <div>
                <p className="text-muted-foreground">MACD</p>
                <p className="font-mono">{entry.technicals.macdHistogram.toFixed(3)}</p>
              </div>
            )}
            {entry.technicals.sma20 && (
              <div>
                <p className="text-muted-foreground">SMA20</p>
                <p className="font-mono">${entry.technicals.sma20.toFixed(2)}</p>
              </div>
            )}
            {entry.technicals.relativeVolume && (
              <div>
                <p className="text-muted-foreground">Rel Vol</p>
                <p className="font-mono">{entry.technicals.relativeVolume.toFixed(1)}x</p>
              </div>
            )}
          </CardContent>
        </Card>
      )}

      {/* Notes */}
      {(entry.notes || entry.lessonsLearned) && (
        <Card>
          <CardHeader>
            <CardTitle>Notes</CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            {entry.notes && (
              <div>
                <p className="text-sm text-muted-foreground">Entry Notes</p>
                <p>{entry.notes}</p>
              </div>
            )}
            {entry.lessonsLearned && (
              <div>
                <p className="text-sm text-muted-foreground">Lessons Learned</p>
                <p>{entry.lessonsLearned}</p>
              </div>
            )}
          </CardContent>
        </Card>
      )}
    </div>
  )
}

function DetailCard({ label, value }: { label: string; value: string }) {
  return (
    <Card>
      <CardContent className="pt-4">
        <p className="text-xs text-muted-foreground">{label}</p>
        <p className="text-lg font-semibold">{value}</p>
      </CardContent>
    </Card>
  )
}

function formatDuration(mins: number): string {
  if (mins < 60) return `${mins}m`
  const hours = Math.floor(mins / 60)
  const remaining = mins % 60
  if (hours < 24) return `${hours}h ${remaining}m`
  const days = Math.floor(hours / 24)
  return `${days}d ${hours % 24}h`
}
```

### Memory Rules List

```tsx
// apps/user-application/src/components/journal/rules-list.tsx

import { useMemoryRules, useToggleRule, useDeleteRule, useCreateRule } from "@/lib/journal-queries"
import { Badge } from "@/components/ui/badge"
import { Button } from "@/components/ui/button"
import { Switch } from "@/components/ui/switch"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogTrigger } from "@/components/ui/dialog"
import { Input } from "@/components/ui/input"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { Textarea } from "@/components/ui/textarea"
import { useState } from "react"
import type { RuleType } from "@repo/data-ops/journal"

export function RulesList() {
  const { data: rules, isLoading } = useMemoryRules()
  const toggleMutation = useToggleRule()
  const deleteMutation = useDeleteRule()

  if (isLoading) {
    return <div className="animate-pulse">Loading rules...</div>
  }

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <h2 className="text-lg font-semibold">Memory Rules</h2>
        <CreateRuleDialog />
      </div>

      {!rules?.length ? (
        <p className="text-muted-foreground">No rules created yet</p>
      ) : (
        <div className="space-y-2">
          {rules.map((rule) => (
            <Card key={rule.id}>
              <CardContent className="py-4 flex items-center justify-between">
                <div className="flex items-center gap-3">
                  <Switch
                    checked={rule.active}
                    onCheckedChange={(active) =>
                      toggleMutation.mutate({ ruleId: rule.id, active })
                    }
                  />
                  <Badge variant={getRuleVariant(rule.ruleType)}>
                    {rule.ruleType}
                  </Badge>
                  <span className={!rule.active ? "text-muted-foreground" : ""}>
                    {rule.description}
                  </span>
                </div>
                <Button
                  variant="ghost"
                  size="sm"
                  onClick={() => deleteMutation.mutate({ ruleId: rule.id })}
                >
                  Delete
                </Button>
              </CardContent>
            </Card>
          ))}
        </div>
      )}
    </div>
  )
}

function CreateRuleDialog() {
  const [open, setOpen] = useState(false)
  const [ruleType, setRuleType] = useState<RuleType>("insight")
  const [description, setDescription] = useState("")
  const createMutation = useCreateRule()

  const handleSubmit = () => {
    if (!description.trim()) return

    createMutation.mutate(
      { ruleType, description },
      {
        onSuccess: () => {
          setOpen(false)
          setDescription("")
        },
      }
    )
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button size="sm">Add Rule</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Create Memory Rule</DialogTitle>
        </DialogHeader>
        <div className="space-y-4">
          <div>
            <label className="text-sm text-muted-foreground">Type</label>
            <Select value={ruleType} onValueChange={(v) => setRuleType(v as RuleType)}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="avoid">Avoid</SelectItem>
                <SelectItem value="prefer">Prefer</SelectItem>
                <SelectItem value="caution">Caution</SelectItem>
                <SelectItem value="insight">Insight</SelectItem>
              </SelectContent>
            </Select>
          </div>
          <div>
            <label className="text-sm text-muted-foreground">Description</label>
            <Textarea
              value={description}
              onChange={(e) => setDescription(e.target.value)}
              placeholder="e.g., Avoid trading TSLA on earnings day"
            />
          </div>
          <Button onClick={handleSubmit} disabled={createMutation.isPending}>
            Create Rule
          </Button>
        </div>
      </DialogContent>
    </Dialog>
  )
}

function getRuleVariant(type: string) {
  switch (type) {
    case "avoid": return "destructive"
    case "prefer": return "default"
    case "caution": return "secondary"
    default: return "outline"
  }
}
```
