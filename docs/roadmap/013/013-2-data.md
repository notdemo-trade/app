## Database Schema

```sql
-- packages/data-ops/src/db/migrations/0011_trade_journal.sql

CREATE TABLE trade_journal (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES "user"(id) ON DELETE CASCADE,
  trade_id TEXT REFERENCES trades(id),
  symbol TEXT NOT NULL,
  side TEXT NOT NULL,
  entry_price REAL,
  entry_at TIMESTAMPTZ,
  exit_price REAL,
  exit_at TIMESTAMPTZ,
  qty REAL NOT NULL,
  pnl_usd REAL,
  pnl_pct REAL,
  hold_duration_mins INTEGER,
  signals_json JSONB,
  technicals_json JSONB,
  regime_tags TEXT[],
  event_ids TEXT[],
  outcome TEXT CHECK (outcome IN ('win', 'loss', 'scratch')),
  notes TEXT,
  lessons_learned TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_journal_user ON trade_journal(user_id);
CREATE INDEX idx_journal_symbol ON trade_journal(user_id, symbol);
CREATE INDEX idx_journal_outcome ON trade_journal(user_id, outcome);
CREATE INDEX idx_journal_created ON trade_journal(user_id, created_at DESC);

CREATE TABLE memory_rules (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES "user"(id) ON DELETE CASCADE,
  rule_type TEXT NOT NULL CHECK (rule_type IN ('avoid', 'prefer', 'caution', 'insight')),
  description TEXT NOT NULL,
  conditions_json JSONB,
  confidence REAL,
  source TEXT NOT NULL DEFAULT 'manual',
  active BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_rules_user ON memory_rules(user_id);
CREATE INDEX idx_rules_active ON memory_rules(user_id, active);
```

## Drizzle Schema

```ts
// packages/data-ops/src/db/schema/journal.ts

import { pgTable, text, real, integer, boolean, timestamp, jsonb, index } from "drizzle-orm/pg-core"
import { user } from "./auth"

export const tradeJournal = pgTable(
  "trade_journal",
  {
    id: text("id").primaryKey(),
    userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
    tradeId: text("trade_id"),
    symbol: text("symbol").notNull(),
    side: text("side").notNull(),
    entryPrice: real("entry_price"),
    entryAt: timestamp("entry_at", { withTimezone: true }),
    exitPrice: real("exit_price"),
    exitAt: timestamp("exit_at", { withTimezone: true }),
    qty: real("qty").notNull(),
    pnlUsd: real("pnl_usd"),
    pnlPct: real("pnl_pct"),
    holdDurationMins: integer("hold_duration_mins"),
    signalsJson: jsonb("signals_json").$type<SignalSnapshot | null>(),
    technicalsJson: jsonb("technicals_json").$type<TechnicalsSnapshot | null>(),
    regimeTags: text("regime_tags").array(),
    eventIds: text("event_ids").array(),
    outcome: text("outcome").$type<TradeOutcome>(),
    notes: text("notes"),
    lessonsLearned: text("lessons_learned"),
    createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    userIdx: index("idx_journal_user").on(table.userId),
    symbolIdx: index("idx_journal_symbol").on(table.userId, table.symbol),
    outcomeIdx: index("idx_journal_outcome").on(table.userId, table.outcome),
    createdIdx: index("idx_journal_created").on(table.userId, table.createdAt),
  })
)

export const memoryRules = pgTable(
  "memory_rules",
  {
    id: text("id").primaryKey(),
    userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
    ruleType: text("rule_type").notNull().$type<"avoid" | "prefer" | "caution" | "insight">(),
    description: text("description").notNull(),
    conditionsJson: jsonb("conditions_json").$type<RuleConditions | null>(),
    confidence: real("confidence"),
    source: text("source").notNull().default("manual"),
    active: boolean("active").notNull().default(true),
    createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  },
  (table) => ({
    userIdx: index("idx_rules_user").on(table.userId),
    activeIdx: index("idx_rules_active").on(table.userId, table.active),
  })
)

export type TradeJournalRow = typeof tradeJournal.$inferSelect
export type MemoryRuleRow = typeof memoryRules.$inferSelect
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/journal.ts

import { z } from "zod"

export const TradeOutcomeSchema = z.enum(["win", "loss", "scratch"])

export const SignalSnapshotSchema = z.object({
  sources: z.array(z.object({
    source: z.string(),
    sentiment: z.number(),
    mentions: z.number(),
  })),
  aggregateScore: z.number(),
  topSignals: z.array(z.string()),
})

export const TechnicalsSnapshotSchema = z.object({
  rsi14: z.number().nullable(),
  macdHistogram: z.number().nullable(),
  sma20: z.number().nullable(),
  sma50: z.number().nullable(),
  atr14: z.number().nullable(),
  relativeVolume: z.number().nullable(),
})

export const JournalEntrySchema = z.object({
  id: z.string(),
  userId: z.string(),
  tradeId: z.string().nullable(),
  symbol: z.string(),
  side: z.enum(["buy", "sell"]),
  entryPrice: z.number().nullable(),
  entryAt: z.string().nullable(),
  exitPrice: z.number().nullable(),
  exitAt: z.string().nullable(),
  qty: z.number(),
  pnlUsd: z.number().nullable(),
  pnlPct: z.number().nullable(),
  holdDurationMins: z.number().nullable(),
  signals: SignalSnapshotSchema.nullable(),
  technicals: TechnicalsSnapshotSchema.nullable(),
  regimeTags: z.array(z.string()),
  eventIds: z.array(z.string()),
  outcome: TradeOutcomeSchema.nullable(),
  notes: z.string().nullable(),
  lessonsLearned: z.string().nullable(),
  createdAt: z.string(),
  updatedAt: z.string(),
})

export const JournalStatsSchema = z.object({
  totalTrades: z.number(),
  wins: z.number(),
  losses: z.number(),
  scratches: z.number(),
  totalPnl: z.number(),
  avgPnl: z.number(),
  winRate: z.number(),
  avgHoldMins: z.number(),
})

export const QueryJournalRequestSchema = z.object({
  symbol: z.string().optional(),
  outcome: TradeOutcomeSchema.optional(),
  startDate: z.string().optional(),
  endDate: z.string().optional(),
  limit: z.coerce.number().min(1).max(100).optional().default(50),
  offset: z.coerce.number().min(0).optional().default(0),
})

export const JournalStatsRequestSchema = z.object({
  days: z.coerce.number().min(1).max(365).optional().default(30),
})

export const RuleTypeSchema = z.enum(["avoid", "prefer", "caution", "insight"])

export const RuleConditionsSchema = z.object({
  symbols: z.array(z.string()).optional(),
  regimes: z.array(z.string()).optional(),
  indicators: z.record(z.object({
    min: z.number().optional(),
    max: z.number().optional(),
  })).optional(),
  minWinRate: z.number().optional(),
  maxLoss: z.number().optional(),
})

export const MemoryRuleSchema = z.object({
  id: z.string(),
  userId: z.string(),
  ruleType: RuleTypeSchema,
  description: z.string(),
  conditions: RuleConditionsSchema.nullable(),
  confidence: z.number().nullable(),
  source: z.enum(["manual", "extracted"]),
  active: z.boolean(),
  createdAt: z.string(),
})

export const CreateRuleRequestSchema = z.object({
  ruleType: RuleTypeSchema,
  description: z.string().min(1).max(500),
  conditions: RuleConditionsSchema.optional(),
  confidence: z.number().min(0).max(1).optional(),
})

export const UpdateEntryRequestSchema = z.object({
  notes: z.string().max(2000).optional(),
  lessonsLearned: z.string().max(2000).optional(),
})

export type TradeOutcome = z.infer<typeof TradeOutcomeSchema>
export type JournalEntry = z.infer<typeof JournalEntrySchema>
export type JournalStats = z.infer<typeof JournalStatsSchema>
export type MemoryRule = z.infer<typeof MemoryRuleSchema>
```
