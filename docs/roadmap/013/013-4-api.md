## API Handlers

```ts
// apps/data-service/src/hono/handlers/journal-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { z } from "zod"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  QueryJournalRequestSchema,
  JournalStatsRequestSchema,
  CreateRuleRequestSchema,
  UpdateEntryRequestSchema,
} from "@repo/data-ops/journal"
import { queryJournal, getJournalEntry, getJournalStats } from "@repo/data-ops/db/queries/journal"
import { createMemoryRule, getActiveRules, getAllRules, toggleRule, deleteRule } from "@repo/data-ops/db/queries/memory-rules"

const journalRouter = new Hono<{ Bindings: Env }>()

// List journal entries
journalRouter.get(
  "/",
  apiTokenMiddleware,
  zValidator("query", QueryJournalRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const params = c.req.valid("query")

    const entries = await queryJournal(c.get("db"), { userId, ...params })
    return c.json({ data: entries })
  }
)

// Get single entry
journalRouter.get(
  "/:id",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const id = c.req.param("id")

    const entry = await getJournalEntry(c.get("db"), userId, id)
    if (!entry) {
      return c.json({ error: "Journal entry not found" }, 404)
    }

    return c.json({ data: entry })
  }
)

// Update entry (notes/lessons)
journalRouter.patch(
  "/:id",
  apiTokenMiddleware,
  zValidator("json", UpdateEntryRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const id = c.req.param("id")
    const { notes, lessonsLearned } = c.req.valid("json")

    const entry = await getJournalEntry(c.get("db"), userId, id)
    if (!entry) {
      return c.json({ error: "Journal entry not found" }, 404)
    }

    await c.get("db")
      .update(tradeJournal)
      .set({
        ...(notes !== undefined && { notes }),
        ...(lessonsLearned !== undefined && { lessonsLearned }),
        updatedAt: new Date(),
      })
      .where(eq(tradeJournal.id, id))

    return c.json({ success: true })
  }
)

// Get stats
journalRouter.get(
  "/stats",
  apiTokenMiddleware,
  zValidator("query", JournalStatsRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const { days } = c.req.valid("query")

    const stats = await getJournalStats(c.get("db"), userId, days)
    return c.json({ data: stats })
  }
)

// Memory rules
journalRouter.get(
  "/rules",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const rules = await getAllRules(c.get("db"), userId)
    return c.json({ data: rules })
  }
)

journalRouter.post(
  "/rules",
  apiTokenMiddleware,
  zValidator("json", CreateRuleRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    const id = await createMemoryRule(c.get("db"), { userId, ...data })
    return c.json({ data: { id } }, 201)
  }
)

journalRouter.patch(
  "/rules/:id/toggle",
  apiTokenMiddleware,
  zValidator("json", z.object({ active: z.boolean() })),
  async (c) => {
    const userId = c.get("userId")
    const ruleId = c.req.param("id")
    const { active } = c.req.valid("json")

    await toggleRule(c.get("db"), userId, ruleId, active)
    return c.json({ success: true })
  }
)

journalRouter.delete(
  "/rules/:id",
  apiTokenMiddleware,
  async (c) => {
    const userId = c.get("userId")
    const ruleId = c.req.param("id")

    await deleteRule(c.get("db"), userId, ruleId)
    return c.json({ success: true })
  }
)

export { journalRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/journal` | Bearer | List journal entries |
| GET | `/api/journal/:id` | Bearer | Get single entry |
| PATCH | `/api/journal/:id` | Bearer | Update notes/lessons |
| GET | `/api/journal/stats` | Bearer | Get performance stats |
| GET | `/api/journal/rules` | Bearer | List memory rules |
| POST | `/api/journal/rules` | Bearer | Create rule |
| PATCH | `/api/journal/rules/:id/toggle` | Bearer | Toggle rule active |
| DELETE | `/api/journal/rules/:id` | Bearer | Delete rule |

## Server Functions

```ts
// apps/user-application/src/core/functions/journal/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { QueryJournalRequestSchema, JournalStatsRequestSchema, CreateRuleRequestSchema } from "@repo/data-ops/journal"
import { queryJournal, getJournalEntry, getJournalStats } from "@repo/data-ops/db/queries/journal"
import { createMemoryRule, getAllRules, toggleRule, deleteRule } from "@repo/data-ops/db/queries/memory-rules"
import { z } from "zod"

export const getJournalEntries = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(QueryJournalRequestSchema)
  .handler(async ({ data, context }) => {
    return queryJournal(context.db, {
      userId: context.session.user.id,
      ...data,
    })
  })

export const getJournalEntryById = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string() }))
  .handler(async ({ data, context }) => {
    return getJournalEntry(context.db, context.session.user.id, data.id)
  })

export const getPerformanceStats = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(JournalStatsRequestSchema)
  .handler(async ({ data, context }) => {
    return getJournalStats(context.db, context.session.user.id, data.days)
  })

export const getMemoryRules = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return getAllRules(context.db, context.session.user.id)
  })

export const createRule = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(CreateRuleRequestSchema)
  .handler(async ({ data, context }) => {
    return createMemoryRule(context.db, {
      userId: context.session.user.id,
      ...data,
    })
  })

export const toggleRuleActive = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ ruleId: z.string(), active: z.boolean() }))
  .handler(async ({ data, context }) => {
    await toggleRule(context.db, context.session.user.id, data.ruleId, data.active)
    return { success: true }
  })

export const removeRule = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ ruleId: z.string() }))
  .handler(async ({ data, context }) => {
    await deleteRule(context.db, context.session.user.id, data.ruleId)
    return { success: true }
  })
```
