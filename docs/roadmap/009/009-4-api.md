## API Handlers

```ts
// apps/data-service/src/hono/handlers/risk-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  PolicyConfigSchema,
  EnableKillSwitchSchema,
  DisableKillSwitchSchema,
} from "@repo/data-ops/risk"
import {
  getRiskState,
  enableKillSwitch,
  disableKillSwitch,
  resetDailyLoss,
} from "@repo/data-ops/db/queries/risk-state"
import { getPolicyConfig, updatePolicyConfig } from "@repo/data-ops/db/queries/policy-config"
import { getAlpacaTradingClient } from "../utils/alpaca"

const riskRouter = new Hono<{ Bindings: Env }>()

// Get risk status
riskRouter.get("/status", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")

  const [riskState, policyConfig] = await Promise.all([
    getRiskState(c.get("db"), userId),
    getPolicyConfig(c.get("db"), userId),
  ])

  // Get account for context
  const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
  let account = null
  if (trading) {
    try {
      account = await trading.getAccount()
    } catch {}
  }

  const dailyLossPct = account ? riskState.dailyLossUsd / account.equity : 0

  return c.json({
    data: {
      killSwitch: {
        active: riskState.killSwitchActive,
        reason: riskState.killSwitchReason,
        activatedAt: riskState.killSwitchAt,
      },
      dailyLoss: {
        usd: riskState.dailyLossUsd,
        pct: dailyLossPct,
        limitPct: policyConfig.maxDailyLossPct,
        resetAt: riskState.dailyLossResetAt,
      },
      cooldown: {
        active: riskState.cooldownUntil ? new Date(riskState.cooldownUntil) > new Date() : false,
        until: riskState.cooldownUntil,
      },
      limits: policyConfig,
    },
  })
})

// Get policy config
riskRouter.get("/policy", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const config = await getPolicyConfig(c.get("db"), userId)
  return c.json({ data: config })
})

// Update policy config
riskRouter.patch(
  "/policy",
  apiTokenMiddleware,
  zValidator("json", PolicyConfigSchema),
  async (c) => {
    const userId = c.get("userId")
    const updates = c.req.valid("json")

    await updatePolicyConfig(c.get("db"), userId, updates)
    const config = await getPolicyConfig(c.get("db"), userId)

    return c.json({ data: config })
  }
)

// Enable kill switch
riskRouter.post(
  "/kill-switch/enable",
  apiTokenMiddleware,
  zValidator("json", EnableKillSwitchSchema),
  async (c) => {
    const userId = c.get("userId")
    const { reason } = c.req.valid("json")

    // Enable kill switch
    await enableKillSwitch(c.get("db"), userId, reason)

    // Cancel all open orders
    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (trading) {
      try {
        await trading.cancelAllOrders()
      } catch {}
    }

    return c.json({ data: { message: "Kill switch enabled", reason } })
  }
)

// Disable kill switch
riskRouter.post(
  "/kill-switch/disable",
  apiTokenMiddleware,
  zValidator("json", DisableKillSwitchSchema),
  async (c) => {
    const userId = c.get("userId")

    await disableKillSwitch(c.get("db"), userId)

    return c.json({ data: { message: "Kill switch disabled" } })
  }
)

// Reset daily loss (manual override)
riskRouter.post("/reset-daily-loss", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  await resetDailyLoss(c.get("db"), userId)
  return c.json({ data: { message: "Daily loss reset" } })
})

export { riskRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/risk/status` | Bearer | Get risk status and limits |
| GET | `/api/risk/policy` | Bearer | Get policy configuration |
| PATCH | `/api/risk/policy` | Bearer | Update policy configuration |
| POST | `/api/risk/kill-switch/enable` | Bearer | Enable kill switch |
| POST | `/api/risk/kill-switch/disable` | Bearer | Disable kill switch |
| POST | `/api/risk/reset-daily-loss` | Bearer | Reset daily loss counter |
