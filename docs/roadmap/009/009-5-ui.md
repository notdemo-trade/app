## Query Hooks

```ts
// apps/user-application/src/lib/risk-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  getRiskStatus,
  getPolicyConfig,
  updatePolicyConfig,
  enableKillSwitch,
  disableKillSwitch,
} from "@/core/functions/risk/direct"
import type { PolicyConfigInput } from "@repo/data-ops/zod-schema/risk"

export const riskStatusQueryOptions = () =>
  queryOptions({
    queryKey: ["risk", "status"],
    queryFn: getRiskStatus,
    staleTime: 10_000,
    refetchInterval: 30_000, // Auto-refresh every 30s
  })

export const policyConfigQueryOptions = () =>
  queryOptions({
    queryKey: ["risk", "policy"],
    queryFn: getPolicyConfig,
    staleTime: 60_000,
  })

export const useRiskStatus = () => useQuery(riskStatusQueryOptions())
export const usePolicyConfig = () => useQuery(policyConfigQueryOptions())

export const useUpdatePolicyConfig = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: PolicyConfigInput) => updatePolicyConfig(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["risk", "policy"] })
    },
  })
}

export const useEnableKillSwitch = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (reason: string) => enableKillSwitch({ reason }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["risk"] })
      queryClient.invalidateQueries({ queryKey: ["orders"] })
    },
  })
}

export const useDisableKillSwitch = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: () => disableKillSwitch({ confirmation: "CONFIRM_RESUME_TRADING" }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["risk"] })
    },
  })
}
```

## UI Components

### Kill Switch Toggle

```tsx
// apps/user-application/src/components/risk/kill-switch-toggle.tsx

import { useState } from "react"
import { useRiskStatus, useEnableKillSwitch, useDisableKillSwitch } from "@/lib/risk-queries"
import { Button } from "@/components/ui/button"
import { Switch } from "@/components/ui/switch"
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from "@/components/ui/alert-dialog"
import { Input } from "@/components/ui/input"
import { AlertTriangle, Shield } from "lucide-react"

export function KillSwitchToggle() {
  const { data } = useRiskStatus()
  const enableMutation = useEnableKillSwitch()
  const disableMutation = useDisableKillSwitch()

  const [showEnableDialog, setShowEnableDialog] = useState(false)
  const [showDisableDialog, setShowDisableDialog] = useState(false)
  const [reason, setReason] = useState("")

  const isActive = data?.killSwitch.active ?? false

  const handleToggle = () => {
    if (isActive) {
      setShowDisableDialog(true)
    } else {
      setShowEnableDialog(true)
    }
  }

  const handleEnable = () => {
    enableMutation.mutate(reason || "Manual activation", {
      onSuccess: () => {
        setShowEnableDialog(false)
        setReason("")
      },
    })
  }

  const handleDisable = () => {
    disableMutation.mutate(undefined, {
      onSuccess: () => setShowDisableDialog(false),
    })
  }

  return (
    <>
      <div
        className={`p-4 rounded-lg border-2 ${
          isActive
            ? "border-red-500 bg-red-50 dark:bg-red-900/20"
            : "border-green-500 bg-green-50 dark:bg-green-900/20"
        }`}
      >
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-3">
            {isActive ? (
              <AlertTriangle className="h-6 w-6 text-red-500" />
            ) : (
              <Shield className="h-6 w-6 text-green-500" />
            )}
            <div>
              <div className="font-semibold">
                Kill Switch {isActive ? "ACTIVE" : "Inactive"}
              </div>
              {isActive && data?.killSwitch.reason && (
                <div className="text-sm text-muted-foreground">
                  Reason: {data.killSwitch.reason}
                </div>
              )}
            </div>
          </div>
          <Switch checked={isActive} onCheckedChange={handleToggle} />
        </div>
      </div>

      {/* Enable Dialog */}
      <AlertDialog open={showEnableDialog} onOpenChange={setShowEnableDialog}>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Enable Kill Switch?</AlertDialogTitle>
            <AlertDialogDescription>
              This will immediately halt all trading and cancel open orders.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <Input
            placeholder="Reason (optional)"
            value={reason}
            onChange={(e) => setReason(e.target.value)}
          />
          <AlertDialogFooter>
            <AlertDialogCancel>Cancel</AlertDialogCancel>
            <AlertDialogAction
              onClick={handleEnable}
              className="bg-red-500 hover:bg-red-600"
              disabled={enableMutation.isPending}
            >
              {enableMutation.isPending ? "Enabling..." : "Enable Kill Switch"}
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>

      {/* Disable Dialog */}
      <AlertDialog open={showDisableDialog} onOpenChange={setShowDisableDialog}>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Resume Trading?</AlertDialogTitle>
            <AlertDialogDescription>
              This will disable the kill switch and allow new orders.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel>Cancel</AlertDialogCancel>
            <AlertDialogAction
              onClick={handleDisable}
              disabled={disableMutation.isPending}
            >
              {disableMutation.isPending ? "Disabling..." : "Resume Trading"}
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </>
  )
}
```

### Risk Status Dashboard

```tsx
// apps/user-application/src/components/risk/risk-status-dashboard.tsx

import { useRiskStatus } from "@/lib/risk-queries"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Progress } from "@/components/ui/progress"
import { Badge } from "@/components/ui/badge"

export function RiskStatusDashboard() {
  const { data, isLoading } = useRiskStatus()

  if (isLoading || !data) {
    return <div className="animate-pulse">Loading risk status...</div>
  }

  const dailyLossPct = (data.dailyLoss.pct * 100).toFixed(2)
  const dailyLossLimitPct = (data.dailyLoss.limitPct * 100).toFixed(0)
  const dailyLossProgress = Math.min(100, (data.dailyLoss.pct / data.dailyLoss.limitPct) * 100)

  return (
    <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
      {/* Daily Loss */}
      <Card>
        <CardHeader className="pb-2">
          <CardTitle className="text-sm font-medium">Daily Loss</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">
            ${Math.abs(data.dailyLoss.usd).toFixed(2)}
          </div>
          <p className="text-xs text-muted-foreground">
            {dailyLossPct}% of equity (limit: {dailyLossLimitPct}%)
          </p>
          <Progress
            value={dailyLossProgress}
            className="mt-2"
            indicatorClassName={dailyLossProgress > 80 ? "bg-red-500" : ""}
          />
        </CardContent>
      </Card>

      {/* Cooldown */}
      <Card>
        <CardHeader className="pb-2">
          <CardTitle className="text-sm font-medium">Cooldown Status</CardTitle>
        </CardHeader>
        <CardContent>
          {data.cooldown.active ? (
            <>
              <Badge variant="destructive">In Cooldown</Badge>
              <p className="text-xs text-muted-foreground mt-2">
                Until: {new Date(data.cooldown.until!).toLocaleTimeString()}
              </p>
            </>
          ) : (
            <Badge variant="outline" className="text-green-600">
              No Cooldown
            </Badge>
          )}
        </CardContent>
      </Card>

      {/* Limits Summary */}
      <Card>
        <CardHeader className="pb-2">
          <CardTitle className="text-sm font-medium">Position Limits</CardTitle>
        </CardHeader>
        <CardContent className="space-y-1 text-sm">
          <div className="flex justify-between">
            <span className="text-muted-foreground">Max per position</span>
            <span>{(data.limits.maxPositionPctEquity * 100).toFixed(0)}%</span>
          </div>
          <div className="flex justify-between">
            <span className="text-muted-foreground">Max positions</span>
            <span>{data.limits.maxOpenPositions}</span>
          </div>
          <div className="flex justify-between">
            <span className="text-muted-foreground">Max per trade</span>
            <span>${data.limits.maxNotionalPerTrade.toLocaleString()}</span>
          </div>
        </CardContent>
      </Card>
    </div>
  )
}
```

### Policy Config Form

```tsx
// apps/user-application/src/components/risk/policy-config-form.tsx

import { usePolicyConfig, useUpdatePolicyConfig } from "@/lib/risk-queries"
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { PolicyConfigSchema, type PolicyConfigInput } from "@repo/data-ops/zod-schema/risk"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Switch } from "@/components/ui/switch"
import { useEffect } from "react"

export function PolicyConfigForm() {
  const { data, isLoading } = usePolicyConfig()
  const updateMutation = useUpdatePolicyConfig()

  const form = useForm<PolicyConfigInput>({
    resolver: zodResolver(PolicyConfigSchema),
  })

  useEffect(() => {
    if (data) {
      form.reset({
        maxPositionPctEquity: data.maxPositionPctEquity,
        maxOpenPositions: data.maxOpenPositions,
        maxNotionalPerTrade: data.maxNotionalPerTrade,
        maxDailyLossPct: data.maxDailyLossPct,
        cooldownMinutesAfterLoss: data.cooldownMinutesAfterLoss,
        tradingHoursOnly: data.tradingHoursOnly,
        allowShortSelling: data.allowShortSelling,
        useCashOnly: data.useCashOnly,
      })
    }
  }, [data, form])

  const onSubmit = (values: PolicyConfigInput) => {
    updateMutation.mutate(values)
  }

  if (isLoading) {
    return <div className="animate-pulse">Loading config...</div>
  }

  return (
    <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
      <div className="grid gap-4 md:grid-cols-2">
        <div>
          <Label>Max Position (% of equity)</Label>
          <Input
            type="number"
            step="0.01"
            {...form.register("maxPositionPctEquity", { valueAsNumber: true })}
          />
        </div>

        <div>
          <Label>Max Open Positions</Label>
          <Input
            type="number"
            {...form.register("maxOpenPositions", { valueAsNumber: true })}
          />
        </div>

        <div>
          <Label>Max $ Per Trade</Label>
          <Input
            type="number"
            {...form.register("maxNotionalPerTrade", { valueAsNumber: true })}
          />
        </div>

        <div>
          <Label>Max Daily Loss (% of equity)</Label>
          <Input
            type="number"
            step="0.001"
            {...form.register("maxDailyLossPct", { valueAsNumber: true })}
          />
        </div>

        <div>
          <Label>Cooldown After Loss (minutes)</Label>
          <Input
            type="number"
            {...form.register("cooldownMinutesAfterLoss", { valueAsNumber: true })}
          />
        </div>
      </div>

      <div className="space-y-4">
        <div className="flex items-center justify-between">
          <Label>Trading hours only</Label>
          <Switch
            checked={form.watch("tradingHoursOnly")}
            onCheckedChange={(v) => form.setValue("tradingHoursOnly", v)}
          />
        </div>

        <div className="flex items-center justify-between">
          <Label>Allow short selling</Label>
          <Switch
            checked={form.watch("allowShortSelling")}
            onCheckedChange={(v) => form.setValue("allowShortSelling", v)}
          />
        </div>

        <div className="flex items-center justify-between">
          <Label>Cash only (no margin)</Label>
          <Switch
            checked={form.watch("useCashOnly")}
            onCheckedChange={(v) => form.setValue("useCashOnly", v)}
          />
        </div>
      </div>

      <Button type="submit" disabled={updateMutation.isPending}>
        {updateMutation.isPending ? "Saving..." : "Save Configuration"}
      </Button>
    </form>
  )
}
```
