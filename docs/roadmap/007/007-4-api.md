```ts
// apps/data-service/src/hono/handlers/strategy-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { z } from "zod"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  CreateStrategySchema,
  UpdateStrategySchema,
  TestStrategySchema,
  GetStrategiesSchema,
  StrategyExportSchema,
} from "@repo/data-ops/strategies"
import * as strategyQueries from "@repo/data-ops/strategies"
import { executeStrategyPrompt } from "@repo/data-ops/services/strategy-execution-service"
import { analyzeSymbol } from "@repo/data-ops/services/analysis-service"
import { createLLMProviderForUser } from "@repo/data-ops/providers/llm/factory"
import { getAlpacaClientForUser, getLLMCredentialsForUser } from "../utils/user-credentials"

const strategyRouter = new Hono<{ Bindings: Env }>()

// List user's strategies
strategyRouter.get(
  "/",
  apiTokenMiddleware,
  zValidator("query", GetStrategiesSchema.pick({ limit: true, offset: true })),
  async (c) => {
    const userId = c.get("userId")
    const { limit, offset } = c.req.valid("query")
    const strategies = await strategyQueries.getUserStrategies(c.get("db"), userId, { limit, offset })
    return c.json({ data: strategies })
  }
)

// List public strategies
strategyRouter.get(
  "/public",
  apiTokenMiddleware,
  zValidator("query", GetStrategiesSchema),
  async (c) => {
    const { limit, offset, sortBy } = c.req.valid("query")
    const strategies = await strategyQueries.getPublicStrategies(c.get("db"), { limit, offset, sortBy })
    return c.json({ data: strategies })
  }
)

// Get single strategy
strategyRouter.get(
  "/:id",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const strategy = await strategyQueries.getStrategy(c.get("db"), id, userId)

    if (!strategy) {
      return c.json({ error: "Strategy not found" }, 404)
    }

    const isStarred = await strategyQueries.isStrategyStarred(c.get("db"), userId, id)
    return c.json({ data: { ...strategy, isStarred } })
  }
)

// Create strategy
strategyRouter.post(
  "/",
  apiTokenMiddleware,
  zValidator("json", CreateStrategySchema),
  async (c) => {
    const userId = c.get("userId")
    const input = c.req.valid("json")
    const id = await strategyQueries.createStrategy(c.get("db"), userId, input)
    return c.json({ data: { id } }, 201)
  }
)

// Update strategy
strategyRouter.patch(
  "/:id",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  zValidator("json", UpdateStrategySchema),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const input = c.req.valid("json")

    const result = await strategyQueries.updateStrategy(c.get("db"), userId, id, input)
    if (!result.success) {
      return c.json({ error: "Strategy not found or unauthorized" }, 404)
    }

    return c.json({ data: { success: true, version: result.newVersion } })
  }
)

// Delete strategy
strategyRouter.delete(
  "/:id",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const deleted = await strategyQueries.deleteStrategy(c.get("db"), userId, id)

    if (!deleted) {
      return c.json({ error: "Strategy not found or unauthorized" }, 404)
    }

    return c.json({ success: true })
  }
)

// Fork strategy
strategyRouter.post(
  "/:id/fork",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const result = await strategyQueries.forkStrategy(c.get("db"), userId, id)

    if (!result.success) {
      return c.json({ error: result.error }, 400)
    }

    return c.json({ data: { id: result.newId } }, 201)
  }
)

// Star strategy
strategyRouter.post(
  "/:id/star",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const success = await strategyQueries.starStrategy(c.get("db"), userId, id)
    return c.json({ success })
  }
)

// Unstar strategy
strategyRouter.delete(
  "/:id/star",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const success = await strategyQueries.unstarStrategy(c.get("db"), userId, id)
    return c.json({ success })
  }
)

// Set active strategy
strategyRouter.post(
  "/:id/activate",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const success = await strategyQueries.setActiveStrategy(c.get("db"), userId, id)

    if (!success) {
      return c.json({ error: "Strategy not found or unauthorized" }, 404)
    }

    return c.json({ success: true })
  }
)

// Get strategy versions
strategyRouter.get(
  "/:id/versions",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const versions = await strategyQueries.getStrategyVersions(c.get("db"), userId, id)
    return c.json({ data: versions })
  }
)

// Rollback to version
strategyRouter.post(
  "/:id/rollback/:version",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid(), version: z.coerce.number() })),
  async (c) => {
    const userId = c.get("userId")
    const { id, version } = c.req.valid("param")
    const success = await strategyQueries.rollbackToVersion(c.get("db"), userId, id, version)

    if (!success) {
      return c.json({ error: "Version not found" }, 404)
    }

    return c.json({ success: true })
  }
)

// Test strategy on symbol
strategyRouter.post(
  "/:id/test",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  zValidator("json", TestStrategySchema),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const { symbol } = c.req.valid("json")

    const strategy = await strategyQueries.getStrategy(c.get("db"), id, userId)
    if (!strategy) {
      return c.json({ error: "Strategy not found" }, 404)
    }

    const llmCreds = await getLLMCredentialsForUser(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!llmCreds) {
      return c.json({ error: "LLM credentials not configured" }, 400)
    }

    const alpacaClient = await getAlpacaClientForUser(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!alpacaClient) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    // Get market data
    const taResult = await analyzeSymbol({ alpacaClient }, symbol)

    // Execute strategy
    const llm = createLLMProviderForUser(llmCreds)
    const result = await executeStrategyPrompt(llm, strategy, {
      symbol: symbol.toUpperCase(),
      technicals: taResult,
      signals: taResult.signals,
      price: { current: taResult.indicators.price, change_pct: 0 },
    })

    return c.json({
      data: {
        recommendation: result.recommendation,
        usage: result.usage,
        validationErrors: result.validationErrors,
        fallbackUsed: result.fallbackUsed,
      },
    })
  }
)

// Export strategy as JSON
strategyRouter.get(
  "/:id/export",
  apiTokenMiddleware,
  zValidator("param", z.object({ id: z.string().uuid() })),
  async (c) => {
    const userId = c.get("userId")
    const { id } = c.req.valid("param")
    const strategy = await strategyQueries.getStrategy(c.get("db"), id, userId)

    if (!strategy) {
      return c.json({ error: "Strategy not found" }, 404)
    }

    const exportData = {
      name: strategy.name,
      description: strategy.description,
      systemPrompt: strategy.systemPrompt,
      parameters: strategy.parameters,
      requiredVariables: strategy.requiredVariables,
      exportedAt: new Date().toISOString(),
      version: "1.0",
    }

    return c.json({ data: exportData })
  }
)

// Import strategy from JSON
strategyRouter.post(
  "/import",
  apiTokenMiddleware,
  zValidator("json", StrategyExportSchema),
  async (c) => {
    const userId = c.get("userId")
    const input = c.req.valid("json")

    const id = await strategyQueries.createStrategy(c.get("db"), userId, {
      name: input.name,
      description: input.description ?? undefined,
      systemPrompt: input.systemPrompt,
      parameters: input.parameters,
      requiredVariables: input.requiredVariables,
      isPublic: false,
    })

    return c.json({ data: { id } }, 201)
  }
)

export { strategyRouter }
```

## Server Functions

```ts
// apps/user-application/src/core/functions/strategies/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { z } from "zod"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import {
  CreateStrategySchema,
  UpdateStrategySchema,
  TestStrategySchema,
  GetStrategiesSchema,
  StrategyExportSchema,
} from "@repo/data-ops/strategies"
import * as strategyQueries from "@repo/data-ops/strategies"
import { executeStrategyPrompt } from "@repo/data-ops/services/strategy-execution-service"
import { analyzeSymbol } from "@repo/data-ops/services/analysis-service"
import { createLLMProviderForUser } from "@repo/data-ops/providers/llm/factory"
import { getAlpacaClientForUser, getLLMCredentialsForUser } from "@repo/data-ops/utils/user-credentials"

export const getStrategies = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(GetStrategiesSchema.pick({ limit: true, offset: true }))
  .handler(async ({ data, context }) => {
    return strategyQueries.getUserStrategies(context.db, context.session.user.id, data)
  })

export const getPublicStrategies = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(GetStrategiesSchema)
  .handler(async ({ data, context }) => {
    return strategyQueries.getPublicStrategies(context.db, data)
  })

export const getStrategy = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    const strategy = await strategyQueries.getStrategy(context.db, data.id, context.session.user.id)
    if (!strategy) throw new Error("Strategy not found")
    const isStarred = await strategyQueries.isStrategyStarred(
      context.db,
      context.session.user.id,
      data.id
    )
    return { ...strategy, isStarred }
  })

export const createStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(CreateStrategySchema)
  .handler(async ({ data, context }) => {
    const id = await strategyQueries.createStrategy(context.db, context.session.user.id, data)
    return { id }
  })

export const updateStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid(), updates: UpdateStrategySchema }))
  .handler(async ({ data, context }) => {
    const result = await strategyQueries.updateStrategy(
      context.db,
      context.session.user.id,
      data.id,
      data.updates
    )
    if (!result.success) throw new Error("Strategy not found or unauthorized")
    return { version: result.newVersion }
  })

export const deleteStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    const deleted = await strategyQueries.deleteStrategy(
      context.db,
      context.session.user.id,
      data.id
    )
    if (!deleted) throw new Error("Strategy not found or unauthorized")
    return { success: true }
  })

export const forkStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    const result = await strategyQueries.forkStrategy(
      context.db,
      context.session.user.id,
      data.id
    )
    if (!result.success) throw new Error(result.error)
    return { id: result.newId }
  })

export const starStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    return strategyQueries.starStrategy(context.db, context.session.user.id, data.id)
  })

export const unstarStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    return strategyQueries.unstarStrategy(context.db, context.session.user.id, data.id)
  })

export const setActiveStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    const success = await strategyQueries.setActiveStrategy(
      context.db,
      context.session.user.id,
      data.id
    )
    if (!success) throw new Error("Strategy not found or unauthorized")
    return { success: true }
  })

export const getActiveStrategy = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return strategyQueries.getActiveStrategy(context.db, context.session.user.id)
  })

export const getStrategyVersions = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid() }))
  .handler(async ({ data, context }) => {
    return strategyQueries.getStrategyVersions(context.db, context.session.user.id, data.id)
  })

export const rollbackStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid(), version: z.number() }))
  .handler(async ({ data, context }) => {
    const success = await strategyQueries.rollbackToVersion(
      context.db,
      context.session.user.id,
      data.id,
      data.version
    )
    if (!success) throw new Error("Version not found")
    return { success: true }
  })

export const testStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ id: z.string().uuid(), symbol: z.string().min(1).max(10) }))
  .handler(async ({ data, context }) => {
    const strategy = await strategyQueries.getStrategy(
      context.db,
      data.id,
      context.session.user.id
    )
    if (!strategy) throw new Error("Strategy not found")

    const llmCreds = await getLLMCredentialsForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )
    if (!llmCreds) throw new Error("LLM credentials not configured")

    const alpacaClient = await getAlpacaClientForUser(
      context.db,
      context.session.user.id,
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )
    if (!alpacaClient) throw new Error("Alpaca credentials not configured")

    const taResult = await analyzeSymbol({ alpacaClient }, data.symbol)

    const llm = createLLMProviderForUser(llmCreds)
    const result = await executeStrategyPrompt(llm, strategy, {
      symbol: data.symbol.toUpperCase(),
      technicals: taResult,
      signals: taResult.signals,
      price: { current: taResult.indicators.price, change_pct: 0 },
    })

    return result
  })

export const importStrategy = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(StrategyExportSchema)
  .handler(async ({ data, context }) => {
    const id = await strategyQueries.createStrategy(context.db, context.session.user.id, {
      name: data.name,
      description: data.description ?? undefined,
      systemPrompt: data.systemPrompt,
      parameters: data.parameters,
      requiredVariables: data.requiredVariables,
      isPublic: false,
    })
    return { id }
  })
```

## Rate Limiting (P0 Blocker)

Protect strategy testing (triggers LLM analysis).

### Middleware Implementation

```ts
// apps/data-service/src/hono/middleware/rate-limit.ts

import { rateLimiter } from 'hono-rate-limiter'

export const strategyTestLimit = rateLimiter({
  windowMs: 60_000,
  limit: 5,
  keyGenerator: (c) => c.get('userId') || c.req.header('cf-connecting-ip') || 'anonymous'
})
```

### Apply to Strategy Test Endpoint

```ts
// apps/data-service/src/hono/handlers/strategy-handlers.ts

import { strategyTestLimit } from '../middleware/rate-limit'

strategiesRouter.post('/:id/test', apiTokenMiddleware, strategyTestLimit, zValidator('json', TestStrategySchema), testStrategyHandler)
```

**Limits**:
- Strategy testing: 5 req/min per user

**Why**: Each test runs full LLM analysis with user's custom prompt. Lower limit prevents excessive LLM costs during template iteration.
