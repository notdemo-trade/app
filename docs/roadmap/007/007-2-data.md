## Database Schema

```ts
// packages/data-ops/src/drizzle/schema/strategies.ts

import {
  pgTable,
  text,
  timestamp,
  uuid,
  boolean,
  integer,
  jsonb,
  index,
  unique,
} from "drizzle-orm/pg-core"
import { users } from "./auth-schema"
import type { StrategyParameters } from "../../strategies/types"

export const strategyTemplates = pgTable(
  "strategy_templates",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    userId: text("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" }),
    name: text("name").notNull(),
    description: text("description"),
    systemPrompt: text("system_prompt").notNull(),
    parameters: jsonb("parameters").$type<StrategyParameters>().notNull(),
    requiredVariables: jsonb("required_variables").$type<string[]>().notNull().default([]),
    isPublic: boolean("is_public").notNull().default(false),
    isActive: boolean("is_active").notNull().default(false),
    starCount: integer("star_count").notNull().default(0),
    forkCount: integer("fork_count").notNull().default(0),
    forkedFrom: uuid("forked_from").references((): AnyPgColumn => strategyTemplates.id, {
      onDelete: "set null",
    }),
    version: integer("version").notNull().default(1),
    createdAt: timestamp("created_at").defaultNow().notNull(),
    updatedAt: timestamp("updated_at").defaultNow().notNull(),
  },
  (table) => ({
    userIdIdx: index("strategy_templates_user_id_idx").on(table.userId),
    isPublicIdx: index("strategy_templates_is_public_idx").on(table.isPublic),
    starCountIdx: index("strategy_templates_star_count_idx").on(table.starCount),
    forkedFromIdx: index("strategy_templates_forked_from_idx").on(table.forkedFrom),
  })
)

export const strategyVersions = pgTable(
  "strategy_versions",
  {
    id: uuid("id").primaryKey().defaultRandom(),
    strategyId: uuid("strategy_id")
      .notNull()
      .references(() => strategyTemplates.id, { onDelete: "cascade" }),
    version: integer("version").notNull(),
    systemPrompt: text("system_prompt").notNull(),
    parameters: jsonb("parameters").$type<StrategyParameters>().notNull(),
    createdAt: timestamp("created_at").defaultNow().notNull(),
    createdBy: text("created_by")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" }),
  },
  (table) => ({
    strategyVersionIdx: unique("strategy_version_unique").on(table.strategyId, table.version),
  })
)

export const strategyStars = pgTable(
  "strategy_stars",
  {
    userId: text("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" }),
    strategyId: uuid("strategy_id")
      .notNull()
      .references(() => strategyTemplates.id, { onDelete: "cascade" }),
    starredAt: timestamp("starred_at").defaultNow().notNull(),
  },
  (table) => ({
    pk: unique("strategy_stars_pk").on(table.userId, table.strategyId),
  })
)

export type StrategyTemplate = typeof strategyTemplates.$inferSelect
export type NewStrategyTemplate = typeof strategyTemplates.$inferInsert
export type StrategyVersion = typeof strategyVersions.$inferSelect
export type StrategyStar = typeof strategyStars.$inferSelect
```

## SQL Migration

```sql
-- packages/data-ops/migrations/0010_strategy_templates.sql

CREATE TABLE strategy_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  description TEXT,
  system_prompt TEXT NOT NULL,
  parameters JSONB NOT NULL,
  required_variables JSONB NOT NULL DEFAULT '[]',
  is_public BOOLEAN NOT NULL DEFAULT false,
  is_active BOOLEAN NOT NULL DEFAULT false,
  star_count INTEGER NOT NULL DEFAULT 0,
  fork_count INTEGER NOT NULL DEFAULT 0,
  forked_from UUID REFERENCES strategy_templates(id) ON DELETE SET NULL,
  version INTEGER NOT NULL DEFAULT 1,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_strategy_templates_user ON strategy_templates(user_id);
CREATE INDEX idx_strategy_templates_public ON strategy_templates(is_public);
CREATE INDEX idx_strategy_templates_stars ON strategy_templates(star_count DESC);
CREATE INDEX idx_strategy_templates_forked ON strategy_templates(forked_from);

CREATE TABLE strategy_versions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  strategy_id UUID NOT NULL REFERENCES strategy_templates(id) ON DELETE CASCADE,
  version INTEGER NOT NULL,
  system_prompt TEXT NOT NULL,
  parameters JSONB NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  created_by TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  UNIQUE(strategy_id, version)
);

CREATE TABLE strategy_stars (
  user_id TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  strategy_id UUID NOT NULL REFERENCES strategy_templates(id) ON DELETE CASCADE,
  starred_at TIMESTAMP NOT NULL DEFAULT NOW(),
  PRIMARY KEY (user_id, strategy_id)
);
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/strategies.ts

import { z } from "zod"

export const RiskToleranceSchema = z.enum(["conservative", "moderate", "aggressive"])
export const StrategyTimeframeSchema = z.enum(["intraday", "swing", "position"])
export const AssetClassSchema = z.enum(["stocks", "crypto", "options"])

export const CustomVariableSchema = z.object({
  name: z.string().min(1).max(50).regex(/^[a-z_][a-z0-9_]*$/i, "Invalid variable name"),
  type: z.enum(["number", "string", "boolean"]),
  defaultValue: z.union([z.string(), z.number(), z.boolean()]),
  description: z.string().max(200).optional(),
})

export const StrategyParametersSchema = z.object({
  riskTolerance: RiskToleranceSchema,
  timeframe: StrategyTimeframeSchema,
  preferredAssets: z.array(AssetClassSchema).min(1),
  customVariables: z.array(CustomVariableSchema).max(10).default([]),
})

export const CreateStrategySchema = z.object({
  name: z.string().min(1).max(100),
  description: z.string().max(500).optional(),
  systemPrompt: z.string().min(100).max(10000),
  parameters: StrategyParametersSchema,
  requiredVariables: z.array(z.string()).default([]),
  isPublic: z.boolean().default(false),
})

export const UpdateStrategySchema = z.object({
  name: z.string().min(1).max(100).optional(),
  description: z.string().max(500).optional().nullable(),
  systemPrompt: z.string().min(100).max(10000).optional(),
  parameters: StrategyParametersSchema.partial().optional(),
  requiredVariables: z.array(z.string()).optional(),
  isPublic: z.boolean().optional(),
})

export const StrategyExportSchema = z.object({
  name: z.string(),
  description: z.string().nullable(),
  systemPrompt: z.string(),
  parameters: StrategyParametersSchema,
  requiredVariables: z.array(z.string()),
  originalAuthor: z.string().optional(),
  exportedAt: z.string(),
  version: z.string(),
})

export const TestStrategySchema = z.object({
  symbol: z.string().min(1).max(10),
})

export const GetStrategiesSchema = z.object({
  limit: z.coerce.number().min(1).max(100).default(20),
  offset: z.coerce.number().min(0).default(0),
  publicOnly: z.coerce.boolean().optional(),
  sortBy: z.enum(["stars", "forks", "recent"]).default("recent"),
})

export type RiskTolerance = z.infer<typeof RiskToleranceSchema>
export type StrategyTimeframe = z.infer<typeof StrategyTimeframeSchema>
export type AssetClass = z.infer<typeof AssetClassSchema>
export type CustomVariable = z.infer<typeof CustomVariableSchema>
export type StrategyParameters = z.infer<typeof StrategyParametersSchema>
export type CreateStrategyInput = z.infer<typeof CreateStrategySchema>
export type UpdateStrategyInput = z.infer<typeof UpdateStrategySchema>
export type StrategyExport = z.infer<typeof StrategyExportSchema>
```
