```ts
// apps/user-application/src/lib/strategy-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  getStrategies,
  getPublicStrategies,
  getStrategy,
  createStrategy,
  updateStrategy,
  deleteStrategy,
  forkStrategy,
  starStrategy,
  unstarStrategy,
  setActiveStrategy,
  getActiveStrategy,
  getStrategyVersions,
  rollbackStrategy,
  testStrategy,
  importStrategy,
} from "@/core/functions/strategies/direct"
import type { CreateStrategyInput, UpdateStrategyInput, StrategyExport } from "@repo/data-ops/strategies"

export const strategiesQueryOptions = (params?: { limit?: number; offset?: number }) =>
  queryOptions({
    queryKey: ["strategies", "user", params],
    queryFn: () => getStrategies(params ?? {}),
    staleTime: 30_000,
  })

export const publicStrategiesQueryOptions = (params?: {
  limit?: number
  offset?: number
  sortBy?: "stars" | "forks" | "recent"
}) =>
  queryOptions({
    queryKey: ["strategies", "public", params],
    queryFn: () => getPublicStrategies(params ?? {}),
    staleTime: 60_000,
  })

export const strategyQueryOptions = (id: string) =>
  queryOptions({
    queryKey: ["strategies", id],
    queryFn: () => getStrategy({ id }),
    staleTime: 30_000,
  })

export const activeStrategyQueryOptions = () =>
  queryOptions({
    queryKey: ["strategies", "active"],
    queryFn: () => getActiveStrategy(),
    staleTime: 60_000,
  })

export const strategyVersionsQueryOptions = (id: string) =>
  queryOptions({
    queryKey: ["strategies", id, "versions"],
    queryFn: () => getStrategyVersions({ id }),
    staleTime: 30_000,
  })

export const useStrategies = (params?: { limit?: number; offset?: number }) =>
  useQuery(strategiesQueryOptions(params))

export const usePublicStrategies = (params?: {
  limit?: number
  offset?: number
  sortBy?: "stars" | "forks" | "recent"
}) => useQuery(publicStrategiesQueryOptions(params))

export const useStrategy = (id: string) => useQuery(strategyQueryOptions(id))

export const useActiveStrategy = () => useQuery(activeStrategyQueryOptions())

export const useStrategyVersions = (id: string) => useQuery(strategyVersionsQueryOptions(id))

export const useCreateStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: CreateStrategyInput) => createStrategy(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["strategies"] })
    },
  })
}

export const useUpdateStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: { id: string; updates: UpdateStrategyInput }) => updateStrategy(data),
    onSuccess: (_data, variables) => {
      queryClient.invalidateQueries({ queryKey: ["strategies", variables.id] })
      queryClient.invalidateQueries({ queryKey: ["strategies", "user"] })
    },
  })
}

export const useDeleteStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => deleteStrategy({ id }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["strategies"] })
    },
  })
}

export const useForkStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => forkStrategy({ id }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["strategies"] })
    },
  })
}

export const useStarStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => starStrategy({ id }),
    onSuccess: (_data, id) => {
      queryClient.invalidateQueries({ queryKey: ["strategies", id] })
      queryClient.invalidateQueries({ queryKey: ["strategies", "public"] })
    },
  })
}

export const useUnstarStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => unstarStrategy({ id }),
    onSuccess: (_data, id) => {
      queryClient.invalidateQueries({ queryKey: ["strategies", id] })
      queryClient.invalidateQueries({ queryKey: ["strategies", "public"] })
    },
  })
}

export const useSetActiveStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => setActiveStrategy({ id }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["strategies"] })
    },
  })
}

export const useRollbackStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: { id: string; version: number }) => rollbackStrategy(data),
    onSuccess: (_data, variables) => {
      queryClient.invalidateQueries({ queryKey: ["strategies", variables.id] })
    },
  })
}

export const useTestStrategy = () =>
  useMutation({
    mutationFn: (data: { id: string; symbol: string }) => testStrategy(data),
  })

export const useImportStrategy = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: StrategyExport) => importStrategy(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["strategies"] })
    },
  })
}
```

```tsx
import type { StrategyTemplate } from "@repo/data-ops/drizzle/schema/strategies"

interface StrategyCardProps {
  strategy: StrategyTemplate
  isOwner: boolean
  isStarred?: boolean
  onActivate?: () => void
  onEdit?: () => void
  onDelete?: () => void
  onFork?: () => void

  isStarred,
  onActivate,
```

```tsx
// apps/user-application/src/components/strategies/strategy-editor.tsx

import { useState } from "react"
import { useForm } from "@tanstack/react-form"
import { zodValidator } from "@tanstack/zod-form-adapter"
import type { StrategyTemplate } from "@repo/data-ops/drizzle/schema/strategies"
import {
  CreateStrategySchema,
  RiskToleranceSchema,
  StrategyTimeframeSchema,
  AssetClassSchema,
  type CreateStrategyInput,
} from "@repo/data-ops/strategies"
import { extractVariablesFromPrompt, validatePromptVariables } from "@repo/data-ops/services/strategy-execution-service"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Textarea } from "@/components/ui/textarea"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { Switch } from "@/components/ui/switch"
import { Badge } from "@/components/ui/badge"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs"

interface StrategyEditorProps {
  strategy?: StrategyTemplate
  onSubmit: (data: CreateStrategyInput) => void
  onCancel: () => void
  isSubmitting?: boolean
}

export function StrategyEditor({ strategy, onSubmit, onCancel, isSubmitting }: StrategyEditorProps) {
  const [promptVars, setPromptVars] = useState<{
    standard: string[]
    custom: string[]
    unknown: string[]
  }>({ standard: [], custom: [], unknown: [] })

  const form = useForm({
    defaultValues: {
      name: strategy?.name ?? "",
      description: strategy?.description ?? "",
      systemPrompt: strategy?.systemPrompt ?? "",
      riskTolerance: strategy?.parameters.riskTolerance ?? "moderate",
      timeframe: strategy?.parameters.timeframe ?? "swing",
      preferredAssets: strategy?.parameters.preferredAssets ?? ["stocks"],
      isPublic: strategy?.isPublic ?? false,
    },
    validatorAdapter: zodValidator(),
    onSubmit: async ({ value }) => {
      onSubmit({
        name: value.name,
        description: value.description || undefined,
        systemPrompt: value.systemPrompt,
        parameters: {
          riskTolerance: value.riskTolerance as "conservative" | "moderate" | "aggressive",
          timeframe: value.timeframe as "intraday" | "swing" | "position",
          preferredAssets: value.preferredAssets as ("stocks" | "crypto" | "options")[],
          customVariables: [],
        },
        requiredVariables: promptVars.standard,
        isPublic: value.isPublic,
      })
    },
  })

  const handlePromptChange = (prompt: string) => {
    form.setFieldValue("systemPrompt", prompt)
    setPromptVars(validatePromptVariables(prompt))
  }

  return (
    <form
      onSubmit={(e) => {
        e.preventDefault()
        form.handleSubmit()
      }}
      className="space-y-6"
    >
      <div className="grid gap-4 md:grid-cols-2">
        <div className="space-y-2">
          <Label htmlFor="name">Strategy Name</Label>
          <form.Field
            name="name"
            validators={{ onChange: CreateStrategySchema.shape.name }}
            children={(field) => (
              <Input
                id="name"
                value={field.state.value}
                onChange={(e) => field.handleChange(e.target.value)}
                placeholder="My Trading Strategy"
              />
            )}
          />
        </div>

        <div className="space-y-2">
          <Label htmlFor="riskTolerance">Risk Tolerance</Label>
          <form.Field
            name="riskTolerance"
            children={(field) => (
              <Select value={field.state.value} onValueChange={field.handleChange}>
                <SelectTrigger>
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="conservative">Conservative</SelectItem>
                  <SelectItem value="moderate">Moderate</SelectItem>
                  <SelectItem value="aggressive">Aggressive</SelectItem>
                </SelectContent>
              </Select>
            )}
          />
        </div>
      </div>

      <div className="space-y-2">
        <Label htmlFor="description">Description</Label>
        <form.Field
          name="description"
          children={(field) => (
            <Textarea
              id="description"
              value={field.state.value}
              onChange={(e) => field.handleChange(e.target.value)}
              placeholder="Brief description of your strategy..."
              rows={2}
            />
          )}
        />
      </div>

      <Tabs defaultValue="prompt">
        <TabsList>
          <TabsTrigger value="prompt">Prompt Editor</TabsTrigger>
          <TabsTrigger value="parameters">Parameters</TabsTrigger>
        </TabsList>

        <TabsContent value="prompt" className="space-y-4">
          <div className="space-y-2">
            <div className="flex items-center justify-between">
              <Label htmlFor="systemPrompt">System Prompt</Label>
              <div className="flex gap-2">
                {promptVars.standard.length > 0 && (
                  <div className="flex gap-1">
                    {promptVars.standard.map((v) => (
                      <Badge key={v} variant="outline" className="text-xs">
                        {`{${v}}`}
                      </Badge>
                    ))}
                  </div>
                )}
              </div>
            </div>
            <form.Field
              name="systemPrompt"
              children={(field) => (
                <Textarea
                  id="systemPrompt"
                  value={field.state.value}
                  onChange={(e) => handlePromptChange(e.target.value)}
                  placeholder="Enter your strategy prompt..."
                  rows={20}
                  className="font-mono text-sm"
                />
              )}
            />
            <p className="text-xs text-muted-foreground">
              Available variables: {`{symbol}`}, {`{price}`}, {`{technicals}`}, {`{signals}`},{" "}
              {`{portfolio}`}, {`{news}`}, {`{position_state}`}
            </p>
            {promptVars.unknown.length > 0 && (
              <p className="text-xs text-yellow-600">
                Unknown variables: {promptVars.unknown.map((v) => `{${v}}`).join(", ")}
              </p>
            )}
          </div>
        </TabsContent>

        <TabsContent value="parameters" className="space-y-4">
          <div className="grid gap-4 md:grid-cols-2">
            <div className="space-y-2">
              <Label>Timeframe</Label>
              <form.Field
                name="timeframe"
                children={(field) => (
                  <Select value={field.state.value} onValueChange={field.handleChange}>
                    <SelectTrigger>
                      <SelectValue />
                    </SelectTrigger>
                    <SelectContent>
                      <SelectItem value="intraday">Intraday</SelectItem>
                      <SelectItem value="swing">Swing</SelectItem>
                      <SelectItem value="position">Position</SelectItem>
                    </SelectContent>
                  </Select>
                )}
              />
            </div>

            <div className="space-y-2">
              <Label>Preferred Assets</Label>
              <form.Field
                name="preferredAssets"
                children={(field) => (
                  <div className="flex gap-2">
                    {["stocks", "crypto", "options"].map((asset) => (
                      <Badge
                        key={asset}
                        variant={field.state.value.includes(asset as never) ? "default" : "outline"}
                        className="cursor-pointer"
                        onClick={() => {
                          const current = field.state.value
                          if (current.includes(asset as never)) {
                            field.handleChange(current.filter((a) => a !== asset))
                          } else {
                            field.handleChange([...current, asset])
                          }
                        }}
                      >
                        {asset}
                      </Badge>
                    ))}
                  </div>
                )}
              />
            </div>
          </div>

          <div className="flex items-center justify-between">
            <div>
              <Label>Make Public</Label>
              <p className="text-xs text-muted-foreground">
                Others can view and fork this strategy
              </p>
            </div>
            <form.Field
              name="isPublic"
              children={(field) => (
                <Switch checked={field.state.value} onCheckedChange={field.handleChange} />
              )}
            />
          </div>
        </TabsContent>
      </Tabs>

      <div className="flex justify-end gap-2">
        <Button type="button" variant="outline" onClick={onCancel}>
          Cancel
        </Button>
        <Button type="submit" disabled={isSubmitting}>
          {isSubmitting ? "Saving..." : strategy ? "Update Strategy" : "Create Strategy"}
        </Button>
      </div>
    </form>
  )
}
```

```tsx
import { useState } from "react"

interface TestStrategyPanelProps {

export function TestStrategyPanel({ strategyId }: TestStrategyPanelProps) {
  const [symbol, setSymbol] = useState("")
  const { mutate, isPending, data, error, reset } = useTestStrategy()

  const handleTest = () => {
    if (symbol.trim()) {
      mutate({ id: strategyId, symbol: symbol.trim().toUpperCase() })
    }
  }

  const actionColors = {
    buy: "bg-green-500/10 text-green-600 border-green-500/30",
    sell: "bg-red-500/10 text-red-600 border-red-500/30",

    <Card>
      <CardHeader>
        <CardTitle className="text-base">Test Strategy</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="flex gap-2">
          <div className="flex-1">
            <Label htmlFor="testSymbol" className="sr-only">
              Symbol
            </Label>
            <Input
              id="testSymbol"
              value={symbol}
              onChange={(e) => {
                setSymbol(e.target.value.toUpperCase())
                reset()
              }}
              placeholder="Enter symbol (e.g., AAPL)"
              onKeyDown={(e) => e.key === "Enter" && handleTest()}
```

### Version History

```tsx
// apps/user-application/src/components/strategies/version-history.tsx

import { useStrategyVersions, useRollbackStrategy } from "@/lib/strategy-queries"
import { Button } from "@/components/ui/button"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { formatDistanceToNow } from "date-fns"
import { History, RotateCcw } from "lucide-react"

interface VersionHistoryProps {
  strategyId: string
  currentVersion: number
}

export function VersionHistory({ strategyId, currentVersion }: VersionHistoryProps) {
  const { data: versions, isLoading } = useStrategyVersions(strategyId)
  const rollbackMutation = useRollbackStrategy()

  if (isLoading) {
    return <div className="animate-pulse">Loading versions...</div>
  }

  if (!versions || versions.length === 0) {
    return (
      <Card>
        <CardContent className="py-6 text-center text-sm text-muted-foreground">
          <History className="h-8 w-8 mx-auto mb-2 opacity-50" />
          No previous versions
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="text-base flex items-center gap-2">
          <History className="h-4 w-4" />
          Version History
        </CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-2">
          {versions.map((version) => (
            <div
              key={version.id}
              className="flex items-center justify-between p-2 rounded border text-sm"
            >
              <div>
                <span className="font-mono">v{version.version}</span>
                <span className="text-muted-foreground ml-2">
                  {formatDistanceToNow(new Date(version.createdAt), { addSuffix: true })}
                </span>
              </div>
              <Button
                size="sm"
                variant="outline"
                disabled={rollbackMutation.isPending}
                onClick={() =>
                  rollbackMutation.mutate({ id: strategyId, version: version.version })
                }
              >
                <RotateCcw className="h-3 w-3 mr-1" />
                Restore
              </Button>
            </div>
          ))}
        </div>
      </CardContent>
    </Card>
  )
}
```

```tsx
import {
  useStrategies,
  usePublicStrategies,
  useCreateStrategy,
  useDeleteStrategy,
  useForkStrategy,
  useStarStrategy,
  useUnstarStrategy,
  useSetActiveStrategy,
} from "@/lib/strategy-queries"
import { StrategyCard } from "@/components/strategies/strategy-card"
import { StrategyEditor } from "@/components/strategies/strategy-editor"
import { Button } from "@/components/ui/button"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs"
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog"
import { Plus } from "lucide-react"

export const Route = createFileRoute("/_auth/strategies/")({
  component: StrategiesPage,
})

function StrategiesPage() {
  const navigate = useNavigate()
  const [showCreateDialog, setShowCreateDialog] = useState(false)

  const { data: myStrategies, isLoading: loadingMine } = useStrategies()
  const { data: publicStrategies, isLoading: loadingPublic } = usePublicStrategies({
    sortBy: "stars",
  })

  const createMutation = useCreateStrategy()
  const deleteMutation = useDeleteStrategy()
  const forkMutation = useForkStrategy()
  const starMutation = useStarStrategy()
  const unstarMutation = useUnstarStrategy()
  const activateMutation = useSetActiveStrategy()

  const handleCreate = async (data: Parameters<typeof createMutation.mutate>[0]) => {
    const result = await createMutation.mutateAsync(data)
    setShowCreateDialog(false)
    navigate({ to: "/strategies/$id", params: { id: result.id } })
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold">Strategy Templates</h1>
        <Button onClick={() => setShowCreateDialog(true)}>
          <Plus className="h-4 w-4 mr-1" />
          New Strategy
        </Button>
      </div>

      <Tabs defaultValue="mine">
        <TabsList>
          <TabsTrigger value="mine">My Strategies</TabsTrigger>
          <TabsTrigger value="public">Public Strategies</TabsTrigger>
        </TabsList>

        <TabsContent value="mine" className="space-y-4">
          {loadingMine ? (
            <div className="animate-pulse">Loading...</div>
          ) : myStrategies?.length === 0 ? (
            <div className="text-center py-12 text-muted-foreground">
              <p>No strategies yet.</p>
              <Button variant="link" onClick={() => setShowCreateDialog(true)}>
                Create your first strategy
              </Button>
            </div>
          ) : (
            <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-4">
              {myStrategies?.map((strategy) => (
                <StrategyCard
                  key={strategy.id}
                  strategy={strategy}
                  isOwner={true}
                  onActivate={() => activateMutation.mutate(strategy.id)}
                  onEdit={() => navigate({ to: "/strategies/$id", params: { id: strategy.id } })}
                  onDelete={() => deleteMutation.mutate(strategy.id)}
                />
              ))}
            </div>
          )}
        </TabsContent>

        <TabsContent value="public" className="space-y-4">
          {loadingPublic ? (
            <div className="animate-pulse">Loading...</div>
          ) : (
            <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-4">
              {publicStrategies?.map((strategy) => (
                <StrategyCard
                  key={strategy.id}
                  strategy={strategy}
                  isOwner={false}
                  isStarred={(strategy as { isStarred?: boolean }).isStarred}
                  onFork={() => forkMutation.mutate(strategy.id)}
                  onStar={() => starMutation.mutate(strategy.id)}
                  onUnstar={() => unstarMutation.mutate(strategy.id)}
                />
              ))}
            </div>
          )}
        </TabsContent>
      </Tabs>

      <Dialog open={showCreateDialog} onOpenChange={setShowCreateDialog}>
        <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
          <DialogHeader>
            <DialogTitle>Create Strategy</DialogTitle>
          </DialogHeader>
          <StrategyEditor
            onSubmit={handleCreate}
            onCancel={() => setShowCreateDialog(false)}
            isSubmitting={createMutation.isPending}
          />
        </DialogContent>
      </Dialog>
    </div>
  )
}
```

```ts
// apps/data-service/src/agents/trading-agent.ts (additions)

import { getActiveStrategy } from "@repo/data-ops/strategies"
import { executeStrategyPrompt } from "@repo/data-ops/services/strategy-execution-service"

// In runAnalysis scheduled method:
async runAnalysis(): Promise<void> {
  if (!this.state.enabled) return
  const config = this.getConfig()
  await this.logActivity("analysis_started")

  // Get user's active strategy
  const activeStrategy = await getActiveStrategy(this.env.DB, this.name)

  if (!activeStrategy) {
    await this.logActivity("error", undefined, { error: "No active strategy configured" })
    return
  }

  // Get cached signals and build context
  const signals = await this.getCachedSignals()

  for (const symbol of config.watchlistSymbols.slice(0, 5)) {
    const context = await this.buildAnalysisContext(symbol, signals)

    try {
      const result = await executeStrategyPrompt(this.llm, activeStrategy, context)

      if (result.recommendation.action !== "hold" &&
          result.recommendation.confidence >= config.minAnalystConfidence) {
        await this.proposeTradeFromRecommendation({
          symbol,
          action: result.recommendation.action,
          confidence: result.recommendation.confidence,
          rationale: result.recommendation.rationale,
        }, config)
      }
    } catch (err) {
      await this.logActivity("error", symbol, { error: String(err) })
    }
  }

  await this.logActivity("analysis_completed", undefined, {
    strategy: activeStrategy.name,
  })

  this.setState({
    ...this.state,
    lastAnalysisAt: new Date().toISOString(),
    cycleCount: this.state.cycleCount + 1,
  })
}
```
