# Phase 11: Telegram Approvals — Part 4: API & Server Functions

## Webhook Handler

The webhook receives Telegram updates and bridges them to SessionAgent RPC. Since proposals are per-user and SessionAgent is keyed by `userId`, the webhook URL includes the userId.

```ts
// apps/data-service/src/hono/handlers/telegram-webhook-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { z } from "zod"
import { getAgentByName } from "agents"
import { TelegramWebhookUpdateSchema } from "@repo/data-ops/zod-schema/telegram"
import { getCredential, saveCredential } from "@repo/data-ops/credential/queries"
import { TelegramService } from "@repo/data-ops/services/telegram-service"
import type { TelegramCredential } from "@repo/data-ops/credential/schema"
import type { SessionAgent } from "../../agents/session-agent"

const telegramWebhook = new Hono<{ Bindings: Env }>()

// Telegram sends updates here for a specific user's bot
// Route: POST /api/telegram/webhook/:webhookSecret
telegramWebhook.post(
  "/webhook/:webhookSecret",
  async (c) => {
    const { webhookSecret } = c.req.param()

    // Resolve userId from webhook secret (see security section in ops)
    const userId = await resolveUserFromWebhookSecret(c.env, webhookSecret)
    if (!userId) {
      return c.json({ ok: true }) // Silent 200 — don't leak info
    }

    // Parse update
    const parseResult = TelegramWebhookUpdateSchema.safeParse(await c.req.json())
    if (!parseResult.success) {
      return c.json({ ok: true })
    }
    const update = parseResult.data

    // Handle /start command — capture chat_id
    if (update.message?.text === "/start") {
      const chatId = String(update.message.chat.id)

      // Update the telegram credential with the captured chatId
      const existingCred = await getCredential<TelegramCredential>(
        c.get("db"), userId, "telegram", c.env.CREDENTIALS_ENCRYPTION_KEY
      )
      if (existingCred) {
        await saveCredential(c.get("db"), userId, "telegram", {
          botToken: existingCred.botToken,
          chatId,
        }, c.env.CREDENTIALS_ENCRYPTION_KEY)

        const telegram = new TelegramService({ botToken: existingCred.botToken, chatId })
        await telegram.sendMessage(
          "<b>Setup Complete!</b>\n\nYou'll now receive trade proposals and notifications here."
        )
      }

      return c.json({ ok: true })
    }

    // Handle callback queries (button clicks)
    if (update.callback_query) {
      const { id: callbackId, data } = update.callback_query

      // Parse: "approve:{proposalId}" or "reject:{proposalId}"
      const [action, proposalId] = data.split(":")
      if (!proposalId || (action !== "approve" && action !== "reject")) {
        return c.json({ ok: true })
      }

      // Get SessionAgent for this user
      const session = await getAgentByName<Env, SessionAgent>(
        c.env.SessionAgent,
        userId
      )

      try {
        if (action === "approve") {
          await session.approveProposal(proposalId)
        } else {
          await session.rejectProposal(proposalId)
        }
      } catch (err) {
        // Proposal may be expired or already processed — not an error
        console.warn("Telegram callback failed:", err)
      }

      // Answer the callback to remove loading spinner in Telegram
      const cred = await getCredential<TelegramCredential>(
        c.get("db"), userId, "telegram", c.env.CREDENTIALS_ENCRYPTION_KEY
      )
      if (cred?.chatId) {
        const telegram = new TelegramService(cred)
        await telegram.answerCallbackQuery(
          callbackId,
          action === "approve" ? "Trade approved!" : "Trade rejected"
        )
      }

      return c.json({ ok: true })
    }

    return c.json({ ok: true })
  }
)

export { telegramWebhook }
```

## API Endpoints

```ts
// apps/data-service/src/hono/handlers/notification-settings-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { sessionMiddleware } from "../middleware/auth"
import { NotificationSettingsSchema } from "@repo/data-ops/zod-schema/telegram"
import {
  getNotificationSettings,
  upsertNotificationSettings,
} from "@repo/data-ops/queries/notification-settings"
import { getCredential } from "@repo/data-ops/credential/queries"
import { TelegramService } from "@repo/data-ops/services/telegram-service"
import type { TelegramCredential } from "@repo/data-ops/credential/schema"

const notificationSettings = new Hono<{ Bindings: Env }>()

// GET /api/notification-settings
notificationSettings.get("/", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const settings = await getNotificationSettings(c.get("db"), userId)

  return c.json({
    data: settings ?? {
      userId,
      enableTradeProposals: true,
      enableTradeResults: true,
      enableDailySummary: true,
      enableRiskAlerts: true,
      dailySummaryTime: "17:00",
      quietHoursStart: null,
      quietHoursEnd: null,
    },
  })
})

// PATCH /api/notification-settings
notificationSettings.patch(
  "/",
  sessionMiddleware,
  zValidator("json", NotificationSettingsSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")
    await upsertNotificationSettings(c.get("db"), userId, data)
    return c.json({ success: true })
  }
)

// POST /api/telegram/test — send test notification
notificationSettings.post("/telegram/test", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const cred = await getCredential<TelegramCredential>(
    c.get("db"), userId, "telegram", c.env.CREDENTIALS_ENCRYPTION_KEY
  )

  if (!cred?.chatId) {
    return c.json({ success: false, error: "Telegram not configured" }, 400)
  }

  const telegram = new TelegramService(cred)
  try {
    await telegram.sendMessage("<b>Test Notification</b>\n\nYour Telegram is configured correctly!")
    return c.json({ success: true })
  } catch (err) {
    return c.json({ success: false, error: String(err) }, 400)
  }
})

// GET /api/telegram/status — check bot connection
notificationSettings.get("/telegram/status", sessionMiddleware, async (c) => {
  const userId = c.get("userId")
  const cred = await getCredential<TelegramCredential>(
    c.get("db"), userId, "telegram", c.env.CREDENTIALS_ENCRYPTION_KEY
  )

  if (!cred) {
    return c.json({ connected: false, reason: "no_credentials" })
  }

  const telegram = new TelegramService({ botToken: cred.botToken, chatId: cred.chatId || "" })
  const result = await telegram.testConnection()

  return c.json({
    connected: result.ok && !!cred.chatId,
    botUsername: result.username,
    hasChatId: !!cred.chatId,
  })
})

export { notificationSettings }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| POST | `/api/telegram/webhook/:webhookSecret` | None (public) | Telegram webhook callback |
| GET | `/api/telegram/status` | Session | Check bot connection + chatId |
| POST | `/api/telegram/test` | Session | Send test message |
| GET | `/api/notification-settings` | Session | Get notification preferences |
| PATCH | `/api/notification-settings` | Session | Update notification preferences |

## Server Functions

```ts
// apps/user-application/src/core/functions/telegram/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import { NotificationSettingsSchema } from "@repo/data-ops/zod-schema/telegram"
import {
  getNotificationSettings,
  upsertNotificationSettings,
} from "@repo/data-ops/queries/notification-settings"
import { getCredential } from "@repo/data-ops/credential/queries"
import { TelegramService } from "@repo/data-ops/services/telegram-service"
import type { TelegramCredential } from "@repo/data-ops/credential/schema"

export const getSettings = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    const settings = await getNotificationSettings(context.db, context.session.user.id)
    return settings ?? {
      userId: context.session.user.id,
      enableTradeProposals: true,
      enableTradeResults: true,
      enableDailySummary: true,
      enableRiskAlerts: true,
      dailySummaryTime: "17:00",
      quietHoursStart: null,
      quietHoursEnd: null,
    }
  })

export const updateSettings = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(NotificationSettingsSchema)
  .handler(async ({ data, context }) => {
    await upsertNotificationSettings(context.db, context.session.user.id, data)
    return { success: true }
  })

export const getTelegramStatus = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    const cred = await getCredential<TelegramCredential>(
      context.db,
      context.session.user.id,
      "telegram",
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!cred) {
      return { connected: false as const, reason: "no_credentials" }
    }

    const telegram = new TelegramService({ botToken: cred.botToken, chatId: cred.chatId || "" })
    const result = await telegram.testConnection()

    return {
      connected: result.ok && !!cred.chatId,
      botUsername: result.username,
      hasChatId: !!cred.chatId,
    }
  })

export const sendTestMessage = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    const cred = await getCredential<TelegramCredential>(
      context.db,
      context.session.user.id,
      "telegram",
      context.env.CREDENTIALS_ENCRYPTION_KEY
    )

    if (!cred?.chatId) {
      throw new Error("Telegram not configured")
    }

    const telegram = new TelegramService(cred)
    await telegram.sendMessage("<b>Test Notification</b>\n\nYour Telegram is configured correctly!")
    return { success: true }
  })
```
