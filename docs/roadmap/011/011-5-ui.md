## Query Hooks

```ts
// apps/user-application/src/lib/telegram-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  getSettings,
  updateSettings,
  getTelegramStatus,
  sendTestMessage,
  getPendingApprovals,
} from "@/core/functions/telegram/direct"
import type { NotificationSettingsInput } from "@repo/data-ops/telegram"

export const notificationSettingsQueryOptions = () =>
  queryOptions({
    queryKey: ["telegram", "settings"],
    queryFn: getSettings,
    staleTime: 60_000,
  })

export const telegramStatusQueryOptions = () =>
  queryOptions({
    queryKey: ["telegram", "status"],
    queryFn: getTelegramStatus,
    staleTime: 30_000,
  })

export const pendingApprovalsQueryOptions = () =>
  queryOptions({
    queryKey: ["telegram", "pending"],
    queryFn: getPendingApprovals,
    staleTime: 5_000,
    refetchInterval: 10_000,
  })

export const useNotificationSettings = () => useQuery(notificationSettingsQueryOptions())
export const useTelegramStatus = () => useQuery(telegramStatusQueryOptions())
export const usePendingApprovals = () => useQuery(pendingApprovalsQueryOptions())

export const useUpdateSettings = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: NotificationSettingsInput) => updateSettings({ data }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["telegram", "settings"] })
    },
  })
}

export const useSendTestMessage = () => {
  return useMutation({
    mutationFn: sendTestMessage,
  })
}
```

## UI Components

### Telegram Setup Wizard

```tsx
// apps/user-application/src/components/telegram/telegram-setup-wizard.tsx

import { useState } from "react"
import { useTelegramStatus, useSendTestMessage } from "@/lib/telegram-queries"
import { useSaveCredential } from "@/lib/credential-queries"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card"
import { Badge } from "@/components/ui/badge"
import { CheckCircle, Circle, Loader2 } from "lucide-react"

type Step = 1 | 2 | 3

export function TelegramSetupWizard() {
  const [step, setStep] = useState<Step>(1)
  const [botToken, setBotToken] = useState("")
  const { data: status, refetch } = useTelegramStatus()
  const saveMutation = useSaveCredential()
  const testMutation = useSendTestMessage()

  const handleSaveToken = async () => {
    await saveMutation.mutateAsync({
      provider: "telegram",
      data: { botToken, chatId: "" },
      validate: false,
    })
    setStep(2)
  }

  const handleTestConnection = async () => {
    await refetch()
    if (status?.connected) {
      await testMutation.mutateAsync()
      setStep(3)
    }
  }

  if (status?.connected && step === 1) {
    return (
      <Card>
        <CardHeader>
          <div className="flex items-center justify-between">
            <CardTitle>Telegram Connected</CardTitle>
            <Badge variant="default">Connected</Badge>
          </div>
          <CardDescription>
            Bot: @{status.botUsername}
          </CardDescription>
        </CardHeader>
        <CardContent>
          <Button onClick={() => testMutation.mutate()} disabled={testMutation.isPending}>
            {testMutation.isPending ? "Sending..." : "Send Test Message"}
          </Button>
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle>Setup Telegram Notifications</CardTitle>
        <CardDescription>3 steps to enable trade approvals on Telegram</CardDescription>
      </CardHeader>
      <CardContent className="space-y-6">
        {/* Step indicators */}
        <div className="flex items-center gap-4">
          <StepIndicator step={1} current={step} label="Create Bot" />
          <div className="h-px flex-1 bg-border" />
          <StepIndicator step={2} current={step} label="Get Chat ID" />
          <div className="h-px flex-1 bg-border" />
          <StepIndicator step={3} current={step} label="Test" />
        </div>

        {/* Step 1: Create Bot */}
        {step === 1 && (
          <div className="space-y-4">
            <div className="space-y-2">
              <p className="text-sm">
                1. Open Telegram and search for <code>@BotFather</code>
              </p>
              <p className="text-sm">
                2. Send <code>/newbot</code> and follow the prompts
              </p>
              <p className="text-sm">
                3. Copy the bot token and paste below
              </p>
            </div>
            <Input
              placeholder="123456789:ABCdefGHIjklMNOpqrsTUVwxyz"
              value={botToken}
              onChange={(e) => setBotToken(e.target.value)}
            />
            <Button onClick={handleSaveToken} disabled={!botToken || saveMutation.isPending}>
              {saveMutation.isPending ? "Saving..." : "Continue"}
            </Button>
          </div>
        )}

        {/* Step 2: Get Chat ID */}
        {step === 2 && (
          <div className="space-y-4">
            <div className="space-y-2">
              <p className="text-sm">
                1. Open your bot in Telegram (search for @{status?.botUsername || "your_bot"})
              </p>
              <p className="text-sm">
                2. Send <code>/start</code> to the bot
              </p>
              <p className="text-sm">
                3. Click below to verify connection
              </p>
            </div>
            <Button onClick={handleTestConnection}>
              Check Connection
            </Button>
            {status && !status.connected && (
              <p className="text-sm text-muted-foreground">
                Waiting for /start message...
              </p>
            )}
          </div>
        )}

        {/* Step 3: Complete */}
        {step === 3 && (
          <div className="space-y-4 text-center">
            <CheckCircle className="h-12 w-12 text-green-500 mx-auto" />
            <p className="text-lg font-medium">Setup Complete!</p>
            <p className="text-sm text-muted-foreground">
              You'll now receive trade approvals and notifications on Telegram.
            </p>
          </div>
        )}
      </CardContent>
    </Card>
  )
}

function StepIndicator({ step, current, label }: { step: Step; current: Step; label: string }) {
  const isComplete = current > step
  const isCurrent = current === step

  return (
    <div className="flex flex-col items-center gap-1">
      {isComplete ? (
        <CheckCircle className="h-6 w-6 text-green-500" />
      ) : isCurrent ? (
        <div className="h-6 w-6 rounded-full border-2 border-primary flex items-center justify-center">
          <span className="text-xs font-medium">{step}</span>
        </div>
      ) : (
        <Circle className="h-6 w-6 text-muted-foreground" />
      )}
      <span className="text-xs text-muted-foreground">{label}</span>
    </div>
  )
}
```

### Pending Approvals UI

```tsx
// apps/user-application/src/components/telegram/pending-approvals-list.tsx

import { usePendingApprovals } from "@/lib/telegram-queries"
import { Badge } from "@/components/ui/badge"
import { Card, CardContent, CardHeader } from "@/components/ui/card"
import { useState, useEffect } from "react"
import { cn } from "@/lib/utils"

export function PendingApprovalsList() {
  const { data: approvals, isLoading } = usePendingApprovals()

  if (isLoading) {
    return <div className="animate-pulse">Loading approvals...</div>
  }

  if (!approvals?.length) {
    return (
      <p className="text-sm text-muted-foreground text-center py-8">
        No pending approvals
      </p>
    )
  }

  return (
    <div className="space-y-4">
      {approvals.map((approval) => (
        <ApprovalCard key={approval.id} approval={approval} />
      ))}
    </div>
  )
}

interface ApprovalCardProps {
  approval: {
    id: string
    symbol: string
    action: "buy" | "sell"
    quantity: number
    estimatedPrice: number
    rationale: string
    confidence: number
    timeRemaining: number
    createdAt: string
  }
}

function ApprovalCard({ approval }: ApprovalCardProps) {
  const [timeLeft, setTimeLeft] = useState(approval.timeRemaining)

  useEffect(() => {
    const timer = setInterval(() => {
      setTimeLeft((t) => Math.max(0, t - 1))
    }, 1000)
    return () => clearInterval(timer)
  }, [])

  const minutes = Math.floor(timeLeft / 60)
  const seconds = timeLeft % 60
  const isExpiringSoon = timeLeft < 60
  const totalValue = (approval.quantity * approval.estimatedPrice).toFixed(2)

  return (
    <Card className={cn(isExpiringSoon && "border-yellow-500")}>
      <CardHeader className="pb-2">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <span className="font-mono text-lg font-bold">{approval.symbol}</span>
            <Badge variant={approval.action === "buy" ? "default" : "destructive"}>
              {approval.action.toUpperCase()}
            </Badge>
          </div>
          <Badge variant="outline" className={cn(isExpiringSoon && "text-yellow-600")}>
            {minutes}:{seconds.toString().padStart(2, "0")}
          </Badge>
        </div>
      </CardHeader>
      <CardContent>
        <div className="grid grid-cols-2 gap-2 text-sm">
          <div>
            <span className="text-muted-foreground">Qty:</span> {approval.quantity}
          </div>
          <div>
            <span className="text-muted-foreground">Price:</span> ${approval.estimatedPrice.toFixed(2)}
          </div>
          <div>
            <span className="text-muted-foreground">Total:</span> ${totalValue}
          </div>
          <div>
            <span className="text-muted-foreground">Conf:</span> {(approval.confidence * 100).toFixed(0)}%
          </div>
        </div>
        <p className="text-sm text-muted-foreground mt-2 line-clamp-2">
          {approval.rationale}
        </p>
        <p className="text-xs text-muted-foreground mt-2">
          Approve/reject via Telegram
        </p>
      </CardContent>
    </Card>
  )
}
```
