### pending_approvals table

```ts
// packages/data-ops/src/drizzle/schema.ts

import { pgTable, text, timestamp, uuid, real, integer, pgEnum } from "drizzle-orm/pg-core"
import { auth_user } from "./auth-schema"

export const approvalStatusEnum = pgEnum("approval_status", [
  "pending",
  "approved",
  "rejected",
  "expired"
])

export const pending_approvals = pgTable("pending_approvals", {
  id: uuid("id").defaultRandom().primaryKey(),
  userId: text("user_id")
    .notNull()
    .references(() => auth_user.id, { onDelete: "cascade" }),
  symbol: text("symbol").notNull(),
  action: text("action").notNull().$type<"buy" | "sell">(),
  quantity: real("quantity").notNull(),
  estimatedPrice: real("estimated_price").notNull(),
  rationale: text("rationale").notNull(),
  confidence: real("confidence").notNull(),
  status: approvalStatusEnum("status").notNull().default("pending"),
  telegramMessageId: integer("telegram_message_id"),
  expiresAt: timestamp("expires_at").notNull(),
  approvedAt: timestamp("approved_at"),
  rejectedAt: timestamp("rejected_at"),
  rejectedReason: text("rejected_reason"),
  createdAt: timestamp("created_at").defaultNow().notNull(),
}, (table) => ({
  userIdIdx: index("pending_approvals_user_id_idx").on(table.userId),
  statusIdx: index("pending_approvals_status_idx").on(table.status),
}))

export type PendingApprovalRecord = typeof pending_approvals.$inferSelect
export type NewPendingApproval = typeof pending_approvals.$inferInsert
```

```ts
// packages/data-ops/src/drizzle/schema.ts

export const notification_settings = pgTable("notification_settings", {
  userId: text("user_id")
    .primaryKey()
    .references(() => auth_user.id, { onDelete: "cascade" }),
  enableTradeApprovals: boolean("enable_trade_approvals").notNull().default(true),
  enableTradeResults: boolean("enable_trade_results").notNull().default(true),
  enableDailySummary: boolean("enable_daily_summary").notNull().default(true),
  enableRiskAlerts: boolean("enable_risk_alerts").notNull().default(true),
  enableSignals: boolean("enable_signals").notNull().default(false),
  enablePositionUpdates: boolean("enable_position_updates").notNull().default(false),
  dailySummaryTime: text("daily_summary_time").notNull().default("17:00"),
  positionUpdateFrequency: text("position_update_frequency")
    .notNull()
    .default("never")
    .$type<"never" | "hourly" | "daily">(),
  quietHoursStart: text("quiet_hours_start"),
  quietHoursEnd: text("quiet_hours_end"),
  updatedAt: timestamp("updated_at").defaultNow().$onUpdate(() => new Date()).notNull(),
})

export type NotificationSettingsRecord = typeof notification_settings.$inferSelect
```

### Migration SQL

```sql
-- 0011_telegram_approvals.sql

CREATE TYPE approval_status AS ENUM ('pending', 'approved', 'rejected', 'expired');

CREATE TABLE pending_approvals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  symbol TEXT NOT NULL,
  action TEXT NOT NULL CHECK (action IN ('buy', 'sell')),
  quantity REAL NOT NULL,
  estimated_price REAL NOT NULL,
  rationale TEXT NOT NULL,
  confidence REAL NOT NULL,
  status approval_status NOT NULL DEFAULT 'pending',
  telegram_message_id INTEGER,
  expires_at TIMESTAMP NOT NULL,
  approved_at TIMESTAMP,
  rejected_at TIMESTAMP,
  rejected_reason TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_pending_approvals_user_id ON pending_approvals(user_id);
CREATE INDEX idx_pending_approvals_status ON pending_approvals(status);
CREATE INDEX idx_pending_approvals_expires ON pending_approvals(expires_at) WHERE status = 'pending';

CREATE TABLE notification_settings (
  user_id TEXT PRIMARY KEY REFERENCES auth_user(id) ON DELETE CASCADE,
  enable_trade_approvals BOOLEAN NOT NULL DEFAULT true,
  enable_trade_results BOOLEAN NOT NULL DEFAULT true,
  enable_daily_summary BOOLEAN NOT NULL DEFAULT true,
  enable_risk_alerts BOOLEAN NOT NULL DEFAULT true,
  enable_signals BOOLEAN NOT NULL DEFAULT false,
  enable_position_updates BOOLEAN NOT NULL DEFAULT false,
  daily_summary_time TEXT NOT NULL DEFAULT '17:00',
  position_update_frequency TEXT NOT NULL DEFAULT 'never',
  quiet_hours_start TEXT,
  quiet_hours_end TEXT,
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/telegram.ts

import { z } from "zod"

export const ApprovalStatusSchema = z.enum(["pending", "approved", "rejected", "expired"])

export const PendingApprovalSchema = z.object({
  id: z.string().uuid(),
  userId: z.string(),
  symbol: z.string(),
  action: z.enum(["buy", "sell"]),
  quantity: z.number().positive(),
  estimatedPrice: z.number().positive(),
  rationale: z.string(),
  confidence: z.number().min(0).max(1),
  status: ApprovalStatusSchema,
  telegramMessageId: z.number().nullable(),
  expiresAt: z.date(),
  approvedAt: z.date().nullable(),
  rejectedAt: z.date().nullable(),
  rejectedReason: z.string().nullable(),
  createdAt: z.date(),
})

export const CreateApprovalSchema = z.object({
  symbol: z.string().min(1).max(10),
  action: z.enum(["buy", "sell"]),
  quantity: z.number().positive(),
  estimatedPrice: z.number().positive(),
  rationale: z.string().max(500),
  confidence: z.number().min(0).max(1),
  ttlMinutes: z.number().min(1).max(60).default(15),
})

export const ApprovalCallbackSchema = z.object({
  approvalId: z.string().uuid(),
  action: z.enum(["approve", "reject"]),
})

export const NotificationSettingsSchema = z.object({
  enableTradeApprovals: z.boolean().optional(),
  enableTradeResults: z.boolean().optional(),
  enableDailySummary: z.boolean().optional(),
  enableRiskAlerts: z.boolean().optional(),
  enableSignals: z.boolean().optional(),
  enablePositionUpdates: z.boolean().optional(),
  dailySummaryTime: z.string().regex(/^\d{2}:\d{2}$/).optional(),
  positionUpdateFrequency: z.enum(["never", "hourly", "daily"]).optional(),
  quietHoursStart: z.string().regex(/^\d{2}:\d{2}$/).nullable().optional(),
  quietHoursEnd: z.string().regex(/^\d{2}:\d{2}$/).nullable().optional(),
})

export const TelegramWebhookUpdateSchema = z.object({
  update_id: z.number(),
  message: z.object({
    message_id: z.number(),
    from: z.object({ id: z.number(), username: z.string().optional() }),
    chat: z.object({ id: z.number(), type: z.string() }),
    text: z.string().optional(),
  }).optional(),
  callback_query: z.object({
    id: z.string(),
    from: z.object({ id: z.number(), username: z.string().optional() }),
    message: z.object({
      message_id: z.number(),
      chat: z.object({ id: z.number() }),
    }),
    data: z.string(),
  }).optional(),
})

export type PendingApproval = z.infer<typeof PendingApprovalSchema>
export type CreateApprovalInput = z.infer<typeof CreateApprovalSchema>
export type NotificationSettingsInput = z.infer<typeof NotificationSettingsSchema>
```
