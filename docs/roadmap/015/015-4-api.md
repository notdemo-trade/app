## API Handlers

```ts
// apps/data-service/src/hono/handlers/leaderboard-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  LeaderboardQuerySchema,
  UpdateSettingsRequestSchema,
  LeaderboardPeriodSchema,
  TradingModeSchema,
} from "@repo/data-ops/zod-schema/leaderboard"
import {
  getLeaderboardSettings,
  updateLeaderboardSettings,
  checkAliasAvailable,
  getLeaderboard,
  getUserRanks,
  getUserByAlias,
} from "@repo/data-ops/db/queries/leaderboard"
import { z } from "zod"

const leaderboardRouter = new Hono<{ Bindings: Env }>()

// Get user's settings
leaderboardRouter.get("/settings", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const settings = await getLeaderboardSettings(c.get("db"), userId)

  if (!settings) {
    return c.json({ data: null })
  }

  return c.json({ data: settings })
})

// Update settings
leaderboardRouter.patch(
  "/settings",
  apiTokenMiddleware,
  zValidator("json", UpdateSettingsRequestSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    // Check alias availability if changing
    if (data.customAlias) {
      const available = await checkAliasAvailable(c.get("db"), data.customAlias, userId)
      if (!available) {
        return c.json({ error: "Alias already taken" }, 400)
      }

      // Check 7-day cooldown
      const settings = await getLeaderboardSettings(c.get("db"), userId)
      if (settings?.aliasLastChangedAt) {
        const lastChanged = new Date(settings.aliasLastChangedAt)
        const cooldownEnd = new Date(lastChanged.getTime() + 7 * 24 * 60 * 60 * 1000)
        if (new Date() < cooldownEnd) {
          return c.json({ error: "Alias change on cooldown", cooldownEnds: cooldownEnd.toISOString() }, 400)
        }
      }
    }

    await updateLeaderboardSettings(c.get("db"), userId, data)
    return c.json({ success: true })
  }
)

// Get leaderboard (public)
leaderboardRouter.get(
  "/:period",
  zValidator("param", z.object({ period: LeaderboardPeriodSchema })),
  zValidator("query", LeaderboardQuerySchema.pick({ mode: true, page: true, limit: true, search: true })),
  async (c) => {
    const { period } = c.req.valid("param")
    const { mode, page, limit, search } = c.req.valid("query")

    const { entries, total } = await getLeaderboard(c.get("db"), period, mode, {
      page,
      limit,
      search,
    })

    return c.json({
      data: entries,
      meta: {
        page,
        limit,
        total,
        totalPages: Math.ceil(total / limit),
      },
    })
  }
)

// Get user profile by alias (public)
leaderboardRouter.get(
  "/:period/user/:alias",
  zValidator("param", z.object({ period: LeaderboardPeriodSchema, alias: z.string() })),
  zValidator("query", z.object({ mode: TradingModeSchema.default("paper") })),
  async (c) => {
    const { alias } = c.req.valid("param")
    const { mode } = c.req.valid("query")

    const profile = await getUserByAlias(c.get("db"), alias, mode)
    if (!profile) {
      return c.json({ error: "User not found" }, 404)
    }

    return c.json({ data: profile })
  }
)

// Get my ranks (auth required)
leaderboardRouter.get("/my-rank", apiTokenMiddleware, async (c) => {
  const userId = c.get("userId")
  const settings = await getLeaderboardSettings(c.get("db"), userId)

  if (!settings || !settings.isEnabled) {
    return c.json({ data: null })
  }

  const ranks = await getUserRanks(c.get("db"), userId, settings.tradingMode)
  return c.json({ data: { alias: settings.customAlias ?? settings.generatedAlias, ranks } })
})

export { leaderboardRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/leaderboard/settings` | Bearer | Get user's settings |
| PATCH | `/api/leaderboard/settings` | Bearer | Update settings (opt-in, alias) |
| GET | `/api/leaderboard/:period` | None | Get leaderboard (public) |
| GET | `/api/leaderboard/:period/user/:alias` | None | Get user's profile |
| GET | `/api/leaderboard/my-rank` | Bearer | Get my current ranks |

## Server Functions

```ts
// apps/user-application/src/core/functions/leaderboard/direct.ts

import { createServerFn } from "@tanstack/react-start"
import { protectedFunctionMiddleware } from "@/core/middleware/auth"
import {
  LeaderboardQuerySchema,
  UpdateSettingsRequestSchema,
  LeaderboardPeriodSchema,
  TradingModeSchema,
} from "@repo/data-ops/zod-schema/leaderboard"
import {
  getLeaderboardSettings,
  updateLeaderboardSettings,
  checkAliasAvailable,
  getLeaderboard,
  getUserRanks,
  getUserByAlias,
} from "@repo/data-ops/db/queries/leaderboard"
import { z } from "zod"

export const getMyLeaderboardSettings = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .handler(async ({ context }) => {
    return getLeaderboardSettings(context.db, context.session.user.id)
  })

export const updateMyLeaderboardSettings = createServerFn({ method: "POST" })
  .middleware([protectedFunctionMiddleware])
  .validator(UpdateSettingsRequestSchema)
  .handler(async ({ data, context }) => {
    const userId = context.session.user.id

    if (data.customAlias) {
      const available = await checkAliasAvailable(context.db, data.customAlias, userId)
      if (!available) {
        throw new Error("Alias already taken")
      }
    }

    await updateLeaderboardSettings(context.db, userId, data)
    return { success: true }
  })

export const getLeaderboardData = createServerFn({ method: "GET" })
  .validator(LeaderboardQuerySchema)
  .handler(async ({ data, context }) => {
    return getLeaderboard(context.db, data.period, data.mode, {
      page: data.page,
      limit: data.limit,
      search: data.search,
    })
  })

export const getMyRanks = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ mode: TradingModeSchema.default("paper") }))
  .handler(async ({ data, context }) => {
    return getUserRanks(context.db, context.session.user.id, data.mode)
  })

export const getTraderProfile = createServerFn({ method: "GET" })
  .validator(z.object({ alias: z.string(), mode: TradingModeSchema.default("paper") }))
  .handler(async ({ data, context }) => {
    return getUserByAlias(context.db, data.alias, data.mode)
  })

export const checkAliasAvailability = createServerFn({ method: "GET" })
  .middleware([protectedFunctionMiddleware])
  .validator(z.object({ alias: z.string() }))
  .handler(async ({ data, context }) => {
    const available = await checkAliasAvailable(context.db, data.alias, context.session.user.id)
    return { available }
  })
```
