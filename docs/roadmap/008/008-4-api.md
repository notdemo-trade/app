## API Handlers

```ts
// apps/data-service/src/hono/handlers/order-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { apiTokenMiddleware } from "../middleware/auth"
import {
  CreateOrderSchema,
  CreateOptionsOrderSchema,
  ListOrdersSchema,
  ClosePositionSchema,
  GetOptionsChainSchema,
  GetOptionsByDeltaSchema,
} from "@repo/data-ops/orders"
import { getAlpacaTradingClient, getAlpacaOptionsClient } from "../utils/alpaca"
import { previewOrder, previewOptionsOrder, submitOrder } from "@repo/data-ops/services/order-service"
import { z } from "zod"

const orderRouter = new Hono<{ Bindings: Env }>()

// Preview stock/crypto order
orderRouter.post(
  "/preview",
  apiTokenMiddleware,
  zValidator("json", CreateOrderSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const result = await previewOrder(trading, {
        symbol: data.symbol,
        side: data.side,
        type: data.type,
        time_in_force: data.time_in_force,
        qty: data.qty,
        notional: data.notional,
        limit_price: data.limit_price,
        stop_price: data.stop_price,
      })

      return c.json({ data: result })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Submit order
orderRouter.post(
  "/submit",
  apiTokenMiddleware,
  zValidator("json", CreateOrderSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const order = await submitOrder(trading, {
        symbol: data.symbol,
        side: data.side,
        type: data.type,
        time_in_force: data.time_in_force,
        qty: data.qty,
        notional: data.notional,
        limit_price: data.limit_price,
        stop_price: data.stop_price,
      })

      return c.json({
        data: {
          id: order.id,
          symbol: order.symbol,
          status: order.status,
          side: order.side,
          qty: order.qty,
        },
      })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// List orders
orderRouter.get(
  "/",
  apiTokenMiddleware,
  zValidator("query", ListOrdersSchema),
  async (c) => {
    const userId = c.get("userId")
    const { status, limit, symbol } = c.req.valid("query")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const orders = await trading.listOrders({
        status,
        limit,
        symbols: symbol ? [symbol] : undefined,
      })

      return c.json({ data: orders })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Cancel order
orderRouter.delete(
  "/:orderId",
  apiTokenMiddleware,
  zValidator("param", z.object({ orderId: z.string() })),
  async (c) => {
    const userId = c.get("userId")
    const { orderId } = c.req.valid("param")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      await trading.cancelOrder(orderId)
      return c.json({ data: { message: "Order cancelled" } })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Close position
orderRouter.post(
  "/close-position",
  apiTokenMiddleware,
  zValidator("json", ClosePositionSchema),
  async (c) => {
    const userId = c.get("userId")
    const { symbol, qty, percentage } = c.req.valid("json")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const order = await trading.closePosition(symbol, qty, percentage ? percentage / 100 : undefined)
      return c.json({
        data: {
          id: order.id,
          symbol: order.symbol,
          status: order.status,
        },
      })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Options chain
orderRouter.get(
  "/options/chain",
  apiTokenMiddleware,
  zValidator("query", GetOptionsChainSchema),
  async (c) => {
    const userId = c.get("userId")
    const { underlying, expiration } = c.req.valid("query")

    const options = await getAlpacaOptionsClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!options) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const chain = await options.getChain(underlying, expiration)
      return c.json({ data: chain })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Options by delta
orderRouter.get(
  "/options/by-delta",
  apiTokenMiddleware,
  zValidator("query", GetOptionsByDeltaSchema),
  async (c) => {
    const userId = c.get("userId")
    const { underlying, expiration, type, min_delta, max_delta } = c.req.valid("query")

    const options = await getAlpacaOptionsClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!options) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const contracts = await options.findContractsByDelta(
        underlying,
        type,
        expiration,
        min_delta,
        max_delta
      )
      return c.json({ data: contracts })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Options expirations
orderRouter.get(
  "/options/expirations/:underlying",
  apiTokenMiddleware,
  zValidator("param", z.object({ underlying: z.string() })),
  async (c) => {
    const userId = c.get("userId")
    const { underlying } = c.req.valid("param")

    const options = await getAlpacaOptionsClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!options) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const expirations = await options.getExpirations(underlying)
      return c.json({ data: expirations })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Preview options order
orderRouter.post(
  "/options/preview",
  apiTokenMiddleware,
  zValidator("json", CreateOptionsOrderSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    const options = await getAlpacaOptionsClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading || !options) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const result = await previewOptionsOrder(trading, options, data)
      return c.json({ data: result })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

// Submit options order
orderRouter.post(
  "/options/submit",
  apiTokenMiddleware,
  zValidator("json", CreateOptionsOrderSchema),
  async (c) => {
    const userId = c.get("userId")
    const data = c.req.valid("json")

    const trading = await getAlpacaTradingClient(c.get("db"), userId, c.env.CREDENTIALS_ENCRYPTION_KEY)
    if (!trading) {
      return c.json({ error: "Alpaca credentials not configured" }, 400)
    }

    try {
      const order = await trading.createOrder({
        symbol: data.contract_symbol,
        side: data.side,
        type: data.order_type,
        time_in_force: data.time_in_force,
        qty: data.qty,
        limit_price: data.limit_price,
      })

      return c.json({
        data: {
          id: order.id,
          symbol: order.symbol,
          status: order.status,
        },
      })
    } catch (err) {
      return c.json({ error: String(err) }, 500)
    }
  }
)

export { orderRouter }
```
