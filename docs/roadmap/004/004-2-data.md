# Phase 4: Signal Gathering Agents — Part 2: Data Model

## Shared PG Schema

All signal agents write to the same PG tables. `source_agent` column identifies origin.

```sql
-- packages/data-ops/migrations/0004_signal_gathering.sql

CREATE TABLE watchlists (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id TEXT NOT NULL REFERENCES auth_user(id) ON DELETE CASCADE,
  symbol TEXT NOT NULL,
  added_at TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE(user_id, symbol)
);

CREATE INDEX idx_watchlists_user ON watchlists(user_id);

CREATE TABLE raw_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_agent TEXT NOT NULL,     -- 'stocktwits', 'twitter', 'sec', 'fred'
  source_id TEXT NOT NULL,
  symbol TEXT,                     -- null for FRED (uses series_id)
  series_id TEXT,                  -- FRED series ID
  raw_content JSONB NOT NULL,
  ingested_at TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE(source_agent, source_id)
);

CREATE INDEX idx_raw_events_source ON raw_events(source_agent);
CREATE INDEX idx_raw_events_symbol ON raw_events(symbol);
CREATE INDEX idx_raw_events_ingested ON raw_events(ingested_at);

-- NOTE: signals DDL extracted to Phase 5 (005-2-data.md) for MVP scope.
-- Phase 4 and Phase 5 share the same signals table definition.
CREATE TABLE signals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_agent TEXT NOT NULL,
  symbol TEXT,
  series_id TEXT,
  signal_type TEXT NOT NULL,
  direction TEXT NOT NULL,
  strength DECIMAL(3,2) NOT NULL CHECK (strength >= 0 AND strength <= 1),
  summary TEXT,
  metadata JSONB,
  raw_event_id UUID REFERENCES raw_events(id) ON DELETE SET NULL,
  created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_signals_source ON signals(source_agent);
CREATE INDEX idx_signals_symbol ON signals(symbol);
CREATE INDEX idx_signals_series ON signals(series_id);
CREATE INDEX idx_signals_created ON signals(created_at DESC);
CREATE INDEX idx_signals_symbol_created ON signals(symbol, created_at DESC);

CREATE TABLE ingestion_state (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_agent TEXT NOT NULL,
  instance_id TEXT NOT NULL,       -- userId, ticker, or seriesId
  last_poll_at TIMESTAMP,
  last_event_id TEXT,
  error_count INT DEFAULT 0,
  last_error TEXT,
  UNIQUE(source_agent, instance_id)
);
```

### Key changes from original

- Removed `user_id` FK from `raw_events` and `signals` — shared agents (SEC, FRED) write data not owned by a single user
- Added `source_agent` column replacing `source`
- Added `series_id` for FRED data
- `raw_events` unique on `(source_agent, source_id)` — global dedup across users
- `ingestion_state` keyed by `(source_agent, instance_id)` instead of `(user_id, source)`

### Drizzle Schema

```ts
// packages/data-ops/src/drizzle/schema/signals.ts

export const watchlists = pgTable("watchlists", {
  id: uuid("id").defaultRandom().primaryKey(),
  userId: text("user_id")
    .notNull()
    .references(() => auth_user.id, { onDelete: "cascade" }),
  symbol: text("symbol").notNull(),
  addedAt: timestamp("added_at").defaultNow().notNull(),
}, (table) => ({
  uniqueUserSymbol: unique().on(table.userId, table.symbol),
}))

export const rawEvents = pgTable("raw_events", {
  id: uuid("id").defaultRandom().primaryKey(),
  sourceAgent: text("source_agent").notNull(),
  sourceId: text("source_id").notNull(),
  symbol: text("symbol"),
  seriesId: text("series_id"),
  rawContent: jsonb("raw_content").notNull(),
  ingestedAt: timestamp("ingested_at").defaultNow().notNull(),
}, (table) => ({
  uniqueSourceId: unique().on(table.sourceAgent, table.sourceId),
}))

export const signals = pgTable("signals", {
  id: uuid("id").defaultRandom().primaryKey(),
  sourceAgent: text("source_agent").notNull(),
  symbol: text("symbol"),
  seriesId: text("series_id"),
  signalType: text("signal_type").notNull(),
  direction: text("direction").notNull(),
  strength: decimal("strength", { precision: 3, scale: 2 }).notNull(),
  summary: text("summary"),
  metadata: jsonb("metadata"),
  rawEventId: uuid("raw_event_id")
    .references(() => rawEvents.id, { onDelete: "set null" }),
  createdAt: timestamp("created_at").defaultNow().notNull(),
})

export const ingestionState = pgTable("ingestion_state", {
  id: uuid("id").defaultRandom().primaryKey(),
  sourceAgent: text("source_agent").notNull(),
  instanceId: text("instance_id").notNull(),
  lastPollAt: timestamp("last_poll_at"),
  lastEventId: text("last_event_id"),
  errorCount: integer("error_count").default(0),
  lastError: text("last_error"),
}, (table) => ({
  uniqueAgentInstance: unique().on(table.sourceAgent, table.instanceId),
}))
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/signals.ts

import { z } from "zod"

export const SignalSourceSchema = z.enum(["stocktwits", "twitter", "sec", "fred"])
export const SignalDirectionSchema = z.enum(["bullish", "bearish", "neutral"])
export const SignalTypeSchema = z.enum([
  "sentiment", "trending", "mention", "influencer", "filing", "insider", "macro"
])

export const SignalSchema = z.object({
  id: z.string().uuid(),
  sourceAgent: SignalSourceSchema,
  symbol: z.string().nullable(),
  seriesId: z.string().nullable(),
  signalType: SignalTypeSchema,
  direction: SignalDirectionSchema,
  strength: z.number().min(0).max(1),
  summary: z.string().nullable(),
  metadata: z.record(z.unknown()).nullable(),
  createdAt: z.date(),
})

export const WatchlistItemSchema = z.object({
  id: z.string().uuid(),
  userId: z.string(),
  symbol: z.string(),
  addedAt: z.date(),
})

export const AddToWatchlistRequestSchema = z.object({
  symbol: z.string().min(1).max(10).transform(s => s.toUpperCase().trim()),
})

export const RemoveFromWatchlistRequestSchema = z.object({
  symbol: z.string().min(1).max(10).transform(s => s.toUpperCase().trim()),
})

export const GetSignalsRequestSchema = z.object({
  symbol: z.string().optional(),
  source: SignalSourceSchema.optional(),
  limit: z.coerce.number().int().min(1).max(100).optional().default(50),
  since: z.coerce.date().optional(),
})

export type Signal = z.infer<typeof SignalSchema>
export type WatchlistItem = z.infer<typeof WatchlistItemSchema>
```
