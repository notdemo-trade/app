# Phase 4: Signal Gathering Agents — Part 4: API & Server Functions

## API Handlers

Hono handlers proxy to agents via `getAgentByName()` RPC. Watchlist CRUD remains direct DB.

```ts
// apps/data-service/src/hono/handlers/signals-handlers.ts

import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { getAgentByName } from "agents"
import { apiTokenMiddleware } from "../middleware/auth"
import { AddToWatchlistRequestSchema, RemoveFromWatchlistRequestSchema, GetSignalsRequestSchema } from "@repo/data-ops/signals"
import * as watchlistQueries from "@repo/data-ops/watchlist"
import type { SignalAgentRPC } from "@repo/data-ops/agents/signal/types"

const signalsRouter = new Hono<{ Bindings: Env }>()

// --- Watchlist (direct DB) ---

signalsRouter.get("/watchlist", apiTokenMiddleware, async (c) => {
  const watchlist = await watchlistQueries.getUserWatchlist(c.get("db"), c.get("userId"))
  return c.json({ data: watchlist })
})

signalsRouter.post("/watchlist", apiTokenMiddleware, zValidator("json", AddToWatchlistRequestSchema), async (c) => {
  const result = await watchlistQueries.addToWatchlist(c.get("db"), c.get("userId"), c.req.valid("json").symbol)
  if ("error" in result) return c.json({ error: result.error }, 400)
  return c.json({ data: result }, 201)
})

signalsRouter.delete("/watchlist/:symbol", apiTokenMiddleware, async (c) => {
  const deleted = await watchlistQueries.removeFromWatchlist(c.get("db"), c.get("userId"), c.req.param("symbol"))
  if (!deleted) return c.json({ error: "Symbol not in watchlist" }, 404)
  return c.json({ success: true })
})

// --- Signals (via agent RPC) ---

signalsRouter.get("/signals", apiTokenMiddleware, zValidator("query", GetSignalsRequestSchema), async (c) => {
  const { source, symbol, limit, since } = c.req.valid("query")
  // Query shared PG directly for cross-agent signals
  const signals = await getSignals(c.get("db"), { source, symbol, limit, since })
  return c.json({ data: signals })
})

signalsRouter.get("/signals/feed", apiTokenMiddleware, async (c) => {
  const signals = await getSignalsForWatchlist(c.get("db"), c.get("userId"), 50)
  return c.json({ data: signals })
})

signalsRouter.get("/signals/stats/:symbol", apiTokenMiddleware, async (c) => {
  const symbol = c.req.param("symbol").toUpperCase()
  const stats = await getSignalStats(c.get("db"), symbol)
  return c.json({ data: stats })
})

// --- Agent status ---

signalsRouter.get("/agents/:source/status", apiTokenMiddleware, async (c) => {
  const source = c.req.param("source")
  const userId = c.get("userId")
  const agent = await getAgentByName<SignalAgentRPC>(getAgentBinding(c.env, source), userId)
  return c.json(await agent.getStatus())
})

// Manual trigger
signalsRouter.post("/agents/:source/ingest", apiTokenMiddleware, async (c) => {
  const source = c.req.param("source")
  const userId = c.get("userId")
  const agent = await getAgentByName<SignalAgentRPC>(getAgentBinding(c.env, source), userId)
  const result = await agent.ingestNow()
  return c.json({ data: result })
})

export { signalsRouter }
```

### Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | `/api/watchlist` | Bearer | Get user's watchlist |
| POST | `/api/watchlist` | Bearer | Add symbol |
| DELETE | `/api/watchlist/:symbol` | Bearer | Remove symbol |
| GET | `/api/signals` | Bearer | Query signals (filters: source, symbol, since) |
| GET | `/api/signals/feed` | Bearer | Watchlist signal feed |
| GET | `/api/signals/stats/:symbol` | Bearer | Signal stats per symbol |
| GET | `/api/agents/:source/status` | Bearer | Agent ingestion status |
| POST | `/api/agents/:source/ingest` | Bearer | Manual ingest trigger |
