### SignalFeed

Updated to show `sourceAgent` badge per signal.

```tsx
// apps/user-application/src/components/signals/signal-feed.tsx

function SignalCard({ signal }: { signal: Signal }) {
  const sourceColors: Record<string, string> = {
    stocktwits: "bg-blue-100 text-blue-800",
    twitter: "bg-sky-100 text-sky-800",
    sec: "bg-amber-100 text-amber-800",
    fred: "bg-purple-100 text-purple-800",
  }

  return (
    <div className="flex items-start gap-3 p-3 rounded-lg border">
      <div className="flex flex-col gap-1">
        {signal.symbol && <Badge variant="outline" className="font-mono">{signal.symbol}</Badge>}
        {signal.seriesId && <Badge variant="outline" className="font-mono">{signal.seriesId}</Badge>}
      </div>
      <div className="flex-1 min-w-0">
        <div className="flex items-center gap-2">
          <span className={cn("text-xs px-2 py-0.5 rounded", sourceColors[signal.sourceAgent])}>
            {signal.sourceAgent}
          </span>
          <span className={cn("text-xs px-2 py-0.5 rounded", directionColors[signal.direction])}>
            {signal.direction}
          </span>
          <span className="text-xs text-muted-foreground">{formatRelativeTime(signal.createdAt)}</span>
        </div>
        {signal.summary && <p className="text-sm mt-1 text-muted-foreground line-clamp-2">{signal.summary}</p>}
      </div>
      <div className="text-right">
        <span className="text-xs text-muted-foreground">Strength</span>
        <div className="font-mono text-sm">{(signal.strength * 100).toFixed(0)}%</div>
      </div>
    </div>
  )
}
```
