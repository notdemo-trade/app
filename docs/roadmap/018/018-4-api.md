## Rate Limiting

```ts
// apps/data-service/src/middleware/rate-limit.ts

import { Context, Next } from "hono"
import { RateLimiter } from "@repo/data-ops/utils/rate-limiter"

const RATE_LIMITS = {
  default: { requests: 100, windowMs: 60_000 },      // 100/min
  trading: { requests: 20, windowMs: 60_000 },        // 20/min for orders
  analysis: { requests: 30, windowMs: 60_000 },       // 30/min for LLM calls
  auth: { requests: 10, windowMs: 60_000 },           // 10/min for login
}

type RateLimitTier = keyof typeof RATE_LIMITS

export function rateLimitMiddleware(tier: RateLimitTier = "default") {
  return async (c: Context, next: Next) => {
    const userId = c.get("userId") as string | undefined
    const ip = c.req.header("cf-connecting-ip") ?? "unknown"
    const key = userId ?? ip

    const limiter = new RateLimiter(c.env.RATE_LIMIT_KV)
    const limit = RATE_LIMITS[tier]

    const result = await limiter.check(key, tier, limit.requests, limit.windowMs)

    c.header("X-RateLimit-Limit", String(limit.requests))
    c.header("X-RateLimit-Remaining", String(result.remaining))
    c.header("X-RateLimit-Reset", String(result.resetAt))

    if (!result.allowed) {
      return c.json(
        { error: "Rate limit exceeded", retryAfter: result.retryAfterMs },
        429
      )
    }

    await next()
  }
}
```

### Rate Limiter Implementation

```ts
// packages/data-ops/src/utils/rate-limiter.ts

export interface RateLimitResult {
  allowed: boolean
  remaining: number
  resetAt: number
  retryAfterMs?: number
}

export class RateLimiter {
  constructor(private kv: KVNamespace) {}

  async check(
    key: string,
    tier: string,
    maxRequests: number,
    windowMs: number
  ): Promise<RateLimitResult> {
    const now = Date.now()
    const windowStart = Math.floor(now / windowMs) * windowMs
    const kvKey = `ratelimit:${tier}:${key}:${windowStart}`

    const current = await this.kv.get(kvKey)
    const count = current ? parseInt(current, 10) : 0

    if (count >= maxRequests) {
      const resetAt = windowStart + windowMs
      return {
        allowed: false,
        remaining: 0,
        resetAt,
        retryAfterMs: resetAt - now,
      }
    }

    await this.kv.put(kvKey, String(count + 1), {
      expirationTtl: Math.ceil(windowMs / 1000) + 60,
    })

    return {
      allowed: true,
      remaining: maxRequests - count - 1,
      resetAt: windowStart + windowMs,
    }
  }
}
```

## Rate Limiting (P0 Blocker)

Global rate limits across all endpoints.

### Implementation (Already Documented Above)

Phase 18 implements comprehensive rate limiting strategy referenced throughout previous phases. System uses `hono-rate-limiter` with Cloudflare KV for distributed state.

**Applied Limits**:
- Token generation (Phase 1): 5/min
- Credential validation (Phase 2): 10/min
- LLM analysis (Phase 6): 10/min
- Strategy testing (Phase 7): 5/min
- Notification tests (Phase 16): 3/min
- Default API: 100/min
- Trading operations: 20/min
- Auth endpoints: 10/min

Add binding to `wrangler.jsonc`:

```jsonc
{
  "production": {
    "kv_namespaces": [
      {
        "binding": "RATE_LIMIT_KV",
        "id": "<production-kv-id>"

  }
}
```

### Global Application

```ts
// apps/data-service/src/hono/app.ts

import { tokenRateLimit } from './middleware/rate-limit'
import { credentialValidationLimit } from './middleware/rate-limit'
import { llmAnalysisLimit } from './middleware/rate-limit'
import { strategyTestLimit } from './middleware/rate-limit'
import { notificationTestLimit } from './middleware/rate-limit'

// Apply specific limits
app.post('/api/tokens', tokenRateLimit, ...)
app.put('/api/credentials', credentialValidationLimit, ...)
app.post('/api/llm/analyze', llmAnalysisLimit, ...)
app.post('/api/strategies/:id/test', strategyTestLimit, ...)
app.post('/api/notifications/test', notificationTestLimit, ...)
```

**Why Global Strategy**: Layered approach - specific limits for expensive operations (LLM, notifications), generous limits for reads, auth protection on sensitive endpoints.
