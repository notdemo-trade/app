## Database Schema

```ts
// packages/data-ops/src/db/schema/approvals.ts

import { pgTable, text, timestamp, uuid, jsonb, index } from "drizzle-orm/pg-core"
import { users } from "./users"
import type { OrderPreview, PolicyResult } from "../../policy/types"

export const approvals = pgTable(
  "approvals",
  {
    id: uuid("id").primaryKey(),
    userId: uuid("user_id")
      .notNull()
      .references(() => users.id, { onDelete: "cascade" }),
    previewHash: text("preview_hash").notNull(),
    orderParams: jsonb("order_params").$type<OrderPreview>().notNull(),
    policyResult: jsonb("policy_result").$type<PolicyResult>().notNull(),
    token: text("token").notNull().unique(),
    expiresAt: timestamp("expires_at").notNull(),
    source: text("source").notNull().default("manual"), // manual, llm, agent
    createdAt: timestamp("created_at").defaultNow().notNull(),
    usedAt: timestamp("used_at"),
    rejectedAt: timestamp("rejected_at"),
    rejectionReason: text("rejection_reason"),
  },
  (table) => ({
    userIdIdx: index("approvals_user_id_idx").on(table.userId),
    tokenIdx: index("approvals_token_idx").on(table.token),
    expiresAtIdx: index("approvals_expires_at_idx").on(table.expiresAt),
  })
)

export type ApprovalRecord = typeof approvals.$inferSelect
export type NewApprovalRecord = typeof approvals.$inferInsert
```

## Zod Schemas

```ts
// packages/data-ops/src/zod-schema/approvals.ts

import { z } from "zod"

export const ApproveOrderSchema = z.object({
  approvalId: z.string().uuid(),
})

export const RejectApprovalSchema = z.object({
  approvalId: z.string().uuid(),
  reason: z.string().max(500).optional(),
})

export const ApprovalConfigSchema = z.object({
  ttlSeconds: z.number().min(60).max(3600).optional(),
  autoApproveEnabled: z.boolean().optional(),
})

export const PendingApprovalSchema = z.object({
  id: z.string().uuid(),
  orderParams: z.object({
    symbol: z.string(),
    side: z.enum(["buy", "sell"]),
    qty: z.number().optional(),
    notional: z.number().optional(),
    order_type: z.string(),
    limit_price: z.number().optional(),
    estimated_cost: z.number(),
  }),
  source: z.enum(["manual", "llm", "agent"]),
  expiresAt: z.string(),
  createdAt: z.string(),
  timeRemaining: z.number(), // seconds
})

export type ApproveOrderInput = z.infer<typeof ApproveOrderSchema>
export type RejectApprovalInput = z.infer<typeof RejectApprovalSchema>
export type PendingApproval = z.infer<typeof PendingApprovalSchema>
```
