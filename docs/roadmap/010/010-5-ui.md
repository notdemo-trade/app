## Integration with Order Preview

Update order preview to return approval token when policy passes:

```ts
// In order preview handler, after policy check:

if (policyResult.allowed) {
  const approval = await generateApprovalToken({
    userId,
    preview,
    policyResult,
    secret: c.env.APPROVAL_SECRET,
    db: c.get("db"),
    ttlSeconds: 300, // 5 minutes
    source: "manual",
  })

  return c.json({
    data: {
      preview,
      policy: policyResult,
      approval: {
        id: approval.approvalId,
        token: approval.token,
        expiresAt: approval.expiresAt,
      },
    },
  })
}
```

## Query Hooks

```ts
// apps/user-application/src/lib/approval-queries.ts

import { queryOptions, useMutation, useQuery, useQueryClient } from "@tanstack/react-query"
import {
  getPendingApprovals,
  getApprovalHistory,
  approveOrder,
  rejectApproval,
} from "@/core/functions/approvals/direct"

export const pendingApprovalsQueryOptions = () =>
  queryOptions({
    queryKey: ["approvals", "pending"],
    queryFn: getPendingApprovals,
    staleTime: 5_000,
    refetchInterval: 10_000, // Auto-refresh for countdown
  })

export const approvalHistoryQueryOptions = (limit = 50) =>
  queryOptions({
    queryKey: ["approvals", "history", limit],
    queryFn: () => getApprovalHistory(limit),
    staleTime: 30_000,
  })

export const usePendingApprovals = () => useQuery(pendingApprovalsQueryOptions())
export const useApprovalHistory = (limit?: number) => useQuery(approvalHistoryQueryOptions(limit))

export const useApproveOrder = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (approvalId: string) => approveOrder({ approvalId }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["approvals"] })
      queryClient.invalidateQueries({ queryKey: ["orders"] })
      queryClient.invalidateQueries({ queryKey: ["positions"] })
      queryClient.invalidateQueries({ queryKey: ["account"] })
    },
  })
}

export const useRejectApproval = () => {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: { approvalId: string; reason?: string }) => rejectApproval(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["approvals"] })
    },
  })
}
```

## UI Components

### Pending Approvals List

```tsx
// apps/user-application/src/components/approvals/pending-approvals-list.tsx

import { usePendingApprovals, useApproveOrder, useRejectApproval } from "@/lib/approval-queries"
import { Button } from "@/components/ui/button"
import { Badge } from "@/components/ui/badge"
import { Card, CardContent, CardFooter, CardHeader } from "@/components/ui/card"
import { cn } from "@/lib/utils"
import { useState, useEffect } from "react"

export function PendingApprovalsList() {
  const { data, isLoading } = usePendingApprovals()
  const approveMutation = useApproveOrder()
  const rejectMutation = useRejectApproval()

  if (isLoading) {
    return <div className="animate-pulse">Loading approvals...</div>
  }

  if (!data?.length) {
    return (
      <p className="text-sm text-muted-foreground text-center py-8">
        No pending approvals
      </p>
    )
  }

  return (
    <div className="space-y-4">
      {data.map((approval) => (
        <ApprovalCard
          key={approval.id}
          approval={approval}
          onApprove={() => approveMutation.mutate(approval.id)}
          onReject={() => rejectMutation.mutate({ approvalId: approval.id })}
          isApproving={approveMutation.isPending}
          isRejecting={rejectMutation.isPending}
        />
      ))}
    </div>
  )
}

interface ApprovalCardProps {
  approval: {
    id: string
    orderParams: {
      symbol: string
      side: "buy" | "sell"
      qty?: number
      notional?: number
      order_type: string
      estimated_cost: number
    }
    source: string
    timeRemaining: number
    createdAt: string
  }
  onApprove: () => void
  onReject: () => void
  isApproving: boolean
  isRejecting: boolean
}

function ApprovalCard({
  approval,
  onApprove,
  onReject,
  isApproving,
  isRejecting,
}: ApprovalCardProps) {
  const [timeLeft, setTimeLeft] = useState(approval.timeRemaining)

  useEffect(() => {
    const timer = setInterval(() => {
      setTimeLeft((t) => Math.max(0, t - 1))
    }, 1000)
    return () => clearInterval(timer)
  }, [])

  const minutes = Math.floor(timeLeft / 60)
  const seconds = timeLeft % 60
  const isExpiringSoon = timeLeft < 60

  return (
    <Card className={cn(isExpiringSoon && "border-yellow-500")}>
      <CardHeader className="pb-2">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <span className="font-mono text-lg font-bold">
              {approval.orderParams.symbol}
            </span>
            <Badge
              variant={approval.orderParams.side === "buy" ? "default" : "destructive"}
            >
              {approval.orderParams.side.toUpperCase()}
            </Badge>
          </div>
          <Badge variant="outline" className={cn(isExpiringSoon && "text-yellow-600")}>
            {minutes}:{seconds.toString().padStart(2, "0")}
          </Badge>
        </div>
      </CardHeader>

      <CardContent className="pb-2">
        <div className="grid grid-cols-2 gap-2 text-sm">
          <div>
            <span className="text-muted-foreground">Quantity:</span>{" "}
            {approval.orderParams.qty ?? `$${approval.orderParams.notional}`}
          </div>
          <div>
            <span className="text-muted-foreground">Type:</span>{" "}
            <span className="capitalize">{approval.orderParams.order_type}</span>
          </div>
          <div>
            <span className="text-muted-foreground">Est. Cost:</span>{" "}
            ${approval.orderParams.estimated_cost.toFixed(2)}
          </div>
          <div>
            <span className="text-muted-foreground">Source:</span>{" "}
            <span className="capitalize">{approval.source}</span>
          </div>
        </div>
      </CardContent>

      <CardFooter className="gap-2">
        <Button
          onClick={onApprove}
          disabled={isApproving || isRejecting || timeLeft === 0}
          className="flex-1"
        >
          {isApproving ? "Approving..." : "Approve"}
        </Button>
        <Button
          variant="outline"
          onClick={onReject}
          disabled={isApproving || isRejecting}
          className="flex-1"
        >
          {isRejecting ? "Rejecting..." : "Reject"}
        </Button>
      </CardFooter>
    </Card>
  )
}
```

### Approval History

```tsx
// apps/user-application/src/components/approvals/approval-history.tsx

import { useApprovalHistory } from "@/lib/approval-queries"
import { Badge } from "@/components/ui/badge"
import { formatDistanceToNow } from "date-fns"
import { cn } from "@/lib/utils"

export function ApprovalHistory() {
  const { data, isLoading } = useApprovalHistory(20)

  if (isLoading) {
    return <div className="animate-pulse">Loading history...</div>
  }

  if (!data?.length) {
    return <p className="text-sm text-muted-foreground">No approval history</p>
  }

  return (
    <div className="space-y-2">
      {data.map((approval) => (
        <div
          key={approval.id}
          className="flex items-center justify-between p-3 rounded border"
        >
          <div className="flex items-center gap-3">
            <span className="font-mono font-medium">
              {approval.orderParams.symbol}
            </span>
            <Badge
              variant={approval.orderParams.side === "buy" ? "default" : "destructive"}
              className="text-xs"
            >
              {approval.orderParams.side}
            </Badge>
            <StatusBadge status={approval.status} />
          </div>
          <span className="text-xs text-muted-foreground">
            {formatDistanceToNow(new Date(approval.createdAt), { addSuffix: true })}
          </span>
        </div>
      ))}
    </div>
  )
}

function StatusBadge({ status }: { status: string }) {
  return (
    <Badge
      variant="outline"
      className={cn(
        status === "executed" && "border-green-500 text-green-600",
        status === "rejected" && "border-red-500 text-red-600",
        status === "expired" && "border-gray-500 text-gray-600"
      )}
    >
      {status}
    </Badge>
  )
}
```
